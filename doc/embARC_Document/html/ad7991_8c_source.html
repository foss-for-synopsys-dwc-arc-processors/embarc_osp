163 int32_t ad7991_ref_select(AD7991_DEF_PTR obj, int32_t source)
164 {
165 
166  if(source == AD7991_REF_EXTERNAL) {
167  obj->ref_source = source;
168  obj->ref_voltage_mv = AD2_REF_EXTERNAL;
169  obj->reg_config |= AD7991_CONFIG_REF_SEL;
170  /* choose external reference uses VIN4 as REF, so there are only 3 channels available, disable CH3 here*/
171  ad7991_adc_channel(obj, obj->reg_config & ~AD7991_CONFIG_CH3_EN);
172  } else { /*< AD7991_REF_VDD is default setting*/
173  obj->ref_source = AD7991_REF_VDD;
174  obj->ref_voltage_mv = AD2_REF_VDD;
175  obj->reg_config &= ~AD7991_CONFIG_REF_SEL;
176  }
178  dbg_printf(DBG_MORE_INFO, "[%s]%d: after: reg 0x%x , source 0x%x\r\n", __FUNCTION__, __LINE__, obj->reg_config, obj->ref_source);
179  _ad7991_reg_write(obj, obj->reg_config);
180  return E_OK;
181 }

189 int32_t ad7991_adc_channel(AD7991_DEF_PTR obj, int8_t channel)
190 {
191  uint8_t channel_num = 0;
192  if (channel & AD7991_CONFIG_CH0_EN) {
193  channel_num ++;
194  }
195  if (channel & AD7991_CONFIG_CH1_EN) {
196  channel_num ++;
197  }
198  if (channel & AD7991_CONFIG_CH2_EN) {
199  channel_num ++;
200  }
201  if ((obj->ref_source == AD7991_REF_VDD) && (channel & AD7991_CONFIG_CH3_EN)){
202  channel_num ++;
203  }
205  obj->reg_config = (obj->reg_config & 0x0F) | channel;
206  obj->channel_num = channel_num;
207  _ad7991_reg_write(obj, obj->reg_config);
208  return E_OK;
209 }

218 int32_t ad7991_adc_read(AD7991_DEF_PTR obj, float *val)
219 {
220  int32_t ercd = E_OK;
221  uint8_t data[8] = {0};
222  uint8_t index;
223  int16_t voltage;
224  int16_t actual_ref;
225 
226  EMSK_ADC_CHECK_EXP_NORTN(val!=NULL);
227 
228  ercd = _ad7991_reg_read(obj, data, obj->channel_num);
229 
230  if (ercd != obj->channel_num * 2) {
231  ercd = E_OBJ;
232  } else {
233  dbg_printf(DBG_MORE_INFO, "[%s]%d: ercd 0x%x, channel_num%d\r\n", __FUNCTION__, __LINE__, ercd, obj->channel_num);
234  ercd = E_OK;
235  dbg_printf(DBG_MORE_INFO, "[%s]%d: data[0] 0x%x; data[1] 0x%x\r\n", __FUNCTION__, __LINE__, data[0], data[1]);
236  dbg_printf(DBG_MORE_INFO, "[%s]%d: data[0] 0x%x; data[1] 0x%x\r\n", __FUNCTION__, __LINE__, data[2], data[3]);
237  dbg_printf(DBG_MORE_INFO, "[%s]%d: data[0] 0x%x; data[1] 0x%x\r\n", __FUNCTION__, __LINE__, data[4], data[5]);
238  dbg_printf(DBG_MORE_INFO, "[%s]%d: data[0] 0x%x; data[1] 0x%x\r\n", __FUNCTION__, __LINE__, data[6], data[7]);
239  if(obj->ref_source == AD7991_REF_VDD) {
240  actual_ref = AD2_REF_VDD;
241  } else {
242  actual_ref = AD2_REF_EXTERNAL;
243  }
244  /* use index to put value, so if there are more values than channel_num, new value will overwrite old value */
245  voltage = ((int16_t)(data[0] &0x0F) << 8) + ((int16_t)data[1]);
246  index = data[0]>>4;
247  dbg_printf(DBG_LESS_INFO, "[%s]%d: voltage[0] %x\r\n", __FUNCTION__, __LINE__, voltage);
248  val[index] = voltage * actual_ref /4096.0;
249  voltage = ((int16_t)(data[2] &0x0F) << 8) + ((int16_t)data[3]);
250  index = data[2]>>4;
251  dbg_printf(DBG_LESS_INFO, "[%s]%d: voltage[1] %x\r\n", __FUNCTION__, __LINE__, voltage);
252  val[index] = voltage * actual_ref /4096.0;
253  voltage = ((int16_t)(data[4] &0x0F) << 8) + ((int16_t)data[5]);
254  index = data[4]>>4;
255  dbg_printf(DBG_LESS_INFO, "[%s]%d: voltage[2] %x\r\n", __FUNCTION__, __LINE__, voltage);
256  val[index] = voltage * actual_ref /4096.0;
257  voltage = ((int16_t)(data[6] &0x0F) << 8) + ((int16_t)data[7]);
258  index = data[6]>>4;
259  dbg_printf(DBG_LESS_INFO, "[%s]%d: voltage[3] %x\r\n\r\n", __FUNCTION__, __LINE__, voltage);
260  val[index] = voltage * actual_ref /4096.0;
261  }
262 
263 error_exit:
264  return ercd;
265 }
