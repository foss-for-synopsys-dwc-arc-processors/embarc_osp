246 int32_t adt7420_sensor_read(ADT7420_DEF_PTR obj, float *val)
247 {
248  int32_t ercd = E_OK;
249  uint8_t data[2];
250  int16_t temp;
251 
252  ADT7420_CHECK_EXP_NORTN(val!=NULL);
253 
254  ercd = _adt7420_reg_read(obj, ADT7420_REG_TEMP_MSB, data, 2);
255 
256  if (ercd != 2) {
257  ercd = E_OBJ;
258  } else {
259  ercd = E_OK;
260  temp = ((int16_t)data[0] << 8) + ((int16_t)data[1]);
261  dbg_printf(DBG_MORE_INFO, "[%s]%d: data[0] 0x%x; data[1] 0x%x\r\n", __FUNCTION__, __LINE__, data[0], data[1]);
262  dbg_printf(DBG_LESS_INFO, "[%s]%d: temp %x\r\n", __FUNCTION__, __LINE__, temp);
263  /* conversion (val = temp*0.0625) for 13bits[15:3] resolution and (val = temp*0.0078) for 16bits resolution in Degrees Celsius */
264  if (obj->resolution == ADT7420_RESOLUTION_13BIT) {
265  temp >>= 3;
266  *val = temp*0.0625;
267  } else {
268  *val = temp*0.0078;
269  }
270  }
271 
272 error_exit:
273  return ercd;
