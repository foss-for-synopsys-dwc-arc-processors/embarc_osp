68 static void exc_handler_default(void *p_excinf)
69 {
70  uint32_t excpt_cause_reg = 0;
71  uint32_t excpt_ret_reg = 0;
72  uint32_t exc_no = 0;
73 
74  excpt_cause_reg = _arc_aux_read(AUX_ECR);
75  excpt_ret_reg = _arc_aux_read(AUX_ERRET);
76  exc_no = (excpt_cause_reg >> 16) & 0xff;
77 
78  dbg_printf(DBG_LESS_INFO, "default cpu exception handler\r\n");
79  dbg_printf(DBG_LESS_INFO, "exc_no:%d, last sp:0x%08x, ecr:0x%08x, eret:0x%08x\r\n",
80  exc_no, (uint32_t)p_excinf, excpt_cause_reg, excpt_ret_reg);
81 #if SECURESHIELD_VERSION == 2
82  while (1);
83 #else
84  Asm("kflag 1");
85 #endif
86 }

153 void exc_int_init(void)
154 {
155  uint32_t i;
156  uint32_t status;
157  aux_irq_ctrl_t ictrl;
158 
159  ictrl.value = 0;
160 
161 #ifndef ARC_FEATURE_RF16
162  ictrl.bits.save_nr_gpr_pairs = 6; /* r0 to r11 (r12 saved manually) */
163 #else
164  ictrl.bits.save_nr_gpr_pairs = 3; /* r0 to r3, r10, r11 */
165 #endif
166  ictrl.bits.save_blink = 1;
167  ictrl.bits.save_lp_regs = 1; /* LP_COUNT, LP_START, LP_END */
168  ictrl.bits.save_u_to_u = 0; /* user ctxt saved on kernel stack */
169  ictrl.bits.save_idx_regs = 1; /* JLI, LDI, EI */
170 
171  status = arc_lock_save();
172  for (i = NUM_EXC_CPU; i < NUM_EXC_ALL; i++) {
173  /* interrupt level triggered, disabled, priority is the lowest */
174  _arc_aux_write(AUX_IRQ_SELECT, i);
175  _arc_aux_write(AUX_IRQ_ENABLE, 0);
176  _arc_aux_write(AUX_IRQ_TRIGGER, 0);
177 #if defined(ARC_FEATURE_SEC_PRESENT) && (SECURESHIELD_VERSION < 2)
178  _arc_aux_write(AUX_IRQ_PRIORITY, (1 << AUX_IRQ_PRIORITY_BIT_S)|(INT_PRI_MAX - INT_PRI_MIN));
179 #else
180  _arc_aux_write(AUX_IRQ_PRIORITY, INT_PRI_MAX - INT_PRI_MIN);
181 #endif
182  }
183  _arc_aux_write(AUX_IRQ_CTRL, ictrl.value);
184 
185  arc_unlock_restore(status);
186 
188  arc_int_ipm_set((INT_PRI_MAX - INT_PRI_MIN));
189 }

197 int32_t exc_entry_install(const uint32_t excno, EXC_ENTRY entry)
198 {
199  uint32_t status;
200  EXC_ENTRY *table;
201 
202 #if defined(ARC_FEATURE_SEC_PRESENT) && (SECURESHIELD_VERSION < 2)
203  table = (EXC_ENTRY *)_arc_aux_read(AUX_INT_VECT_BASE_S);
204 #else
205  table = (EXC_ENTRY *)_arc_aux_read(AUX_INT_VECT_BASE);
206 #endif
207 
208  if (excno < NUM_EXC_ALL && entry != NULL
209  && table[excno] != entry) {
210  status = cpu_lock_save();
211  /* directly write to mem, as arc gets exception handler from mem not from cache */
212  /* FIXME, here maybe icache is dirty, need to be invalidated */
213  table[excno] = entry;
214 
215  if (_arc_aux_read(AUX_BCR_D_CACHE) > 0x2) {
216  /* dcache is available */
217  dcache_flush_line((uint32_t)&table[excno]);
218  }
219 
220  if (_arc_aux_read(AUX_BCR_D_CACHE) > 0x2) {
221  /* icache is available */
222  icache_invalidate_line((uint32_t)&table[excno]);
223  }
224  cpu_unlock_restore(status);
225  return 0;
226  }
227  return -1;
228 }
