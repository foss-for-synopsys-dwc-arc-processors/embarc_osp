65 static void exc_handler_default(void *p_excinf)
66 {
67  uint32_t excpt_cause_reg = 0;
68  uint32_t excpt_ret_reg = 0;
69  uint32_t exc_no = 0;
70 
71  excpt_cause_reg = _arc_aux_read(AUX_ECR);
72  excpt_ret_reg = _arc_aux_read(AUX_ERRET);
73  exc_no = (excpt_cause_reg >> 16) & 0xff;
74 
75  dbg_printf(DBG_LESS_INFO, "default cpu exception handler\r\n");
76  dbg_printf(DBG_LESS_INFO, "exc_no:%d, last sp:0x%08x, ecr:0x%08x, eret:0x%08x\r\n",
77  exc_no, (uint32_t)p_excinf, excpt_cause_reg, excpt_ret_reg);
78 #if SECURESHIELD_VERSION == 2
79  while (1);
80 #else
81  Asm("kflag 1");
82 #endif
83 }

150 void exc_int_init(void)
151 {
152  uint32_t i;
153  uint32_t status;
154  aux_irq_ctrl_t ictrl;
155 
156  ictrl.value = 0;
157 
158 #ifndef ARC_FEATURE_RF16
159  ictrl.bits.save_nr_gpr_pairs = 6; /* r0 to r11 (r12 saved manually) */
160 #else
161  ictrl.bits.save_nr_gpr_pairs = 3; /* r0 to r3, r10, r11 */
162 #endif
163  ictrl.bits.save_blink = 1;
164  ictrl.bits.save_lp_regs = 1; /* LP_COUNT, LP_START, LP_END */
165  ictrl.bits.save_u_to_u = 0; /* user ctxt saved on kernel stack */
166  ictrl.bits.save_idx_regs = 1; /* JLI, LDI, EI */
167 
168  status = arc_lock_save();
169  for (i = NUM_EXC_CPU; i < NUM_EXC_ALL; i++) {
170  /* interrupt level triggered, disabled, priority is the lowest */
171  _arc_aux_write(AUX_IRQ_SELECT, i);
172  _arc_aux_write(AUX_IRQ_ENABLE, 0);
173  _arc_aux_write(AUX_IRQ_TRIGGER, 0);
174 #if defined(ARC_FEATURE_SEC_PRESENT) && (SECURESHIELD_VERSION < 2)
175  _arc_aux_write(AUX_IRQ_PRIORITY, (1 << AUX_IRQ_PRIORITY_BIT_S)|(INT_PRI_MAX - INT_PRI_MIN));
176 #else
177  _arc_aux_write(AUX_IRQ_PRIORITY, INT_PRI_MAX - INT_PRI_MIN);
178 #endif
179  }
180  _arc_aux_write(AUX_IRQ_CTRL, ictrl.value);
181 
182  arc_unlock_restore(status);
183 
185  arc_int_ipm_set((INT_PRI_MAX - INT_PRI_MIN));
186 }

194 int32_t exc_entry_install(const uint32_t excno, EXC_ENTRY entry)
195 {
196  uint32_t status;
197  EXC_ENTRY *table;
198 
199 #if defined(ARC_FEATURE_SEC_PRESENT) && (SECURESHIELD_VERSION < 2)
200  table = (EXC_ENTRY *)_arc_aux_read(AUX_INT_VECT_BASE_S);
201 #else
202  table = (EXC_ENTRY *)_arc_aux_read(AUX_INT_VECT_BASE);
203 #endif
204 
205  if (excno < NUM_EXC_ALL && entry != NULL
206  && table[excno] != entry) {
207  status = cpu_lock_save();
208  /* directly write to mem, as arc gets exception handler from mem not from cache */
209  /* FIXME, here maybe icache is dirty, need to be invalidated */
210  table[excno] = entry;
211 
212  if (_arc_aux_read(AUX_BCR_D_CACHE) > 0x2) {
213  /* dcache is available */
214  dcache_flush_line((uint32_t)&table[excno]);
215  }
216 
217  if (_arc_aux_read(AUX_BCR_D_CACHE) > 0x2) {
218  /* icache is available */
219  icache_invalidate_line((uint32_t)&table[excno]);
220  }
221  cpu_unlock_restore(status);
222  return 0;
223  }
224  return -1;
225 }
