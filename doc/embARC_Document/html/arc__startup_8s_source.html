51 .weak _f_sdata /* start of small data, defined in link script */
52 .weak init_hardware_hook /* app hardware init hook */
53 .weak init_software_hook /* app software init hook */

58 /* initial vector table */
59  .section .init_vector, "a"
60  .long _arc_reset
61  .section .init_bootstrap, "ax"
62  .global _arc_reset
63  .global _start
64  .align 4
65 _start:
66 _arc_reset:
67 _arc_reset_stage1:
68  kflag STATUS32_RESET_VALUE
69 
70 /* STAGE 1 */
71 
72 /* necessary hardware should be done first to speed up initialization
73  1. system clk
74  2. mem controller must be initialized before any access to external
75  mem.
76  3. others
77 */
78 _arc_cache_init_start:
79  lr r0, [AUX_BCR_D_CACHE]
80  cmp r0, 2
81  /* invalidate dcache */
82  jle _arc_icache_init
83  mov r0, 1
84  sr r0, [AUX_DC_IVDC]
85  sr r0, [AUX_DC_CTRL]
86 _arc_icache_init:
87  lr r0, [AUX_BCR_I_CACHE]
88  cmp r0, 2
89  jle _arc_cache_init_end
90  /* invalidate icache */
91  mov r0, 1
92  sr r0, [AUX_IC_IVIC]
93  nop_s
94  nop_s
95  nop_s
96  sr r0, [AUX_IC_CTRL]
97 
98 _arc_cache_init_end:
99  mov r0, init_hardware_hook
100  cmp r0, 0
101  jlne [r0]
102 
103 /* STAGE 2: init necessary registers */
104 
105 _arc_reset_stage2:
106  mov r0, 0
107 
108 /* interrupt related init */
109  sr r0, [AUX_IRQ_ACT]
110  sr r0, [AUX_IRQ_CTRL]
111  sr r0, [AUX_IRQ_HINT]
112 
113 /* use the new vector table to replace the old one */
114 #if defined(ARC_FEATURE_SEC_PRESENT) && (SECURESHIELD_VERSION < 2)
115  sr exc_entry_table, [AUX_INT_VECT_BASE_S]
116 #else
117  sr exc_entry_table, [AUX_INT_VECT_BASE]
118 #endif
119 
120 /* init stack */
121 #if ARC_FEATURE_RGF_BANKED_REGS >= 16 && ARC_FEATURE_RGF_BANKED_REGS > 1 && ARC_FEATURE_FIRQ == 1
122 #if _STACKSIZE < 512
123 #error "not enough stack size for irq and firq"
124 #endif
125 
126 /* switch to register bank1 */
127  lr r0, [AUX_STATUS32]
128  bic r0, r0, 0x70000
129  or r0, r0, 0x10000
130  kflag r0
131 /* set sp, gp, fp in bank1 */
132  mov sp, _e_stack
133  mov gp, _f_sdata
134  mov fp, 0
135 /* come back to bank0 */
136  lr r0, [AUX_STATUS32]
137  bic r0, r0, 0x70000
138  kflag r0
139  mov sp, _e_stack-256
140 #else
141  mov sp, _e_stack /* init stack pointer */
142 #endif
143  mov gp, _f_sdata /* init small-data base register */
144  mov fp, 0 /* init fp register */
145 
146 _arc_reset_stage3:
147 _s3_copy_text:
148  mov r0, _f_text
149  mov r1, _load_addr_text
150  cmp r0, r1
151 
152 /* if load addr == run addr, no need to copy */
153  jeq _s3_copy_rodata
154  mov r3, _e_text
155 _s3_copy_text_loop:
156  ld.ab r2, [r1, 4]
157  st.ab r2, [r0, 4]
158  cmp r0, r3
159  jlt _s3_copy_text_loop
160 _s3_copy_rodata:
161  mov r0, _f_rodata
162  mov r1, _load_addr_rodata
163  cmp r0, r1
164 
165 /* if load addr == run addr, no need to copy */
166  jeq _s3_copy_data
167  mov r3, _e_rodata
168 _s3_copy_rodata_loop:
169  ld.ab r2, [r1, 4]
170  st.ab r2, [r0, 4]
171  cmp r0, r3
172  jlt _s3_copy_rodata_loop
173 _s3_copy_data:
174  mov r0, _f_data
175  mov r1, _load_addr_data
176  cmp r0, r1
177  jeq _s3_clear_bss
178 
179 /* if load addr == run addr, no need to copy */
180  mov r3, _e_data
181 _s3_copy_data_loop:
182  ld.ab r2, [r1, 4]
183  st.ab r2, [r0, 4]
184  cmp r0, r3
185  jlt _s3_copy_data_loop
186 _s3_clear_bss:
187  mov r0, _f_bss
188  mov r1, _e_bss
189  cmp r0, r1
190  jge _arc_reset_call_main
191  mov r2, 0
192 _s3_clear_bss_loop:
193  st.ab r2, [r0, 4]
194  cmp r0, r1
195  jlt _s3_clear_bss_loop
196 
197 /* STAGE 3: go to main */
198 
199 _arc_reset_call_main:
200  mov r0, init_software_hook
201  cmp r0, 0
202  jlne [r0]
203 /* board level library init */
204 #ifdef LIB_SECURESHIELD
205  /* init cache */
206  jl arc_cache_init
207  jl secureshield_start
208 #if SECURESHIELD_VERSION == 2
209  jl secureshield_except_bit_clear
210 #endif
211 #else
212 /* early init of interrupt and exception */
213  jl exc_int_init
214  /* init cache */
215  jl arc_cache_init
216 #endif
217 
218 #if defined(__MW__)
219  jl _init
220 #elif defined(__GNU__)
221  jl __do_global_ctors_aux
222  jl __do_init_array_aux
223 #endif
224  jl board_main /* board-level main */
225 #if defined(__MW__)
226  jl _fini
227 #elif defined(__GNU__)
228  jl __do_global_dtors_aux
229 #endif
230  .global _exit_loop
231  .global _exit_halt
232  .align 4
233 _exit_halt:
234 _exit_loop:
235  flag AUX_STATUS_MASK_HALT
236  nop
237  nop
238  nop
239  b _exit_loop
