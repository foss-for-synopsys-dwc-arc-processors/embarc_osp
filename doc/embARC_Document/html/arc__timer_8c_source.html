49 int32_t timer_present(const uint32_t no)
50 {
51  uint32_t bcr = _arc_aux_read(AUX_BCR_TIMERS);
52  switch (no) {
53  case TIMER_0:
54  bcr = (bcr >> 8) & 1;
55  break;
56  case TIMER_1:
57  bcr = (bcr >> 9) & 1;
58  break;
59  case TIMER_RTC:
60  bcr = (bcr >> 10) & 1;
61  break;
62  default:
63  bcr = 0;
64  /* illegal argument so return false */
65  break;
66  }
67 
68  return (int)bcr;
69 }
