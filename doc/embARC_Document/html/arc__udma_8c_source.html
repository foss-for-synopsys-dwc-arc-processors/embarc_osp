300 static dma_status_t _dmac_wait_channel(uint32_t channel)
301 {
302  uint32_t status;
303  status = _arc_aux_read(AUX_DMACSTAT1);
304  if (_arc_usually(status & DMACHANNEL(channel))) {
305  if (_arc_rarely(status & (DMACHANNEL(channel) << 16))) {
306  return DMA_ERROR;
307  } else {
308  status = _arc_aux_read(AUX_DMACSTAT0);
309  if (_arc_rarely(status & DMACHANNEL(channel))) {
310  return DMA_BUSY;
311  } else {
312  return DMA_IDLE;
313  }
314  }
315  }
316  return DMA_BUSY;
317 }

332 static uint32_t _dmac_memory_addr_gap(uint32_t dmac_mode)
333 {
334  uint32_t size = (dmac_mode & DMACTRLx_SIZE_MASK) >> DMACTRLx_SIZE_OFS;
335  uint32_t dwinc = (dmac_mode & DMACTRLx_DWINC_MASK) >> DMACTRLx_DWINC_OFS;
336 
337  switch (dwinc) {
338  case DMA_DW1INC2:
339  size <<= 1;
340  break;
341  case DMA_DW1INC4:
342  size <<= 2;
343  break;
344  case DMA_DW2INC2:
345  size &= ~0x1;
346  break;
347  case DMA_DW2INC4:
348  size &= ~0x1;
349  size <<= 1;
350  break;
351  case DMA_DW4INC4:
352  size &= ~0x3;
353  break;
354  case DMA_DW8INC8:
355  size &= ~0x7;
356  break;
357  case DMA_DW1INC1:
358  case DMA_DWINC_CLR:
359  default:
360  break;
361  }
362  return size;
363 }

365 static uint32_t _dmac_aux_addr_gap(uint32_t dmac_mode)
366 {
367  uint32_t size = (dmac_mode & DMACTRLx_SIZE_MASK) >> DMACTRLx_SIZE_OFS;
368  uint32_t dwinc = (dmac_mode & DMACTRLx_DWINC_MASK) >> DMACTRLx_DWINC_OFS;
369 
370  switch (dwinc) {
371  case DMA_DW2INC2:
372  case DMA_DW2INC4:
373  size >>= 1;
374  break;
375  case DMA_DW4INC4:
376  case DMA_DW8INC8:
377  size >>= 2;
378  break;
379  case DMA_DW1INC1:
380  case DMA_DW1INC2:
381  case DMA_DW1INC4:
382  case DMA_DWINC_CLR:
383  default:
384  break;
385  }
386  return size;
387 }

391 #ifndef USE_DMA_CALC_ADDR /* converted to 1 function called _dmac_calc_addr */
392 static uint32_t _dmac_calc_dst_endaddr(uint32_t dst_addr, uint32_t dmac_mode)
393 {
394  /*
395  * Destination address is not incremented
396  */
397  if (_arc_rarely((dmac_mode & DMACTRLx_AM(2)) == 0)) {
398  return dst_addr;
399  }
400 
401  if (_arc_rarely(DMACTRLx_DTT(0x1) & dmac_mode)) { /* Destination is Auxiliary */
402  dst_addr = dst_addr + _dmac_aux_addr_gap(dmac_mode);
403  } else { /* Destination is Memory */
404  dst_addr = dst_addr + _dmac_memory_addr_gap(dmac_mode);
405  }
406  return dst_addr;
407 }

409 static uint32_t _dmac_calc_src_endaddr(uint32_t src_addr, uint32_t dmac_mode)
410 {
411  /*
412  * Source address is not incremented
413  */
414  if (_arc_rarely((dmac_mode & DMACTRLx_AM(1)) == 0)) {
415  return src_addr;
416  }
417 
418  if (_arc_rarely(DMACTRLx_DTT(0x2) & dmac_mode)) { /* Source is Auxiliary */
419  src_addr = src_addr + _dmac_aux_addr_gap(dmac_mode);
420  } else { /* Source is Memory */
421  src_addr = src_addr + _dmac_memory_addr_gap(dmac_mode);
422  }
423 
424  return src_addr;
425 }

427 static void _dmac_calc_addr(uint32_t *src_addr, uint32_t *dst_addr, uint32_t mode)
428 {
429  if (_arc_usually(mode & DMACTRLx_AM(1))) { /* Source address incremented */
430  if (_arc_rarely(DMACTRLx_DTT(0x2) & mode)) { /* Source is Auxiliary */
431  *src_addr += _dmac_aux_addr_gap(mode);
432  } else { /* Source is Memory */
433  *src_addr += _dmac_memory_addr_gap(mode);
434  }
435  }
436  if (_arc_usually(mode & DMACTRLx_AM(2))) { /* Destination address incremented */
437  if (_arc_rarely(DMACTRLx_DTT(0x1) & mode)) { /* Destination is Auxiliary */
438  *dst_addr += _dmac_aux_addr_gap(mode);
439  } else { /* Destination is Memory */
440  *dst_addr += _dmac_memory_addr_gap(mode);
441  }
442  }
443 }

446 static void _dmac_set_desc(dma_desc_t * desc, void *source, void *dest, uint32_t size,
447  uint32_t dmac_mode)
448 {
449  uint32_t src_addr, dst_addr;
450 
451  size = DMA_CTRL_BLKSZ(size);
452  dmac_mode &= ~DMACTRLx_SIZE_MASK;
453  dmac_mode |= DMACTRLx_SIZE(size);
454 
455 #ifndef USE_DMA_CALC_ADDR
456  src_addr = _dmac_calc_src_endaddr((uint32_t)source, dmac_mode);
457  dst_addr = _dmac_calc_dst_endaddr((uint32_t)dest, dmac_mode);
458 #else /* use dma_calc_addr */
459  src_addr = (uint32_t)source;
460  dst_addr = (uint32_t)dest;
461  _dmac_calc_addr(&src_addr, &dst_addr, dmac_mode);
462 #endif
463  desc->DMACTRLx = dmac_mode;
464  desc->DMASARx = src_addr;
465  desc->DMADARx = dst_addr;
466  desc->DMALLPx = (uint32_t) NULL;
467 }

469 static void _dmac_fill_descriptor(uint32_t channel, dma_desc_t * desc)
470 {
471  if (_arc_usually(DMA_CHECK_REGISTER(channel))) {
472  uint32_t channel_shift = 3 * channel;
473  _arc_aux_write(AUX_DMACTRL0 + channel_shift, desc->DMACTRLx);
474  _arc_aux_write(AUX_DMASAR0 + channel_shift, desc->DMASARx);
475  _arc_aux_write(AUX_DMADAR0 + channel_shift, desc->DMADARx);
476 #if CORE_DMAC_INTERNAL_VERSION > 1
477  uint32_t *dma_llps = (uint32_t *) _arc_aux_read(AUX_DMACBASE);
478  dma_llps[channel] = desc->DMALLPx;
479  _MEMORY_FENCE();
480  //_arc_write_uncached_32((void *)(&dma_llps[channel]), desc->DMALLPx);
481  _DCACHE_FLUSH_MLINES((void *)(&dma_llps[channel]), sizeof(uint32_t));
482 #endif
483  } else {
484 #if CORE_DMAC_INTERNAL_VERSION > 1
485  uint32_t *dma_ptrs = (uint32_t *) _arc_aux_read(AUX_DMACBASE);
486  dma_ptrs[channel] = (uint32_t)desc;
487  _MEMORY_FENCE();
488  //_arc_write_uncached_32((void *)(&dma_ptrs[channel]), (uint32_t)desc);
489  _DCACHE_FLUSH_MLINES((void *)(&dma_ptrs[channel]), sizeof(uint32_t));
490 #else
491  dma_desc_t *dmac_desc = (dma_desc_t *) _arc_aux_read(AUX_DMACBASE);
492  dmac_desc = (dma_desc_t *) (&dmac_desc[channel]);
493  /* more efficient to let compiler do this copy */
494  *dmac_desc = *desc;
495  _MEMORY_FENCE();
496  _DCACHE_FLUSH_MLINES((void *)dmac_desc, sizeof(dma_desc_t));

697 static void dmac_process_desc(dma_desc_t * desc, uint32_t int_enable)
698 {
699  do {
700  if (_arc_rarely(desc->DMALLPx)) {
701  /*
702  * Force it to be auto-linked transfer when it is not
703  * DMA_MANUAL_LINKED_TRANSFER
704  */
705  if (_arc_usually(((desc->DMACTRLx & DMACTRLx_OP_MASK) >> DMACTRLx_OP_OFS) !=
706  DMA_MANUAL_LINKED_TRANSFER)) {
707  desc->DMACTRLx &= ~DMACTRLx_OP_MASK;
708  desc->DMACTRLx |= DMACTRLx_OP(DMA_AUTO_LINKED_TRANSFER);
709  }
710  } else {
711  desc->DMACTRLx &= ~DMACTRLx_OP_MASK;
712  desc->DMACTRLx |= DMACTRLx_OP(DMA_SINGLE_TRANSFER);
713  /* Only forcely change the interrupt bit of the last dma descriptor */
714  desc->DMACTRLx &= ~DMACTRLx_INT_MASK;
715  desc->DMACTRLx |= DMACTRLx_INT(int_enable);
716  }
717  /*
718  * TODO: Make sure all linked DMA channel descriptors
719  * stored into memory, not just in data cache
720  */
721  _MEMORY_FENCE();
722  _DCACHE_FLUSH_MLINES((void *)desc, sizeof(dma_desc_t));
723  desc = (dma_desc_t *) desc->DMALLPx;
724  } while (desc != 0);
725 }
