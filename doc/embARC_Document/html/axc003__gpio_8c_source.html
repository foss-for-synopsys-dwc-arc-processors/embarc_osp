93 void leda_write(uint32_t led_val, uint32_t mask)
94 {
95  AXC003_GPIO_CHECK_EXP_NORTN(leda_port != NULL);
96 
97  led_val &= mask;
98  led_val = (led_val & 0x3) | ((led_val & 0xC) << 0x3) | ((led_val & 0x30) << 0x6) |
99  ((led_val & 0xC0) << 9);
100 
101  mask = (mask & 0x3) | ((mask & 0xC) << 0x3) | ((mask & 0x30) << 0x6) |
102  ((mask & 0xC0) << 9);
103 
104  leda_port->gpio_write(led_val, mask);
105 
106 error_exit:
107  return;
108 }

127 uint32_t button_read(uint32_t mask)
128 {
129  uint32_t value;
130 
131  AXC003_GPIO_CHECK_EXP_NORTN(button_port != NULL);
132 
133  mask = (mask << AXC003_BUTTON_OFFSET) & AXC003_BUTTON_MASK;
134  button_port->gpio_read(&value, mask);
135  value = (~value) & mask;
136 
137  return (value >> AXC003_BUTTON_OFFSET);
138 
139 error_exit:
140  return 0;
141 }

143 void button_install_isr(uint32_t offset, void (*isr_ptr)(void *ptr))
144 {
145 
146 /* note: CPU card's creg ctrl's gpio int source to INTNO_ICTL_MB,
147 default is A12 which is int request from main board, int request from SW2504-SW2507 can not
148 go to INTNO_ICTL_MB */
149  DEV_GPIO_BIT_ISR bit_isr;
150  DEV_GPIO_INT_CFG int_cfg;
151  uint32_t mask;
152 
153  if (offset < AXC003_BUTTON_SW2504_OFFSET || offset > AXC003_BUTTON_SW2507_OFFSET) {
154  return;
155  }
156  AXC003_GPIO_CHECK_EXP_NORTN(button_port != NULL);
157 
158  mask = (0x1 << offset);
159 
160  int_cfg.int_bit_mask = mask;
161  int_cfg.int_bit_type = GPIO_INT_BITS_EDGE_TRIG(mask);
162  int_cfg.int_bit_polarity = GPIO_INT_BITS_POL_FALL_EDGE(mask);
163  int_cfg.int_bit_debounce = GPIO_INT_BITS_ENA_DEBOUNCE(mask);
164  button_port->gpio_control(GPIO_CMD_SET_BIT_INT_CFG, (void *)(&int_cfg));
165  bit_isr.int_bit_ofs = offset;
166  bit_isr.int_bit_handler = isr_ptr;
167  button_port->gpio_control(GPIO_CMD_SET_BIT_ISR, (void *)(&bit_isr));
168  button_port->gpio_control(GPIO_CMD_ENA_BIT_INT, (void *)mask);
169 
170 error_exit:
171  return;
172 }

187 void seven_seg_write(uint32_t val)
188 {
189  uint32_t digit;
190 
191  AXC003_GPIO_CHECK_EXP_NORTN(seven_seg_port != NULL);
192 
193  if (val > 99) {
194  return; // only can show val < 99, two 7-seg display.
195  }
196 
197  digit = num_to_7seg_table[val / 10] | (num_to_7seg_table[val % 10] << 8);
198 
199  seven_seg_port->gpio_write((digit << AXC003_7SEG_OFFSET), AXC003_7SEG_MASK);
200 
201 error_exit:
202  return;
203 }
