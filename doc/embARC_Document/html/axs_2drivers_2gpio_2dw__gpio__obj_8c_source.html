109 static void cpu_porta_install(void)
110 {
111  uint32_t i;
112  DEV_GPIO_PTR port_ptr = &cpu_port_a;
113  DEV_GPIO_INFO_PTR info_ptr = &(cpu_port_a.gpio_info);
114  DW_GPIO_PORT_PTR dw_port_ptr = &(cpu_dw_gpio_port_a);
115 
116  info_ptr->gpio_ctrl = (void *)dw_port_ptr;
117  info_ptr->opn_cnt = 0;
118  info_ptr->method = 0;
119  info_ptr->direction = 0;
120  info_ptr->extra = 0;
121 
122  dw_port_ptr->no = DW_GPIO_PORT_A;
123  dw_port_ptr->regs = (DW_GPIO_REG_PTR)(REGBASE_GPIO_CPU);
124  dw_port_ptr->valid_bit_mask = AXS_CPU_GPIO_A_VALID_MASK;
125  dw_port_ptr->intno = INTNO_ICTL_MB;
126  dw_port_ptr->int_handler = cpu_porta_isr;
127 
128  for (i=0; i < cpu_dw_gpio_bit_isr_a.int_bit_max_cnt; i++) {
129  cpu_dw_gpio_bit_isr_a.int_bit_handler_ptr[i] = NULL;
130  }
131  dw_port_ptr->gpio_bit_isr = &cpu_dw_gpio_bit_isr_a;
132 
133  port_ptr->gpio_open = cpu_porta_open;
134  port_ptr->gpio_close = cpu_porta_close;
135  port_ptr->gpio_control = cpu_porta_control;
136  port_ptr->gpio_write = cpu_porta_write;
137  port_ptr->gpio_read = cpu_porta_read;
138 }
