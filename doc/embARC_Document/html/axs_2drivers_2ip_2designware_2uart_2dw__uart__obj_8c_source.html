269 DEV_UART_PTR uart_get_dev(int32_t uart_id)
270 {
271  static uint32_t install_flag = 0;
272 
273  /* intall device objects */
274  if (install_flag == 0) {
275  install_flag = 1;
276  dw_uart_all_install();
277  }
278 
279  switch (uart_id) {
280 #if (USE_DW_UART_0)
281  case DW_UART_0_ID:
282  return &dw_uart_0;
283  break;
284 #endif
285 #if (USE_DW_UART_1)
286  case DW_UART_1_ID:
287  return &dw_uart_1;
288  break;
289 #endif
290 #if (USE_DW_UART_2)
291  case DW_UART_2_ID:
292  return &dw_uart_2;
293  break;
294 #endif
295  default:
296  break;
297  }
298  return NULL;
299 }
