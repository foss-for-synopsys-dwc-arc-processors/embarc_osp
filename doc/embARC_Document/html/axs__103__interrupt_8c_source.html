249 int32_t int_disable(const uint32_t intno)
250 {
251  uint32_t ictl = intno >> 8;
252 
253  uint32_t sub_intno = intno & 0xff;
254 
255  if (ictl == 0 && sub_intno >= NUM_EXC_CPU && sub_intno < NUM_EXC_ALL) {
256  /* arc interrupt controller */
257  arc_int_disable(sub_intno);
258  } else if (ictl == INTNO_ICTL_MB && sub_intno <= ICTL_INTNO_NUM ) { /* main board ictl */
259  dw_ictl_int_disable(dw_ictl_ptr, sub_intno);
260  } else if (ictl == INTNO_ICTL_CPU && sub_intno <= CPU_ICTL_INTNO_NUM) { /* cpu card ictl */
261  ictl_gpio_int_disable(ictl_gpio_ptr, sub_intno);
262  } else {
263  return -1;
264  }
265 
266  return 0;
