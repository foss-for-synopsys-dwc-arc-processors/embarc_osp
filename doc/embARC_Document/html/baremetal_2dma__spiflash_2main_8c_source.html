150  EMBARC_PRINTF("SPI Flash Erase Finished in %d cycles\r\n", cost_cyc);

153  cost_cyc = perf_end();
154  for (i = 0; i < BUFFER_SIZE; i++) {
155  if (test_buffer_rd[i] != 0xFF) {
156  EMBARC_PRINTF("SPI Erase failed:%d\r\n", i);
157  break;
158  }
159  }
160  EMBARC_PRINTF("SPI Flash Read Finished in %d cycles\r\n", cost_cyc);
161 
162  perf_start();
163  spiflash_write(start_addr, BUFFER_SIZE, test_buffer);
164  cost_cyc = perf_end();
165  EMBARC_PRINTF("SPI Flash Write Finished in %d cycles\r\n", cost_cyc);
166  perf_start();
167  spiflash_read(start_addr, BUFFER_SIZE, test_buffer_rd);
168  cost_cyc = perf_end();
169  EMBARC_PRINTF("SPI Flash Read Finished in %d cycles\r\n", cost_cyc);
170  /* verify read and write */
171  if ((wr_err_cnt = mem_cmp(test_buffer_rd, test_buffer, BUFFER_SIZE)) > 0) {
172  EMBARC_PRINTF("SPI FLASH Write Read Verification Error @%uHz, error count:%u\r\n", \
173  freq, wr_err_cnt);

175  EMBARC_PRINTF("SPI FLASH Write Read Verification successfully\r\n");
176  }

178 }

241 static void perf_start(void)
242 {

249 static unsigned int perf_end(void)
250 {
251  unsigned int end = 0;
252 
253  if (timer_current(perf_id, (void *)(&end)) < 0) {
254  return 0;
255  }
256 
257  if (start < end) {
258  return (end - start);
259  } else {
260  return (0xFFFFFFFF - start + end);
261  }
262 }
