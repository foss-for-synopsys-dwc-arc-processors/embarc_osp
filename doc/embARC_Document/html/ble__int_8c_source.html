114 static void ble_rx_state_second(mode_info *ble_mode_info_ptr, uint8_t *ble_rx_handle_ptr, uint8_t data)
115 {
116  if (data == BLE_RX_MUSIC_MODE) {
117  ble_rx_handle_ptr[BLE_STATE_BYTE] = INITIAL_STATE;
118  ble_mode_info_ptr->mode = LIGHT_MODE_MUSIC;
119  xQueueOverwrite(mode_info_queue, ble_mode_info_ptr);
120  xTaskResumeFromISR( pattern_change_task_handle );
121  } else if (data == BLE_RX_COMFOR_MODE) {
122  ble_rx_handle_ptr[BLE_STATE_BYTE] = INITIAL_STATE;
123  ble_mode_info_ptr->mode = LIGHT_MODE_COMFOR;

135 static void ble_rx_state_third(mode_info *ble_mode_info_ptr, uint8_t *ble_rx_handle_ptr, uint8_t data)
136 {
137  switch (ble_rx_handle_ptr[BLE_MODE_TAG_BYTE]) {
138  case BLE_RX_WEATHER_MODE:
139  if ( data == '0') {
140  ble_rx_handle_ptr[BLE_STATE_BYTE] = INITIAL_STATE;
141  ble_mode_info_ptr->mode = LIGHT_MODE_WEATHER;
142  xQueueOverwrite(mode_info_queue, ble_mode_info_ptr);
143  xTaskResumeFromISR( pattern_change_task_handle );
144  } else {
145  ble_rx_handle_ptr[BLE_STATE_BYTE] = FOURTH_STATE;
146  ble_mode_info_ptr->weather = data - '0';
147  }
148  break;
149 
150  case BLE_RX_RUNNING_MODE:
151  case BLE_RX_ALARM_MODE:
152  case BLE_RX_RIDING_MODE:
153  case BLE_RX_LIGHTING_MODE:
154  ble_rx_handle_ptr[BLE_STATE_BYTE] = FOURTH_STATE;
155  ble_rx_handle_ptr[BLE_MODE_AUX_TAG_BYTE] = data;
156  break;
157 
158  case BLE_RX_CLOCK_MODE:
159  case BLE_RX_TIME_MODE:
160  ble_rx_handle_ptr[ ble_rx_handle_ptr[BLE_DATA_CNT_BYTE]++ ] = data - '0';
161 
162  if (ble_rx_handle_ptr[BLE_DATA_CNT_BYTE] == BLE_MODE_TIME_RX_LEN) {
163  ble_rx_handle_ptr[BLE_STATE_BYTE] = INITIAL_STATE;
164  ble_rx_handle_ptr[BLE_DATA_CNT_BYTE] = 0;
165 
166  if (ble_rx_handle_ptr[BLE_MODE_TAG_BYTE] == BLE_RX_CLOCK_MODE) {
167  ble_mode_info_ptr->mode = LIGHT_MODE_CLOCK;
168  } else {
169  ble_mode_info_ptr->mode = LIGHT_MODE_TIMING_LAND;
170  }
171 
172  ble_mode_info_ptr->time = ble_rx_handle_ptr[0] * 600 + ble_rx_handle_ptr[1] * 60 + \
173  ble_rx_handle_ptr[2] * 10 + ble_rx_handle_ptr[3];
174  xQueueOverwrite(mode_info_queue, ble_mode_info_ptr);
175  xTaskResumeFromISR( pattern_change_task_handle );
176  }
177  break;
178 
179  case BLE_RX_FANSE_MODE:
180  ble_rx_handle_ptr[BLE_STATE_BYTE] = FOURTH_STATE;
181  ble_rx_handle_ptr[BLE_DATA_LEN_BYTE] = data;
182  break;
183 
184  default:
185  ble_rx_handle_ptr[BLE_STATE_BYTE] = INITIAL_STATE;
186  EMBARC_PRINTF("Wrong input %d in state 3.\r\n", ble_rx_handle_ptr[BLE_MODE_TAG_BYTE]);
187  break;
188  }
189 }

195 static void ble_rx_state_forth(mode_info *ble_mode_info_ptr, uint8_t *ble_rx_handle_ptr, uint8_t data)
196 {
197  switch (ble_rx_handle_ptr[BLE_MODE_TAG_BYTE]) {
198  case BLE_RX_WEATHER_MODE:
199  ble_rx_handle_ptr[ ble_rx_handle_ptr[BLE_DATA_CNT_BYTE]++ ] = data - '0';
200 
201  if (ble_rx_handle_ptr[BLE_DATA_CNT_BYTE] == BLE_MODE_WEATHER_RX_LEN) {
202  ble_rx_handle_ptr[BLE_STATE_BYTE] = INITIAL_STATE;
203  ble_rx_handle_ptr[BLE_DATA_CNT_BYTE] = 0;
204 
205  ble_mode_info_ptr->temperture = ble_rx_handle_ptr[0] * 10 + ble_rx_handle_ptr[1];
206  ble_mode_info_ptr->humidity = ble_rx_handle_ptr[2] * 10 + ble_rx_handle_ptr[3];
207  xQueueOverwrite(mode_info_queue, ble_mode_info_ptr);
208  }
209  break;
210 
211  case BLE_RX_RUNNING_MODE:
212  case BLE_RX_ALARM_MODE:
213  case BLE_RX_RIDING_MODE:
214  ble_rx_handle_ptr[BLE_STATE_BYTE] = INITIAL_STATE;
215  if (ble_rx_handle_ptr[BLE_MODE_AUX_TAG_BYTE] == BLE_TAG_BRIGHT) {
216  ble_mode_info_ptr->bright = data;
217  }
218  else if (ble_rx_handle_ptr[BLE_MODE_AUX_TAG_BYTE] == BLE_TAG_FREQUENCE) {
219  switch (data) {
220  case BLE_FREQUENCY_SLOW :
221  ble_mode_info_ptr->frequence = FREQUENCY_SLOW;
222  break;
223  case BLE_FREQUENCY_NORMAL :
224  ble_mode_info_ptr->frequence = FREQUENCY_NORMAL;
225  break;
226  case BLE_FREQUENCY_FAST :
227  ble_mode_info_ptr->frequence = FREQUENCY_FAST;
228  break;
229  case BLE_FREQUENCY_VERY_FAST :
230  ble_mode_info_ptr->frequence = FREQUENCY_VERY_FAST;
231  break;
232  default :
233  ble_mode_info_ptr->frequence = 0;
234  break;
235  }
236  }
237 
238  if (ble_rx_handle_ptr[BLE_MODE_TAG_BYTE] == BLE_RX_RUNNING_MODE) {
239  if (ble_mode_info_ptr->mode != LIGHT_MODE_RUNNING) {
240  ble_mode_info_ptr->frequence = 0;
241  }
242  ble_mode_info_ptr->mode = LIGHT_MODE_RUNNING;
243  } else if (ble_rx_handle_ptr[BLE_MODE_TAG_BYTE] == BLE_RX_ALARM_MODE) {
244  if (ble_mode_info_ptr->mode != LIGHT_MODE_ALARM) {
245  ble_mode_info_ptr->frequence = 0;
246  }
247  ble_mode_info_ptr->mode = LIGHT_MODE_ALARM;
248  } else if (ble_rx_handle_ptr[BLE_MODE_TAG_BYTE] == BLE_RX_RIDING_MODE) {
249  if (ble_mode_info_ptr->mode != LIGHT_MODE_RIDING) {
250  ble_mode_info_ptr->frequence = 0;
251  }
252  ble_mode_info_ptr->mode = LIGHT_MODE_RIDING;
253  }
254 
255  xQueueOverwrite(mode_info_queue, ble_mode_info_ptr);
256  xTaskResumeFromISR( pattern_change_task_handle );
257  break;
258 
259  case BLE_RX_LIGHTING_MODE:
260  ble_rx_handle_ptr[BLE_STATE_BYTE] = INITIAL_STATE;
261 
262  if (ble_rx_handle_ptr[BLE_MODE_AUX_TAG_BYTE] == BLE_TAG_BRIGHT) {
263  ble_mode_info_ptr->bright = data;
264  } else if (ble_rx_handle_ptr[BLE_MODE_AUX_TAG_BYTE] == BLE_TAG_COOL_OR_WARM) {
265  if (data == BLE_TAG_LIGHT_WARM) {
266  ble_rx_handle_ptr[BLE_LIGHT_TAG_BYTE] = BLE_TAG_LIGHT_WARM;
267  } else {
268  ble_rx_handle_ptr[BLE_LIGHT_TAG_BYTE] = BLE_TAG_LIGHT_COOL;
269  }
270  } else if (ble_rx_handle_ptr[BLE_MODE_AUX_TAG_BYTE] == BLE_TAG_ANIMATION) {
271  if (data == BLE_TAG_LIGHT_ANIM) {
272  ble_rx_handle_ptr[BLE_COLOR_TAG_BYTE] = BLE_TAG_LIGHT_ANIM;
273  } else {
274  ble_rx_handle_ptr[BLE_COLOR_TAG_BYTE] = BLE_TAG_LIGHT_NOT_ANIM;
275  }
276  }
277 
278  if (ble_rx_handle_ptr[BLE_COLOR_TAG_BYTE] == BLE_TAG_LIGHT_ANIM) {
279  ble_mode_info_ptr->mode = LIGHT_MODE_ANIMATION;
280  } else {
281  if (ble_rx_handle_ptr[BLE_LIGHT_TAG_BYTE] == BLE_TAG_LIGHT_WARM) {
282  ble_mode_info_ptr->mode = LIGHT_MODE_WARM_LIGHTING;
283  } else {
284  ble_mode_info_ptr->mode = LIGHT_MODE_CO_LIGHTING;
285  }
286  }
287 
288  xQueueOverwrite(mode_info_queue, ble_mode_info_ptr);
289  xTaskResumeFromISR(pattern_change_task_handle);
290  break;
291 
292  case BLE_RX_FANSE_MODE:
293  ble_mode_info_ptr->fance_txt[ ble_rx_handle_ptr[BLE_DATA_CNT_BYTE]++ ] = data;
294 
295  if (ble_rx_handle_ptr[BLE_DATA_CNT_BYTE] == ble_rx_handle_ptr[BLE_DATA_LEN_BYTE]) {
296  ble_rx_handle_ptr[BLE_STATE_BYTE] = INITIAL_STATE;
297  ble_mode_info_ptr->fance_txt[ ble_rx_handle_ptr[BLE_DATA_CNT_BYTE] ] = '\0';
298  ble_rx_handle_ptr[BLE_DATA_CNT_BYTE] = 0;
299 
300  ble_mode_info_ptr->mode = LIGHT_MODE_SHAKING;
301  xQueueOverwrite(mode_info_queue, ble_mode_info_ptr);
302  xTaskResumeFromISR(pattern_change_task_handle);
303  }
304  break;
305 
306  default:
307  ble_rx_handle_ptr[BLE_STATE_BYTE] = INITIAL_STATE;
308  EMBARC_PRINTF("Wrong mode input %d in state 4.\r\n", ble_rx_handle_ptr[BLE_MODE_TAG_BYTE]);
309  break;
310  }

318 Inline void ble_data_recesive(uint8_t data)
319 {
320  static mode_info ble_mode_info = INIT_MODE_INFO;
321  static uint8_t ble_rx_handle[BLE_RX_HANDLE_BUF_MAX];
322 
323  if ((ble_rx_handle[BLE_STATE_BYTE] == INITIAL_STATE) && (data == BLE_RX_HEAD_FIRST)) {
324  ble_rx_handle[BLE_STATE_BYTE] = FIRST_STATE;
325  } else if (ble_rx_handle[BLE_STATE_BYTE] == FIRST_STATE && data == BLE_RX_HEAD_SECOND) {
326  ble_rx_handle[BLE_STATE_BYTE] = SECOND_STATE;
327  } else if ((ble_rx_handle[BLE_STATE_BYTE] == SECOND_STATE) && \
328  (data <= (BLE_RX_MIN_MODE + MAX_MODE)) && (data >= BLE_RX_MIN_MODE)) {
329  ble_rx_state_second(&ble_mode_info, ble_rx_handle, data);
330  } else if (ble_rx_handle[BLE_STATE_BYTE] == THIRD_STATE) {
331  ble_rx_state_third(&ble_mode_info, ble_rx_handle, data);
332  } else if (ble_rx_handle[BLE_STATE_BYTE] == FOURTH_STATE) {
333  ble_rx_state_forth(&ble_mode_info, ble_rx_handle, data);
334  } else {
335  ble_rx_handle[BLE_STATE_BYTE] = INITIAL_STATE;
336  }
337 }
