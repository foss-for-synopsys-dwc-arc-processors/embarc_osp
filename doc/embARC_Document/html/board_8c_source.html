107 static void task_wifi(void *par)
108 {
109  WNIC_AUTH_KEY auth_key;
110  int flag=0;
111 
112  pmwifi_wnic = wnic_get_dev(BOARD_PMWIFI_0_ID);
113 
114 #if WF_HOTSPOT_IS_OPEN
115  auth_key.key = NULL;
116  auth_key.key_len = 0;
117  auth_key.key_idx = 0;
118 #else
119  auth_key.key = (const uint8_t *)WF_HOTSPOT_PASSWD;
120  auth_key.key_len = sizeof(WF_HOTSPOT_PASSWD);
121  auth_key.key_idx = 0;
122 #endif
123  lwip_pmwifi_init();
124  EMBARC_PRINTF("\r\nNow trying to connect to WIFI hotspot, please wait about 30s!\r\n");
125 
126  while (1) {
127  pmwifi_wnic->period_process(par);
128 #if WF_HOTSPOT_IS_OPEN
129  pmwifi_wnic->wnic_connect(AUTH_SECURITY_OPEN, (const uint8_t *)WF_HOTSPOT_NAME, &auth_key);
130 #else
131  pmwifi_wnic->wnic_connect(AUTH_SECURITY_WPA_AUTO_WITH_PASS_PHRASE, (const uint8_t *)WF_HOTSPOT_NAME, &auth_key);
132 #endif
133  if ((flag == 0) && lwip_pmwifi_isup()) {
134  flag = 1;
135  EMBARC_PRINTF("WiFi connected \r\n");
136 #ifndef MID_NTSHELL /* resume main task when ntshell task is not defined */
137  if (task_handle_main) vTaskResume(task_handle_main);
138 #else
139  EMBARC_PRINTF("Please run NT-Shell command(main) to start your application.\r\n");
140  EMBARC_PRINTF("main command may required some arguments, please refer to example's document.\r\n");
141 #endif
142  /* consider to generate a event to notify network is ready */
143  }
144  vTaskDelay(TASK_WIFI_PERIOD);
145  }
146 }
147 #endif

149 static void task_main(void *par)
150 {
151  int ercd;
152 #if defined(OS_FREERTOS) && defined(MID_LWIP) && !defined(MID_NTSHELL)
153  EMBARC_PRINTF("Enter to main function....\r\n");
154  EMBARC_PRINTF("Wait until WiFi connected...\r\n");
155  vTaskSuspend(NULL);
156 #endif
157 
158  if ((par == NULL) || (((int)par) & 0x3)) {
159  /* null or aligned not to 4 bytes */
160  ercd = _arc_goto_main(0, NULL);
161  } else {
162  MAIN_ARGS *main_arg = (MAIN_ARGS *)par;
163  ercd = _arc_goto_main(main_arg->argc, main_arg->argv);
164  }
165 
166 #if defined(OS_FREERTOS)
167  EMBARC_PRINTF("Exit from main function, error code:%d....\r\n", ercd);
168  while (1) {
169  vTaskSuspend(NULL);
170  }
171 #else
172  while (1);
173 #endif
174 }

199 /* NTSHELL related initialization */
200 /* For OS situation, ntshell task will be created; for baremetal ntshell_task will be executed */
201 #ifdef MID_NTSHELL
202  NTSHELL_IO *nt_io;
203  nt_io = get_ntshell_io(BOARD_ONBOARD_NTSHELL_ID);
204 
205 #ifdef OS_FREERTOS
206  xTaskCreate((TaskFunction_t)ntshell_task, "ntshell-console", TASK_STACK_SIZE_NTSHELL,
207  (void *)nt_io, TASK_PRI_NTSHELL, &task_handle_ntshell);
208 #else
209  cpu_unlock(); /* unlock cpu to let interrupt work */
211  ntshell_task((void *)nt_io);
212 #endif
213 
214 #else /* No ntshell */
215 #ifdef OS_FREERTOS
216  xTaskCreate((TaskFunction_t)task_main, "main", TASK_STACK_SIZE_MAIN,
217  (void *)(&s_main_args), TASK_PRI_MAIN, &task_handle_main);
218 #else /* No os and ntshell */
219  cpu_unlock(); /* unlock cpu to let interrupt work */
220 #endif
221 
222 #endif /* MID_NTSHELL */
223 
224 #if defined(OS_FREERTOS) && defined (MID_LWIP)
225  xTaskCreate((TaskFunction_t)task_wifi, "wifi-conn", TASK_STACK_SIZE_WIFI,
226  (void *)1, TASK_PRI_WIFI, &task_handle_wifi);
227 #endif
228 
229 #ifdef OS_FREERTOS
230  vTaskStartScheduler();
231 #endif
232 
233  task_main((void *)(&s_main_args));
