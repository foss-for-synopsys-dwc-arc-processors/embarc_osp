226 int32_t dw_gpio_open(DEV_GPIO *gpio_obj, uint32_t dir)
227 {
228  int32_t ercd = E_OK;
229  DEV_GPIO_INFO_PTR port_info_ptr = &(gpio_obj->gpio_info);
230 
231  /* START ERROR CHECK */
232  VALID_CHK_GPIO_INFO_OBJECT(port_info_ptr);
233  /* END OF ERROR CHECK */
234 
235  DW_GPIO_PORT_PTR port = (DW_GPIO_PORT_PTR)(port_info_ptr->gpio_ctrl);
236  DW_GPIO_CHECK_EXP(port->no <= DW_GPIO_PORT_D, E_OBJ);
237 
238  port_info_ptr->opn_cnt ++;
239  if (port_info_ptr->opn_cnt > 1) { /* opened before */
240  if (dir == port_info_ptr->direction) { /* direction is the same */
241  return E_OK;
242  } else { /* open with different direction */
243  return E_OPNED;
244  }
245  }
246 
247  dw_gpio_write_dir(port, port->valid_bit_mask, dir);
248 
249  if (port->no == DW_GPIO_PORT_A) {
250  dw_gpio_int_clear(port, DW_GPIO_MASK_ALL);
251  dw_gpio_int_disable(port, DW_GPIO_MASK_ALL);
252  dw_gpio_int_unmask(port, DW_GPIO_MASK_ALL);
253  /* install gpio interrupt handler */
254  int_handler_install(port->intno, port->int_handler);
255  int_disable(port->intno);
257  dw_gpio_set_int_cfg(port, (DEV_GPIO_INT_CFG *)(&gpio_int_cfg_default));
258  port_info_ptr->method = dw_gpio_read_mthd(port);
259  } else {
260  port_info_ptr->method = DEV_GPIO_BITS_MTHD_DEFAULT;
261  }
262 
263  dw_gpio_write_dr(port, port->valid_bit_mask, 0);
264 
265  port_info_ptr->direction = dir;
266  port_info_ptr->extra = NULL;
267 
268 error_exit:
269  return ercd;
270 }

352 int32_t dw_gpio_control(DEV_GPIO *gpio_obj, uint32_t ctrl_cmd, void *param)
353 {
354  int32_t ercd = E_OK;
355  DEV_GPIO_INFO_PTR port_info_ptr = &(gpio_obj->gpio_info);
356 
357  /* START ERROR CHECK */
358  VALID_CHK_GPIO_INFO_OBJECT(port_info_ptr);
359  /* END OF ERROR CHECK */
360 
361  DW_GPIO_PORT_PTR port = (DW_GPIO_PORT_PTR)(port_info_ptr->gpio_ctrl);
362  DW_GPIO_CHECK_EXP(port->no <= DW_GPIO_PORT_D, E_OBJ);
363  DW_GPIO_CHECK_EXP(port_info_ptr->opn_cnt > 0, E_CLSED);
364 
365  uint32_t val32;
367  if (ctrl_cmd == GPIO_CMD_SET_BIT_DIR_INPUT) {
368  val32 = (uint32_t)param;
369  dw_gpio_write_dir(port, val32, DW_GPIO_INPUT_ALL);
370  port_info_ptr->direction = dw_gpio_read_dir(port);
371  } else if (ctrl_cmd == GPIO_CMD_SET_BIT_DIR_OUTPUT) {
372  val32 = (uint32_t)param;
373  dw_gpio_write_dir(port, val32, DW_GPIO_OUTPUT_ALL);
374  port_info_ptr->direction = dw_gpio_read_dir(port);
375  } else if (ctrl_cmd == GPIO_CMD_GET_BIT_DIR) {
376  DW_GPIO_CHECK_EXP((param!=NULL) && CHECK_ALIGN_4BYTES(param), E_PAR);
377  port_info_ptr->direction = dw_gpio_read_dir(port);
378  *((int32_t *)param) = port_info_ptr->direction;
379  } else {
380  DW_GPIO_CHECK_EXP(port->no == DW_GPIO_PORT_A, E_NOSPT);
381  /* output pin cannot be used as interrupt */
382  DEV_GPIO_INT_CFG *gpio_int_cfg;
383  DEV_GPIO_BIT_ISR *port_bit_isr;
384 
385  switch (ctrl_cmd) {
386  case GPIO_CMD_SET_BIT_INT_CFG:
387  DW_GPIO_CHECK_EXP((param!=NULL) && CHECK_ALIGN_4BYTES(param), E_PAR);
388  gpio_int_cfg = (DEV_GPIO_INT_CFG *)param;
389  dw_gpio_set_int_cfg(port, gpio_int_cfg);
390  break;
391  case GPIO_CMD_GET_BIT_INT_CFG:
392  DW_GPIO_CHECK_EXP((param!=NULL) && CHECK_ALIGN_4BYTES(param), E_PAR);
393  gpio_int_cfg = (DEV_GPIO_INT_CFG *)param;
395  dw_gpio_get_int_cfg(port, gpio_int_cfg);
396  break;
397  case GPIO_CMD_SET_BIT_ISR:
398  DW_GPIO_CHECK_EXP((param!=NULL) && CHECK_ALIGN_4BYTES(param), E_PAR);
399  port_bit_isr = (DEV_GPIO_BIT_ISR *)param;
400  if (port_bit_isr->int_bit_ofs < port->gpio_bit_isr->int_bit_max_cnt) {
401  port->gpio_bit_isr->int_bit_handler_ptr[port_bit_isr->int_bit_ofs] = port_bit_isr->int_bit_handler;
402  } else {
403  ercd = E_PAR;
404  }
405  break;
406  case GPIO_CMD_GET_BIT_ISR:
407  DW_GPIO_CHECK_EXP((param!=NULL) && CHECK_ALIGN_4BYTES(param), E_PAR);
408  port_bit_isr = (DEV_GPIO_BIT_ISR *)param;
409  if (port_bit_isr->int_bit_ofs < port->gpio_bit_isr->int_bit_max_cnt) {
410  port_bit_isr->int_bit_handler = port->gpio_bit_isr->int_bit_handler_ptr[port_bit_isr->int_bit_ofs];
411  } else {
412  ercd = E_PAR;
413  }
414  break;
415  case GPIO_CMD_ENA_BIT_INT:
416  val32 = (uint32_t)param;
417  dw_gpio_int_enable(port, val32);
418  port_info_ptr->method = dw_gpio_read_mthd(port);
419  if (port_info_ptr->method) {
420  int_enable(port->intno);
421  }
422  break;
423  case GPIO_CMD_DIS_BIT_INT:
424  val32 = (uint32_t)param;
425  dw_gpio_int_disable(port, val32);
426  port_info_ptr->method = dw_gpio_read_mthd(port);
427  if (port_info_ptr->method == 0) {
428  int_disable(port->intno);
429  }
430  break;
431  case GPIO_CMD_GET_BIT_MTHD:
432  DW_GPIO_CHECK_EXP((param!=NULL) && CHECK_ALIGN_4BYTES(param), E_PAR);
433  port_info_ptr->method = dw_gpio_read_mthd(port);
434  *((int32_t *)param) = port_info_ptr->method;
435  break;
436  default:
437  ercd = E_NOSPT;
438  break;
439  }
440  }
441 error_exit:
442  return ercd;
443 }

446 int32_t dw_gpio_isr_handler(DEV_GPIO *gpio_obj, void *ptr)
447 {
448  int32_t ercd = E_OK;
449  DEV_GPIO_INFO_PTR port_info_ptr = &(gpio_obj->gpio_info);
450 
451  /* START ERROR CHECK */
452  VALID_CHK_GPIO_INFO_OBJECT(port_info_ptr);
453  /* END OF ERROR CHECK */
454 
455  DW_GPIO_PORT_PTR port = (DW_GPIO_PORT_PTR)(port_info_ptr->gpio_ctrl);
456  DW_GPIO_CHECK_EXP(port->no == DW_GPIO_PORT_A, E_NOSPT);
457 
458  uint32_t i, gpio_bit_isr_state;
459  uint32_t max_int_bit_count = 0;
460 
462  gpio_bit_isr_state = dw_gpio_int_read_status(port);
463 
464  if (port->gpio_bit_isr) {
465  max_int_bit_count = (port->gpio_bit_isr->int_bit_max_cnt);
466  } else {
467  dw_gpio_int_clear(port, gpio_bit_isr_state);
468  }
469 
470  for (i=0; i<max_int_bit_count; i++) {
471  if (gpio_bit_isr_state & (1<<i)) {
472  /* this bit interrupt enabled */
473  if (port->gpio_bit_isr->int_bit_handler_ptr[i]) {
474  port->gpio_bit_isr->int_bit_handler_ptr[i](gpio_obj);
475  }
476  dw_gpio_int_clear(port, (1<<i));
477  }
478  }
479 
480 error_exit:
481  return ercd;
