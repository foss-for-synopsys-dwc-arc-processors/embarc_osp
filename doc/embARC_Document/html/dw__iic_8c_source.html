283 static uint32_t dw_iic_get_slv_state(DW_IIC_REG *iic_reg_ptr)
284 {
285  uint32_t status;
286  uint32_t slv_state = IIC_SLAVE_STATE_FREE;
287 
288  status = iic_reg_ptr->IC_RAW_INTR_STAT;
289  if (status & IC_INTR_STAT_GEN_CALL) {
290  /* General Call address is received and it is acknowledged */
291  slv_state |= IIC_SLAVE_STATE_GC_REQ;
292  }
293  if (status & IC_INTR_STAT_RX_FULL) {
294  /* master is attempting to write data to this slave */
295  slv_state |= IIC_SLAVE_STATE_WR_REQ;
296  }
297  if (status & IC_INTR_STAT_RD_REQ) {
298  /* master is attempting to read data from this slave */
299  slv_state |= IIC_SLAVE_STATE_RD_REQ;
300  }
301  if (status & IC_INTR_STAT_RX_DONE) {
302  /* master does not acknowledge a transmitted byte, and transmission is done */
303  slv_state |= IIC_SLAVE_STATE_RD_DONE;
304  status = iic_reg_ptr->IC_CLR_RX_DONE;
305  }
306  if (status & IC_INTR_STAT_START_DET) {
307  /* a START or RESTART condition has occurred */
308  slv_state |= IIC_SLAVE_STATE_START;
309  status = iic_reg_ptr->IC_CLR_START_DET; /* Clear it when read */
310  }
311  if (status & IC_INTR_STAT_STOP_DET) {
312  /* a STOP condition has occurred */
313  slv_state |= IIC_SLAVE_STATE_STOP;
314  status = iic_reg_ptr->IC_CLR_STOP_DET; /* Clear it when read */
315  }
316  if (status & (IC_INTR_STAT_TX_ABRT|IC_INTR_STAT_TX_OVER\
317  |IC_INTR_STAT_RX_OVER|IC_INTR_STAT_RX_UNDER)) {
318  /* error case */
319  slv_state |= IIC_SLAVE_STATE_ERROR;
320  status = iic_reg_ptr->IC_CLR_TX_ABRT; /* Clear it when read */
321  status = iic_reg_ptr->IC_CLR_TX_OVER;
322  status = iic_reg_ptr->IC_CLR_RX_OVER;
323  status = iic_reg_ptr->IC_CLR_RX_UNDER;
324  }
325 
326  return slv_state;
327 }

330 static void dw_iic_master_init(DW_IIC_CTRL *iic_ctrl_ptr, uint32_t speed_mode, uint32_t addr_mode, uint32_t tar_addr)
331 {
332  uint32_t ic_con_val = 0;
333  DW_IIC_REG *iic_reg_ptr = iic_ctrl_ptr->dw_iic_regs;
334 
335  dw_iic_disable(iic_reg_ptr);
336 
337  /* disable all iic interrupt */
338  iic_reg_ptr->IC_INTR_MASK = IC_INT_DISABLE_ALL;
339 
340  /* Set to 7bit addressing and update target address */
341  iic_reg_ptr->IC_TAR = (tar_addr & IC_TAR_10BIT_ADDR_MASK) | IC_TAR_SPECIAL | IC_TAR_GC_OR_START;
342  /* master mode, restart enabled */
343  ic_con_val = dw_iic_select_speedmode(speed_mode) | IC_CON_ENA_MASTER_MODE | IC_CON_RESTART_EN;
344 
345 #if DW_IIC_DYNAMIC_TAR_UPDATE_SUPPORT
346  if (addr_mode == IIC_10BIT_ADDRESS) {
347  iic_reg_ptr->IC_TAR |= MST_10_BIT_ADDR_MODE;
348  }
349 #else
350  if (addr_mode == IIC_10BIT_ADDRESS) {
351  ic_con_val |= MST_10_BIT_ADDR_MODE;
352  }
353 #endif
354  /* Set final IC_CON value */
355  iic_reg_ptr->IC_CON = ic_con_val;
356  /* FIFO threshold settings */
357  iic_reg_ptr->IC_TX_TL = IIC_TX_THRESHOLD;
358  iic_reg_ptr->IC_RX_TL = IIC_RX_THRESHOLD;
359  /* Master code settings */
360  iic_reg_ptr->IC_HS_MADDR = iic_ctrl_ptr->iic_master_code;
361  dw_iic_enable(iic_reg_ptr);
362 
363  /* Clock Settings */
364  dw_iic_set_scl_cnt(iic_reg_ptr, &(iic_ctrl_ptr->iic_scl_cnt));
365  dw_iic_set_spike_len(iic_reg_ptr, &(iic_ctrl_ptr->iic_spklen));

398 static int32_t dw_iic_mst_chkerr(DW_IIC_CTRL *iic_ctrl_ptr)
399 {
400  uint32_t status;
401  int32_t ercd = IIC_ERR_NONE;
402  DW_IIC_REG *iic_reg_ptr = (DW_IIC_REG *)(iic_ctrl_ptr->dw_iic_regs);
403 
404  status = iic_reg_ptr->IC_RAW_INTR_STAT;
405  if (status & IC_INTR_STAT_TX_ABRT) {
406  status = iic_reg_ptr->IC_TX_ABRT_SOURCE;
407  if (status & IIC_MST_ABRT_LOST_BUS) {
408  ercd = IIC_ERR_LOST_BUS;
409  } else if (status & IIC_MST_ABRT_ADDR_NOACK) {
410  ercd = IIC_ERR_ADDR_NOACK;
411  } else if (status & IIC_MST_ABRT_DATA_NOACK) {
412  ercd = IIC_ERR_DATA_NOACK;
413  } else {
414  ercd = IIC_ERR_UNDEF;
415  }
416  status = iic_reg_ptr->IC_CLR_TX_ABRT;
417  } else {
418  if (status & IC_INTR_STAT_TX_OVER) {
419  iic_ctrl_ptr->iic_tx_over ++;
420  status = iic_reg_ptr->IC_CLR_TX_OVER;
421  }
422  if (status & (IC_INTR_STAT_RX_OVER|IC_INTR_STAT_RX_UNDER)) {
423  iic_ctrl_ptr->iic_rx_over ++;
424  status = iic_reg_ptr->IC_CLR_RX_OVER;
425  status = iic_reg_ptr->IC_CLR_RX_UNDER;
426  }
427  }
428  return ercd;
429 }
