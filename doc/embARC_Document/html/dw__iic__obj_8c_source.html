262 {
263  static uint32_t install_flag = 0;
264 
265  /* intall device objects */
266  if (install_flag == 0) {

268  dw_iic_all_install();
269  }
270 
271  switch (iic_id) {
272 #if (USE_DW_IIC_0)
273  case DW_IIC_0_ID:
274  return &dw_iic_0;
275  break;
276 #endif
277 #if (USE_DW_IIC_1)
278  case DW_IIC_1_ID:
279  return &dw_iic_1;
280  break;
281 #endif
282  default:
283  break;
284  }
285  return NULL;
286 }
