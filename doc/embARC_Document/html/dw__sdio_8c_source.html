59 Inline void dw_sdio_fifo_read_poll(DW_SDIO_CTRL_PTR sdio, uint32_t *buf, uint32_t size)
60 {
61  uint32_t len;
62  uint32_t i;
63 
64  while (size) {
65  len = dw_sdio_reg_read(sdio, DWSDIO_REG_STATUS);
66  len = (len & DWSDIO_STATUS_MASK_FIFO) >> DWSDIO_STATUS_BIT_FIFO;
67 
68  len = min(size, len);
69 
70  for (i = 0; i < len; i++) {
71  *buf++ = dw_sdio_reg_read(sdio, DWSDIO_REG_DATA);
72  }
73  size = size > len ? (size - len) : 0;
74  }
75 }

95 Inline void dw_sdio_fifo_write_poll(DW_SDIO_CTRL_PTR sdio, uint32_t *buf, uint32_t size)
96 {
97  uint32_t len;
98  uint32_t i;
99  uint32_t fifo_depth = sdio->fifo_depth;
100 
101  while (size) {
102  len = dw_sdio_reg_read(sdio, DWSDIO_REG_STATUS);
103  len = fifo_depth - ((len & DWSDIO_STATUS_MASK_FIFO) >> DWSDIO_STATUS_BIT_FIFO);
104  len = min(size, len);
105 
106  for (i = 0; i < len; i++) {
107  dw_sdio_reg_write(sdio, DWSDIO_REG_DATA, *buf++);

109  size = size > len ? (size - len) : 0;

154 Inline void dw_sdio_ref_clk_set(DW_SDIO_CTRL_PTR sdio, uint32_t ref_clk)
155 {
156  /* 0: 25 Mhz, 1: 50 Mhz, 2: 100 Mhz */
157  uint32_t reg;
158 
159  if (ref_clk == 50000000) {
160  ref_clk = 1;
161  } else if (ref_clk == 100000000) {
162  ref_clk = 2;
163  } else {
164  ref_clk = 0;
165  }
166 
167  reg = dw_sdio_reg_read(sdio, DWSDIO_REG_UHS_REG_EXT);
168  reg &= ~(DWSDIO_UHS_REG_EXT_MASK_REFCLK);
169  reg |= (ref_clk << DWSDIO_UHS_REG_EXT_BIT_REFCLK);
170  dw_sdio_reg_write(sdio, DWSDIO_REG_UHS_REG_EXT, reg);
171 }

173 static int32_t dw_sdio_bus_freq_set(DW_SDIO_CTRL_PTR sdio, uint32_t card_number, uint32_t freq)
174 {
175  uint32_t clk;
176  uint32_t div;
177  uint32_t status;
178  int32_t timeout = 10000;
179 
180  clk = sdio->ref_clk;
181 
182  if (freq == clk) {
183  div = 0;
184  } else {
185  div = DIV_ROUND_UP(clk, 2 * freq);
186  }
187 
188  dw_sdio_clk_disable(sdio, card_number);
189 
190  dw_sdio_reg_write(sdio, DWSDIO_REG_CLKSRC, 0);
191  dw_sdio_reg_write(sdio, DWSDIO_REG_CLKDIV, div);
192 
193 
194  dw_sdio_reg_write(sdio, DWSDIO_REG_CMD, DWSDIO_CMD_PRV_DAT_WAIT |
195  DWSDIO_CMD_UPD_CLK | DWSDIO_CMD_START | DWSDIO_CMD_USE_HOLD_REG);
196 
197  do {
198  status = dw_sdio_reg_read(sdio, DWSDIO_REG_CMD);
199  if (timeout-- < 0) {
200  DBG("%s: Timeout!\n", __func__);
201  return E_TMOUT;
202  }
203  } while (status & DWSDIO_CMD_START);
204 
205  dw_sdio_clk_enable(sdio, card_number);
206 
207  dw_sdio_reg_write(sdio, DWSDIO_REG_CMD, DWSDIO_CMD_PRV_DAT_WAIT |
208  DWSDIO_CMD_UPD_CLK | DWSDIO_CMD_START | DWSDIO_CMD_USE_HOLD_REG);
209 
210  timeout = 10000;
211  do {
212  status = dw_sdio_reg_read(sdio, DWSDIO_REG_CMD);
213  if (timeout-- < 0) {
214  DBG("%s: Timeout!\n", __func__);
215  return E_TMOUT;
216  }
217  } while (status & DWSDIO_CMD_START);
218 
219  return E_OK;
220 }

239 static int32_t dw_sdio_cmd_set(DW_SDIO_CTRL_PTR sdio, SDIO_CMD_PTR cmd, SDIO_DATA_PTR data)
240 {
241  uint32_t reg = 0;
242 
243  dw_sdio_reg_write(sdio, DWSDIO_REG_CMDARG, cmd->arg);
244 
245  if (data) {
246  reg |= DWSDIO_CMD_DATA_EXP;
247 
248  if (data->flags & SDIO_DATA_WRITE) {
249  reg |= DWSDIO_CMD_RW;
250  }
251  }
252 
253  if ((cmd->resp_type & SDIO_RSP_136) && (cmd->resp_type & SDIO_RSP_BUSY)) {
254  return -1;
255  }
256 
257  /* to fix */
258  if (cmd->cmdidx == 12) {
259  reg |= DWSDIO_CMD_ABORT_STOP;
260  } else {
261  reg |= DWSDIO_CMD_PRV_DAT_WAIT;
262  }
263 
264  if (cmd->resp_type & SDIO_RSP_PRESENT) {
265  reg |= DWSDIO_CMD_RESP_EXP;
266  if (cmd->resp_type & SDIO_RSP_136) {
267  reg |= DWSDIO_CMD_RESP_LENGTH;
268  }
269  }
270 
271  if (cmd->resp_type & SDIO_RSP_CRC) {
272  reg |= DWSDIO_CMD_CHECK_CRC;
273  }
274 
275  reg |= (cmd->cmdidx | (cmd->card << DWSDIO_CMD_BIT_CARD_NO) | DWSDIO_CMD_START
276  | DWSDIO_CMD_USE_HOLD_REG);
277 
278  DBG("Sending CMD%d to card %d \r\n",cmd->cmdidx, cmd->card);
279 
280  dw_sdio_reg_write(sdio, DWSDIO_REG_CMD, reg);
281 
282  return 0;
283 }

285 static int32_t dw_sdio_data_transfer_poll(DW_SDIO_CTRL_PTR sdio, SDIO_DATA_PTR data)
286 {
287  uint32_t size;
288  uint32_t status;
289  int32_t ret;
290  uint32_t timeout = 1000;
291  uint32_t start = OSP_GET_CUR_MS();
292 
293  size = data->blksz * data->blkcnt / 4;
294 
295  for (;;) {
296  status = dw_sdio_reg_read(sdio, DWSDIO_REG_RINTSTS);
297 
298  if (status & (DWSDIO_INT_DATA_ERR | DWSDIO_INT_DATA_TMO)) {
299  DBG("%s:, data transfer error!\r\n", __func__);
300  ret = E_SYS;
301  break;
302  }
303 
304  if (data->flags == SDIO_DATA_READ && (status & DWSDIO_INT_RXDR)) {
305  dw_sdio_fifo_read_poll(sdio, (uint32_t *)data->in, size);
306  dw_sdio_reg_write(sdio, DWSDIO_REG_RINTSTS, DWSDIO_INT_RXDR);
307  } else if (data->flags == SDIO_DATA_WRITE && (status & DWSDIO_INT_TXDR)) {
308  dw_sdio_fifo_write_poll(sdio, (uint32_t *)data->out, size);
309  dw_sdio_reg_write(sdio, DWSDIO_REG_RINTSTS, DWSDIO_INT_TXDR);
310  }
311 
312  if (status & DWSDIO_INT_DTO) {
313  ret = E_OK;
314  break;
315  }
316 
317  if ((OSP_GET_CUR_MS() - start) > timeout) {
318  DBG("%s: timeout on data transfer\r\n", __func__);
319  return E_TMOUT;
320  }
321  }
322 
323  return ret;
324 }

363 int32_t dw_sdio_cmd_poll(DEV_SDIO *sdio_obj, SDIO_CMD_PTR cmd, SDIO_DATA_PTR data)
364 {
365  DEV_SDIO_INFO_PTR sdio_info_ptr = &(sdio_obj->sdio_info);
366  DW_SDIO_CTRL_PTR sdio = (DW_SDIO_CTRL_PTR)sdio_info_ptr->sdio_ctrl;
367 
368  uint32_t start = OSP_GET_CUR_MS();
369  uint32_t timeout = 500;
370  uint32_t retry = 100000;
371  uint32_t mask;
372  uint32_t i;
373 
374  while (dw_sdio_reg_read(sdio, DWSDIO_REG_STATUS) & DWSDIO_STATUS_DATA_BUSY) {
375  if ((OSP_GET_CUR_MS() - start) > timeout) {
376  DBG("%s: timeout on data busy\r\n", __func__);
377  return E_TMOUT;
378  }
379  }
380 
381  dw_sdio_reg_write(sdio, DWSDIO_REG_RINTSTS, DWSDIO_INT_ALL);
382 
383  if (data) {
384  dw_sdio_enable_card_threshold(sdio, 1);
385  dw_sdio_block_size_set(sdio, data->blksz * data->blkcnt, data->blksz);
386  dw_sdio_reset_wait(sdio, DWSDIO_CTRL_RESET_FIFO);
387  }
388 
389  if (dw_sdio_cmd_set(sdio, cmd, data) < 0 ) {
390  return E_SYS;
391  }
392 
393  for (i = 0; i < retry; i++) {
394  mask = dw_sdio_reg_read(sdio, DWSDIO_REG_RINTSTS);
395  if (mask & DWSDIO_INT_CD) {
396  if (!data) {
397  dw_sdio_reg_write(sdio, DWSDIO_REG_RINTSTS, mask);
398  }
399  break;
400  }
401  }
402 
403  if (i == retry) {
404  DBG("%s: Timeout.\n", __func__);
405  return E_TMOUT;
406  }
407 
408  if (mask & DWSDIO_INT_RTO) {
409  /*
410  * Timeout here is not necessarily fatal. (e)MMC cards
411  * will splat here when they receive CMD55 as they do
412  * not support this command and that is exactly the way
413  * to tell them apart from SD cards. Thus, this output
414  * below shall be debug(). eMMC cards also do not favor
415  * CMD8, please keep that in mind.
416  */
417  DBG("%s: Response Timeout:%x.\r\n", __func__, mask);
418  return E_TMOUT;
419  } else if (mask & DWSDIO_INT_RE) {
420  DBG("%s: Response Error:%x.\r\n", __func__,mask);
421  return E_SYS;
422  }
423 
424 
425  if (cmd->resp_type & SDIO_RSP_PRESENT) {
426  if (cmd->resp_type & SDIO_RSP_136) {
427  cmd->resp[0] = dw_sdio_reg_read(sdio, DWSDIO_REG_RESP3);
428  cmd->resp[1] = dw_sdio_reg_read(sdio, DWSDIO_REG_RESP2);
429  cmd->resp[2] = dw_sdio_reg_read(sdio, DWSDIO_REG_RESP1);
430  cmd->resp[3] = dw_sdio_reg_read(sdio, DWSDIO_REG_RESP0);
431  } else {
432  cmd->resp[0] = dw_sdio_reg_read(sdio, DWSDIO_REG_RESP0);
433  }
434  }
435 
436  if (data) {
437  return dw_sdio_data_transfer_poll(sdio, data);
438  }
439 
440  return E_OK;
441 }

443 int32_t dw_sdio_open(DEV_SDIO *sdio_obj, uint32_t card_number)
444 {
445  uint32_t fifo_depth;
446  DEV_SDIO_INFO_PTR sdio_info_ptr = &(sdio_obj->sdio_info);
447  DW_SDIO_CTRL_PTR sdio = (DW_SDIO_CTRL_PTR)sdio_info_ptr->sdio_ctrl;
448 
449  if (sdio_info_ptr->opn_cnt > 0) {
450  dw_sdio_power_on(sdio, card_number);
451  dw_sdio_reset(sdio, card_number);
452  dw_sdio_bus_freq_set(sdio, card_number, DWSDIO_ENUMERATION_FREQ);
453 
454  sdio_info_ptr->opn_cnt++;
455  return E_OPNED;
456  }
457 
458  sdio_info_ptr->opn_cnt++;
459 
460  dw_sdio_reset_wait(sdio, DWSDIO_CTRL_RESET_ALL);
461  dw_sdio_ref_clk_set(sdio, sdio->ref_clk);
462 
463  dw_sdio_power_on(sdio, card_number);
464  dw_sdio_reset(sdio, card_number);
465 
466  dw_sdio_bus_type_set(sdio, card_number, 1);
467  dw_sdio_bus_freq_set(sdio, card_number, DWSDIO_ENUMERATION_FREQ);
468 
469 
470  if (sdio->fifo_depth == 0) {
471  fifo_depth = dw_sdio_reg_read(sdio, DWSDIO_REG_FIFOTH);
472  fifo_depth = ((fifo_depth & DWSDIO_FIFOTH_MASK_RX_WMARK) >>
473  DWSDIO_FIFOTH_BIT_RX_WMARK) + 1;
474  sdio->fifo_depth = fifo_depth;
475  }
476 
477  fifo_depth = DWSDIO_FIFOTH_M_SIZE(0x2) | DWSDIO_FIFOTH_RX_WMASK(sdio->fifo_depth / 2 - 1) |
478  DWSDIO_FIFOTH_TX_WMASK(sdio->fifo_depth / 2);
479  dw_sdio_reg_write(sdio, DWSDIO_REG_FIFOTH, fifo_depth);
480 
481  return E_OK;
482 }

517 #endif
518

520  return 1; /* the specific card detected */
521  }
