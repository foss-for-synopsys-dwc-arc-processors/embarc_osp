396 static int32_t dw_spi_dis_cbr(DEV_SPI_INFO *spi_info_ptr, uint32_t cbrtn)
397 {
398  DW_SPI_CTRL *spi_ctrl_ptr = (DW_SPI_CTRL *)(spi_info_ptr->spi_ctrl);
399  int32_t ercd = E_OK;
400 
401  if ((spi_info_ptr->status & DW_SPI_IN_XFER) != 0) { /* only in transfer need do check */
402  switch (cbrtn) {
403  case DW_SPI_RDY_SND:
404  DW_SPI_CHECK_EXP((spi_info_ptr->status & DW_SPI_IN_XFER) == DW_SPI_IN_TX, E_CTX);
405  spi_info_ptr->status &= ~(DW_SPI_IN_TX);
406  break;
407  case DW_SPI_RDY_RCV:
408  DW_SPI_CHECK_EXP((spi_info_ptr->status & DW_SPI_IN_XFER) == DW_SPI_IN_RX, E_CTX);
409  spi_info_ptr->status &= ~(DW_SPI_IN_RX);
410  break;
411  case DW_SPI_RDY_XFER:
412  DW_SPI_CHECK_EXP((spi_info_ptr->status & DW_SPI_IN_XFER) == DW_SPI_IN_XFER, E_CTX);
413  spi_info_ptr->status &= ~(DW_SPI_IN_XFER);
414  break;
415  default:
416  break;
417  }
418  }
419 
420  dw_spi_mask_interrupt(spi_ctrl_ptr->dw_spi_regs, DW_SPI_IMR_XFER);
421 
422  if (spi_ctrl_ptr->int_status & DW_SPI_GINT_ENABLE) {
423  if (spi_ctrl_ptr->intno != DW_SPI_INVALID_INTNO) {
424  int_disable(spi_ctrl_ptr->intno);
425  }
426  spi_ctrl_ptr->int_status &= ~DW_SPI_GINT_ENABLE;
427  }
428 
429 error_exit:
430  return ercd;
431 }

837  DW_SPI_CHECK_EXP((param>=SPI_CPOL_0_CPHA_0) && (param<=SPI_CPOL_1_CPHA_1), E_PAR);

844

848  (((support_modes)&DW_SPI_SLAVE_SUPPORTED)&&(mode == DEV_SLAVE_MODE)), E_NOSPT);

902  if (spi_ctrl_ptr->intno != DW_SPI_INVALID_INTNO) {
903  int_handler_install(spi_ctrl_ptr->intno, spi_ctrl_ptr->dw_spi_int_handler);
904  }

906  memset(&(spi_info_ptr->spi_cbs), 0, sizeof(DEV_SPI_CBS));
907 
908 error_exit:

910 }
