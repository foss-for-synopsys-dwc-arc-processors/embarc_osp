216 static int32_t dw_uart_set_dps(DW_UART_REG *uart_reg_ptr, UART_DPS_FORMAT *dps)
217 {
218  uint32_t dps_value = 0;
219 
220  if (dps == NULL) return -1;
221  /* data bits check */
222  if ((dps->databits < 5) || (dps->databits > 8)) return -1;
223  /* stop bits check */
224  if (dps->stopbits > UART_STPBITS_TWO) return -1;
225  /* parity bit type check */
226  if (dps->parity > UART_PARITY_SPACE) return -1;
227 
228  dps_value |= (uint32_t)dw_uart_databits[dps->databits-5];
229  dps_value |= (uint32_t)dw_uart_stopbits[dps->stopbits];
230  dps_value |= (uint32_t)dw_uart_parity[dps->parity];
231 
232  /* clear dps bits */
233  uart_reg_ptr->LCR &= (~DW_UART_LCR_DPS_MASK);
234  /* set dps bits */
235  uart_reg_ptr->LCR |= dps_value;
236 
237  return 0;
238 }

318 static void dw_uart_flush_output(DEV_UART_INFO *uart_info_ptr)
319 {
320  uint32_t i;
321  char *p_charbuf;
322 
323  DW_UART_CTRL *uart_ctrl_ptr = (DW_UART_CTRL_PTR)(uart_info_ptr->uart_ctrl);
324  DW_UART_REG *uart_reg_ptr = (DW_UART_REG_PTR)(uart_ctrl_ptr->dw_uart_regbase);
325 
326  if (uart_info_ptr->tx_buf.buf != NULL) {
327  p_charbuf = (char *)(uart_info_ptr->tx_buf.buf);
328  for (i = uart_info_ptr->tx_buf.ofs; i < uart_info_ptr->tx_buf.len; i ++) {
329  dw_uart_psnd_chr(uart_reg_ptr, p_charbuf[i]);
330  }
331  /* clear transmit buffer */
332  uart_info_ptr->tx_buf.buf = NULL;
333  uart_info_ptr->tx_buf.len = 0;
334  uart_info_ptr->tx_buf.ofs = 0;
335  }
336  /* wait until transmit fifo is empty */
337  while ((uart_reg_ptr->USR & DW_UART_USR_TFE) == 0);
338  while (uart_reg_ptr->USR & DW_UART_USR_BUSY);
339 }

346 static void dw_uart_dis_cbr(DEV_UART_INFO *uart_info_ptr, uint32_t cbrtn)
347 {
348  DW_UART_CTRL *uart_ctrl_ptr = (DW_UART_CTRL_PTR)(uart_info_ptr->uart_ctrl);
349  DW_UART_REG *uart_reg_ptr = (DW_UART_REG_PTR)(uart_ctrl_ptr->dw_uart_regbase);
350 
351  switch (cbrtn) {
352  case DW_UART_RDY_SND:
353  uart_reg_ptr->IER &= ~DW_UART_IER_XMIT_EMPTY;
354  uart_ctrl_ptr->int_status &= ~DW_UART_TXINT_ENABLE;
355  break;
356  case DW_UART_RDY_RCV:
357  uart_reg_ptr->IER &= ~DW_UART_IER_DATA_AVAIL;
358  uart_ctrl_ptr->int_status &= ~DW_UART_RXINT_ENABLE;
359  break;
360  default:
361  break;
362  }
363  if (uart_ctrl_ptr->int_status & DW_UART_GINT_ENABLE) {
364  if ((uart_ctrl_ptr->int_status & (DW_UART_RXINT_ENABLE|DW_UART_TXINT_ENABLE)) == 0) {
365  int_disable(uart_ctrl_ptr->intno);
366  uart_ctrl_ptr->int_status &= ~DW_UART_GINT_ENABLE;
367  }
368  }
369 }

534 int32_t dw_uart_open (DEV_UART *uart_obj, uint32_t baud)
535 {
536  int32_t ercd = E_OK;
537  DEV_UART_INFO *uart_info_ptr = &(uart_obj->uart_info);
538 
539  /* START ERROR CHECK */
540  VALID_CHK_UART_INFO_OBJECT(uart_info_ptr);
541  DW_UART_CHECK_EXP(baud>0, E_PAR);
542  /* END OF ERROR CHECK */
543 
544  uart_info_ptr->opn_cnt ++;
545  if (uart_info_ptr->opn_cnt > 1) { /* opened before */
546  if (baud == uart_info_ptr->baudrate) { /* baudrate is the same */
547  return E_OK;
548  } else { /* open with different baudrate */
549  return E_OPNED;
550  }
551  }
552 
553  int32_t baud_divisor = 0;
554 
555  DW_UART_CTRL *uart_ctrl_ptr = (DW_UART_CTRL_PTR)(uart_info_ptr->uart_ctrl);
556  DW_UART_REG *uart_reg_ptr = (DW_UART_REG_PTR)(uart_ctrl_ptr->dw_uart_regbase);
557 
558  /* Get FIFO Length */
559  uart_ctrl_ptr->tx_fifo_len = dw_uart_get_txfifo_len(uart_reg_ptr);
560  uart_ctrl_ptr->rx_fifo_len = dw_uart_get_rxfifo_len(uart_reg_ptr);
561 
563  uart_info_ptr->baudrate = baud;
564  baud_divisor = DW_UART_BAUD2DIV(uart_ctrl_ptr->dw_apb_bus_freq, baud);
565  uart_info_ptr->dps_format = dps_format_default;
566  uart_info_ptr->hwfc = hwfc_default;
567  dw_uart_init(uart_reg_ptr, baud_divisor, &(uart_info_ptr->dps_format), uart_info_ptr->hwfc);
568 
569  uart_info_ptr->status = DEV_ENABLED;
570  uart_info_ptr->extra = NULL;
571 
575  dw_uart_disable_interrupt(uart_info_ptr);
577  int_handler_install(uart_ctrl_ptr->intno, uart_ctrl_ptr->dw_uart_int_handler);
578 
579  memset(&(uart_info_ptr->tx_buf), 0, sizeof(DEV_BUFFER));
580  memset(&(uart_info_ptr->rx_buf), 0, sizeof(DEV_BUFFER));
581  memset(&(uart_info_ptr->uart_cbs), 0, sizeof(DEV_UART_CBS));
582 
583 error_exit:
584  return ercd;
585 }

594 int32_t dw_uart_close (DEV_UART *uart_obj)
595 {
596  int32_t ercd = E_OK;
597  DEV_UART_INFO *uart_info_ptr = &(uart_obj->uart_info);
598 
599  /* START ERROR CHECK */
600  VALID_CHK_UART_INFO_OBJECT(uart_info_ptr);
601  DW_UART_CHECK_EXP(uart_info_ptr->opn_cnt > 0, E_OK);
602  /* END OF ERROR CHECK */
603 
604  uart_info_ptr->opn_cnt --;
605  if (uart_info_ptr->opn_cnt == 0) {
606  dw_uart_disable_interrupt(uart_info_ptr);
607  dw_uart_abort_tx(uart_obj);
608  dw_uart_abort_rx(uart_obj);
609  dw_uart_flush_output(uart_info_ptr);
610  memset(&(uart_info_ptr->tx_buf), 0, sizeof(DEV_BUFFER));
611  memset(&(uart_info_ptr->rx_buf), 0, sizeof(DEV_BUFFER));
612  memset(&(uart_info_ptr->uart_cbs), 0, sizeof(DEV_UART_CBS));
613  dw_uart_disable_device(uart_info_ptr);
614  uart_info_ptr->status = 0;
615  uart_info_ptr->extra = NULL;
616  } else {
617  ercd = E_OPNED;
618  }
619 
620 error_exit:
621  return ercd;
622 }

638 int32_t dw_uart_control (DEV_UART *uart_obj, uint32_t ctrl_cmd, void *param)
639 {
640  int32_t ercd = E_OK;
641  DEV_UART_INFO *uart_info_ptr = &(uart_obj->uart_info);
642 
643  /* START ERROR CHECK */
644  VALID_CHK_UART_INFO_OBJECT(uart_info_ptr);
645  DW_UART_CHECK_EXP(uart_info_ptr->opn_cnt > 0, E_CLSED);
646  /* END OF ERROR CHECK */
647 
648  uint32_t val32;
649  int32_t baud_divisor = 0;
650  DEV_BUFFER *devbuf;
651  UART_DPS_FORMAT *dps_ptr;
652  UART_HW_FLOW_CONTROL hwfc_local;
653 
654  DW_UART_CTRL *uart_ctrl_ptr = (DW_UART_CTRL_PTR)(uart_info_ptr->uart_ctrl);
655  DW_UART_REG *uart_reg_ptr = (DW_UART_REG_PTR)(uart_ctrl_ptr->dw_uart_regbase);
656 
657  /* check whether current device is disabled */
658  if ((uart_info_ptr->status & DEV_ENABLED) == 0) {
663  if ((ctrl_cmd != UART_CMD_ENA_DEV) && \
664  (ctrl_cmd != UART_CMD_DIS_DEV) && \
665  (ctrl_cmd != UART_CMD_GET_STATUS) ) {
666  return E_SYS;
667  }
668  }
669 
670  switch (ctrl_cmd) {
671  case UART_CMD_SET_BAUD:
672  val32 = (uint32_t)param;
673  DW_UART_CHECK_EXP(val32>0, E_PAR);
674  if (val32 != uart_info_ptr->baudrate) {
675  baud_divisor = DW_UART_BAUD2DIV(uart_ctrl_ptr->dw_apb_bus_freq, val32);
676  dw_uart_set_baud(uart_reg_ptr, baud_divisor);
677  uart_info_ptr->baudrate = val32;
678  }
679  break;
680  case UART_CMD_GET_STATUS:
681  DW_UART_CHECK_EXP((param!=NULL) && CHECK_ALIGN_4BYTES(param), E_PAR);
682  *((int32_t *)param) = uart_info_ptr->status;
683  break;
684  case UART_CMD_ENA_DEV:
685  dw_uart_enable_device(uart_info_ptr);
686  break;

870 void dw_uart_isr (DEV_UART *uart_obj, void *ptr)
871 {
872  int32_t ercd = E_OK;
873  DEV_UART_INFO *uart_info_ptr = &(uart_obj->uart_info);
874 
875  /* START ERROR CHECK */
876  VALID_CHK_UART_INFO_OBJECT(uart_info_ptr);
877  /* END OF ERROR CHECK */
878 
879  uint32_t uart_int_status;
880  volatile uint32_t temp;
881  DEV_BUFFER *buf_ptr;
882  char *p_charbuf;
883 
884  DW_UART_CTRL *uart_ctrl_ptr = (DW_UART_CTRL_PTR)(uart_info_ptr->uart_ctrl);
885  DW_UART_REG *uart_reg_ptr = (DW_UART_REG_PTR)(uart_ctrl_ptr->dw_uart_regbase);
886 
888  uart_int_status = (uart_reg_ptr->IIR) & DW_UART_IIR_INT_ID_MASK;
889 
890  switch (uart_int_status) {
891  case DW_UART_IIR_MDM_STATUS:
892  temp = (volatile uint32_t)(uart_reg_ptr->MSR);
893  break;
894  case DW_UART_IIR_LINE_STATUS:
895  if (uart_info_ptr->uart_cbs.err_cb) {
896  uart_info_ptr->uart_cbs.err_cb(uart_info_ptr);
897  }
898  temp = (volatile uint32_t)(uart_reg_ptr->LSR);
899  break;
900  case DW_UART_IIR_XMIT_EMPTY:
901  buf_ptr = &(uart_info_ptr->tx_buf);
902  p_charbuf = (char *)buf_ptr->buf;
903  if (p_charbuf != NULL) {
904  while (dw_uart_putready(uart_reg_ptr)) {
905  dw_uart_putchar(uart_reg_ptr, p_charbuf[buf_ptr->ofs]);
906  buf_ptr->ofs ++;
907  if (buf_ptr->ofs >= buf_ptr->len) {
908  dw_uart_dis_cbr(uart_info_ptr, DW_UART_RDY_SND);
909  if (uart_info_ptr->uart_cbs.tx_cb) {
910  uart_info_ptr->uart_cbs.tx_cb(uart_obj);
911  }
912  /* clear the send buffer pointer */
913  memset(buf_ptr, 0, sizeof(DEV_BUFFER));
914  break;
915  }
916  }
917  } else {
918  if (uart_info_ptr->uart_cbs.tx_cb) {
919  uart_info_ptr->uart_cbs.tx_cb(uart_obj);
920  }
921  }
922  break;
923  case DW_UART_IIR_RX_TIMEOUT:
924  temp = dw_uart_getchar(uart_reg_ptr);
925  break;
926  case DW_UART_IIR_DATA_AVAIL:
927  buf_ptr = &(uart_info_ptr->rx_buf);
928  p_charbuf = (char *)buf_ptr->buf;
929  if (p_charbuf != NULL) {
930  while (dw_uart_getready(uart_reg_ptr)) {
931  p_charbuf[buf_ptr->ofs] = (char)dw_uart_getchar(uart_reg_ptr);
932  buf_ptr->ofs ++;
933  if (buf_ptr->ofs >= buf_ptr->len) {
934  dw_uart_dis_cbr(uart_info_ptr, DW_UART_RDY_RCV);
935  if (uart_info_ptr->uart_cbs.rx_cb) {
936  uart_info_ptr->uart_cbs.rx_cb(uart_obj);
937  }
938  /* clear the send buffer pointer */
939  memset(buf_ptr, 0, sizeof(DEV_BUFFER));
940  break;
941  }
942  }
943  } else {
944  if (uart_info_ptr->uart_cbs.rx_cb) {
945  uart_info_ptr->uart_cbs.rx_cb(uart_obj);
946  }
947  }
948  break;
949  default:
950  temp = (volatile uint32_t)(uart_reg_ptr->USR);
951  break;
952  }
953 
954 error_exit:
955  return;
