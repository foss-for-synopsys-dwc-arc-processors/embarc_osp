104 static void porta_install(void)
105 {
106  uint32_t i;
107  DEV_GPIO_PTR port_ptr = &port_a;
108  DEV_GPIO_INFO_PTR info_ptr = &(port_a.gpio_info);
109  DW_GPIO_PORT_PTR dw_port_ptr = &(dw_gpio_port_a);
110 
111  info_ptr->gpio_ctrl = (void *)dw_port_ptr;
112  info_ptr->opn_cnt = 0;
113  info_ptr->method = 0;
114  info_ptr->direction = 0;
115  info_ptr->extra = 0;
116 
117  dw_port_ptr->no = DW_GPIO_PORT_A;
118  dw_port_ptr->regs = (DW_GPIO_REG_PTR)(PERIPHERAL_BASE|REL_REGBASE_GPIO0);
119  dw_port_ptr->valid_bit_mask = EMSK_GPIO_A_VALID_MASK;
120  dw_port_ptr->intno = INTNO_GPIO;
121  dw_port_ptr->int_handler = porta_isr;
122 
123  for (i=0; i<dw_gpio_bit_isr_a.int_bit_max_cnt; i++) {
124  dw_gpio_bit_isr_a.int_bit_handler_ptr[i] = NULL;
125  }
126  dw_port_ptr->gpio_bit_isr = &dw_gpio_bit_isr_a;
127 
128  port_ptr->gpio_open = porta_open;
129  port_ptr->gpio_close = porta_close;
130  port_ptr->gpio_control = porta_control;
131  port_ptr->gpio_write = porta_write;
132  port_ptr->gpio_read = porta_read;
133 }

323 DEV_GPIO_PTR gpio_get_dev(int32_t gpio_id)
324 {
325  static uint32_t install_flag = 0;
326 
327  /* intall device objects */
328  if (install_flag == 0) {
329  install_flag = 1;
330  dw_gpio_all_install();
331  }
332 
333  switch (gpio_id) {
334 #if (USE_DW_GPIO_PORT_A)
335  case DW_GPIO_PORT_A: return &port_a;
336 #endif
337 #if (USE_DW_GPIO_PORT_B)
338  case DW_GPIO_PORT_B: return &port_b;
339 #endif
340 #if (USE_DW_GPIO_PORT_C)
341  case DW_GPIO_PORT_C: return &port_c;
342 #endif
343 #if (USE_DW_GPIO_PORT_D)
344  case DW_GPIO_PORT_D: return &port_d;
345 #endif
346  default:
347  break;
348  }
349  return NULL;
350 }
