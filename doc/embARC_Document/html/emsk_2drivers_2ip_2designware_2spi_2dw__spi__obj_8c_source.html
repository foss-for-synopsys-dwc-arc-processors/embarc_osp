276 {
277  static uint32_t install_flag = 0;
278 
279  /* install device objects */
280  if (install_flag == 0) {
281  install_flag = 1;
282  dw_spi_all_install();
283  }
284 
285  switch (spi_id) {
286 #if (USE_DW_SPI_0)
287  case DW_SPI_0_ID:
288  return &dw_spi_0;
289  break;
290 #endif
291 #if (USE_DW_SPI_1)
292  case DW_SPI_1_ID:
293  return &dw_spi_1;
294  break;
295 #endif
296  default:
297  break;
298  }
299  return NULL;
300 }
