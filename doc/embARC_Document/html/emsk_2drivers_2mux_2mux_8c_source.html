127 int32_t io_pmod_config(uint32_t pmod, uint32_t type, uint32_t config)
128 {
129  uint32_t val = get_pmod_mux();
130  switch (pmod) {
131  case PMOD_1: /* PMOD GPIO, PMOD_UART */
132  if (type == PMOD_GPIO) {
133  val &= ~(PM1_UR_UART_0 | PM1_LR_SPI_S);
134  } else if (type == PMOD_UART) {
135  val |= PM1_UR_UART_0;
136  } else {
137  return E_NOSPT;
138  }
139  break;
140  case PMOD_2: /* PMOD GPIO, PMOD_I2C */
141  case PMOD_3:
142  case PMOD_4:
143  pmod--;
144  pmod <<= 2; /* (pmode - 1) * 4*/
145  if (type == PMOD_GPIO) {
146  val &= ~(1 << pmod);
147  } else if (type == PMOD_I2C) {
148  val |= (1 << pmod);
149  } else {
150  return E_NOSPT;
151  }
152  break;
153  case PMOD_5: /* PMOD GPIO, PMOD_SPI */
154  if (type == PMOD_GPIO) {
155  val &= ~(PM5_UR_SPI_M1);
156  } else if (type == PMOD_SPI) {
157  val |= PM5_UR_SPI_M1;
158  } else {
159  return E_NOSPT;
160  }
161  break;
162  case PMOD_6: /* PMOD GPIO, PMOD_SPI */
163  if (type == PMOD_GPIO) {
164  val &= ~(PM6_UR_SPI_M0);
165  } else if (type == PMOD_SPI) {
166  val |= PM6_UR_SPI_M0;
167  } else {
168  return E_NOSPT;
169  }
170  break;
171  default:
172  return E_PAR;
173  }
174 
175  set_pmod_mux(val);
176 
177  return E_OK;
178 }
