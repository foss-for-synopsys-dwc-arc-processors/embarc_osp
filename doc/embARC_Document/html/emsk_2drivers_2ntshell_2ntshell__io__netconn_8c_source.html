133 static void ntshell_netconn_thread(void *arg)
134 {
135  err_t ercd = E_OK;
136  NTSHELL_IO *nt_io = (NTSHELL_IO *)arg;
137  NTSHELL_IO_NETCONN *nt_netconn;
138  struct netconn *temp_conn;
139  uint8_t *p_dont_echo = &dont_echo[0];
140 
141  EMSK_NTSHELL_NETCONN_CHECK_EXP(nt_io!=NULL, E_OBJ);
142  EMSK_NTSHELL_NETCONN_CHECK_EXP(nt_io->extra_info!=NULL, E_OBJ);
143 
144  nt_netconn = (NTSHELL_IO_NETCONN *)nt_io->extra_info;
145 
146  nt_netconn->local_conn = netconn_new(NETCONN_TCP);
147  netconn_bind(nt_netconn->local_conn, NULL, 23);
148  netconn_listen(nt_netconn->local_conn);
149 
150  nt_netconn->conn_status = CONN_NOTCONNECT;
151  while (1) {
152  ercd = netconn_accept(nt_netconn->local_conn, &temp_conn);
153  nt_netconn->conn_status = CONN_NOTCONNECT;
154  if (ercd == ERR_OK) {
155  if (nt_netconn->remote_conn != NULL) { /* kill previous connection */
156  sys_mbox_trypost(&nt_netconn->remote_conn->recvmbox, NULL);
157  sys_msleep(200);
158  netconn_close(nt_netconn->remote_conn);
159  netconn_delete(nt_netconn->remote_conn);
160  DBG("Delete old connection\n");
161  }
162  DBG("New connection connected\n");
163  nt_netconn->remote_conn = temp_conn;
164  nt_netconn->conn_status = CONN_SUCCESS;
165  ntshell_netconn_nt_write(nt_netconn, p_dont_echo, 3);
166  } else {
167  nt_netconn->conn_status = CONN_FAILED;
168  }
169  }
170 
171 error_exit:
172  return;
173 }

195 static int32_t netconn_nt_read_chr_raw(NTSHELL_IO_NETCONN *nt_netconn)
196 {
197  int8_t rcv_chr;
198  int ercd;
199  if (nt_netconn->conn_status == CONN_SUCCESS) {
200  if (nt_netconn->rcv_ofs >= nt_netconn->rcv_len) {
201  if (nt_netconn->pbuf_rcv != NULL) { /* free it when valid */
202  pbuf_free(nt_netconn->pbuf_rcv);
203  }
204  ercd = netconn_recv_tcp_pbuf(nt_netconn->remote_conn, &nt_netconn->pbuf_rcv);
205  if (ercd == ERR_OK) {
206  nt_netconn->rcv_ofs = 0;
207  nt_netconn->rcv_len = nt_netconn->pbuf_rcv->tot_len;
208  } else {
209  nt_netconn->conn_status = CONN_LOST;
210  return E_OBJ;
211  }
212  }
213  if (nt_netconn->pbuf_rcv == NULL) {
214  return E_OBJ;
215  }
216  pbuf_copy_partial(nt_netconn->pbuf_rcv, &rcv_chr, 1, nt_netconn->rcv_ofs);
217  nt_netconn->rcv_ofs ++;
218  return ((int32_t)rcv_chr) & 0xff;
219  }
221  if (nt_netconn->pbuf_rcv != NULL) { /* free it when valid */
222  pbuf_free(nt_netconn->pbuf_rcv);
223  }
224  nt_netconn->rcv_ofs = 0;
225  nt_netconn->rcv_len = 0;
226  return E_OBJ;
227 }
228 static int32_t netconn_nt_read_chr(NTSHELL_IO_NETCONN *nt_netconn)
229 {
230  int32_t ch, req;
231 
232  while ( (ch = netconn_nt_read_chr_raw(nt_netconn)) == TELNET_OPT_IAC \
233  || ch == '\0' && ch != E_OBJ) {
234  if (ch != '\0') {
235  switch (ch = netconn_nt_read_chr_raw(nt_netconn)) {
236  case TELNET_OPT_WILL:
237  case TELNET_OPT_WONT:
238  case TELNET_OPT_DO:
239  case TELNET_OPT_DONT:
240  req = netconn_nt_read_chr_raw(nt_netconn);
241  if (req != E_OBJ) {
242  DBG("Telnet Request:%x\n\r", req);
243  } else {
244  return req;
245  }
246  break;
247  case TELNET_OPT_SB:
248  while ((ch = netconn_nt_read_chr_raw(nt_netconn)) != E_OBJ && ch != TELNET_OPT_IAC) {
249  if ((ch = netconn_nt_read_chr_raw(nt_netconn)) == E_OBJ || ch == TELNET_OPT_SE) {
250  break;
251  }
252  }
253  break;
254  default:
255  break;
256  }
257  }
258  }
259 
260  return ch;
261 }

294 static int32_t ntshell_netconn_nt_write(NTSHELL_IO_NETCONN *nt_netconn, const void *buf, uint32_t cnt)
295 {
296  int32_t ercd = E_OK;
297  uint32_t wricnt = 0;
298  uint8_t crlf_snd = 1, crlf_snd_save = 1;
299  const uint8_t *p_charbuf = (const uint8_t *)buf;
300  uint8_t sndchr = 0;
301 
302  EMSK_NTSHELL_NETCONN_CHECK_EXP(nt_netconn!=NULL, E_OBJ);
303  EMSK_NTSHELL_NETCONN_CHECK_EXP(buf!=NULL, E_OBJ);
304 
305  while (wricnt < cnt) {
306  crlf_snd_save = crlf_snd;
307  sndchr = p_charbuf[wricnt];
308  if (crlf_snd && sndchr == '\n' && nt_netconn->conn_ctrl & CONN_CTRL_CRLF) {
309  crlf_snd = 0;
310  sndchr = '\r';
311  } else {
312  crlf_snd = 1;
313  }
314  netconn_nt_write_chr(nt_netconn, sndchr);
315  if (nt_netconn->conn_status != CONN_SUCCESS) {
316  break;
317  }
318  wricnt += crlf_snd;
319  }
320 
321  ercd = wricnt;
322 error_exit:
323  return ercd;
324 }
