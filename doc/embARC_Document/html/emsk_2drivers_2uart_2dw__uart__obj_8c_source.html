302 {
303  static uint32_t install_flag = 0;
304 
305  /* intall device objects */
306  if (install_flag == 0) {

308  dw_uart_all_install();
309  }
310 
311  switch (uart_id) {
312 #if (USE_DW_UART_0)
313  case DW_UART_0_ID:
314  return &dw_uart_0;
315  break;
316 #endif
317 #if (USE_DW_UART_1)

320  break;
321 #endif
322 #if (USE_DW_UART_2)
323  case DW_UART_2_ID:

326 #endif
327  default:

329  }
