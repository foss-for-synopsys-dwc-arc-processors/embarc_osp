79 {

82  EMSK_GPIO_CHECK_EXP_NORTN(led_port != NULL);
83 
84  if (led_port->gpio_open(EMSK_LED_DIR) == E_OPNED) {
85  led_port->gpio_control(GPIO_CMD_SET_BIT_DIR_OUTPUT, (void *)(EMSK_LED_MASK));

131 static uint32_t g_led_val;
133 void led_write(uint32_t led_val, uint32_t mask)
134 {
135  EMSK_GPIO_CHECK_EXP_NORTN(led_port != NULL);
136 
137  led_val = (~led_val) & mask;
138  led_val = led_val << EMSK_LED_OFFSET;
139  mask = (mask << EMSK_LED_OFFSET) & EMSK_LED_MASK;
140  led_port->gpio_write(led_val, mask);
141 
142  g_led_val = led_val;
143 
144 error_exit:
145  return;
146 }

150 uint32_t led_read(uint32_t mask)
151 {
152  uint32_t value;
153 
154  EMSK_GPIO_CHECK_EXP_NORTN(led_port != NULL);
155 
156  mask = (mask << EMSK_LED_OFFSET) & EMSK_LED_MASK;
157  led_port->gpio_read(&value, mask);
158  value = (~value) & EMSK_LED_MASK;
