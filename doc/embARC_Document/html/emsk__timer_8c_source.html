87 void board_timer_update(uint32_t precision)
88 {
89  static uint32_t sys_hz_update = 0;
90  static uint32_t sys_ms_update = 0;
91  uint32_t hz_conv = 0;
92 
94  hz_conv = HZ_COUNT_CONV(precision, BOARD_SYS_TIMER_HZ);
95  sys_hz_update ++;
96  if (sys_hz_update >= hz_conv) {
97  sys_hz_update = 0;
98  gl_emsk_sys_hz_cnt ++;
99  }
100 
102  hz_conv = HZ_COUNT_CONV(precision, BOARD_SYS_TIMER_MS_HZ);
103  sys_ms_update ++;
104  if (sys_ms_update >= hz_conv) {
105  sys_ms_update = 0;
106  gl_emsk_ms_cnt ++;
107 
108  emsk_sdcard_1ms_update();
109  }
110 }

129 void emsk_timer_init(void)
130 {
131  if (timer_present(BOARD_SYS_TIMER_ID)) {
132  int_disable(BOARD_SYS_TIMER_INTNO); /* disable first then enable */
133  int_handler_install(BOARD_SYS_TIMER_INTNO, emsk_timer_isr);
134  timer_start(BOARD_SYS_TIMER_ID, TIMER_CTRL_IE|TIMER_CTRL_NH, cyc_hz_count); /* start 1ms timer interrupt */
135 
136  int_enable(BOARD_SYS_TIMER_INTNO);
137  }
138 }

160 uint64_t board_get_cur_us(void)
161 {
162  uint32_t sub_us;
163  uint64_t total_us;
164  timer_current(TIMER_0, &sub_us);
165 
166  sub_us = ((uint64_t)sub_us * 1000000) / BOARD_CPU_CLOCK;
167  total_us = ((uint64_t)OSP_GET_CUR_MS()) * 1000 + (uint64_t)sub_us;
168 
169  return total_us;
170 }

186 void board_delay_ms(uint32_t ms, uint8_t os_compat)
187 {

198 #endif
199  us_delayed = ((uint64_t)ms * 1000);
200  start_us = board_get_cur_us();
