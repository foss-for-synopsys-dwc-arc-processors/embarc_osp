129 static void hnd_get_temp(coap_context_t *ctx, struct coap_resource_t *resource,
130  coap_address_t *peer, coap_pdu_t *request, str *token,
131  coap_pdu_t *response) {
132 
133  coap_opt_iterator_t opt_iter;
134  unsigned char buf[40];
135  size_t len;
136  int32_t val;
137  int32_t temp_val;
138  coap_subscription_t *subscription;
139 
140  response->hdr->code = COAP_RESPONSE_CODE(205);
141 
142  /* handle observe request */
143  if (request != NULL &&
144  coap_check_option(request, COAP_OPTION_OBSERVE, &opt_iter)) {
145  subscription = coap_add_observer(resource, peer, token);
146  if (subscription) {
147  subscription->non = request->hdr->type == COAP_MESSAGE_NON;
148  coap_add_option(response, COAP_OPTION_OBSERVE, 0, NULL);
149  }
150  }
151 
152  if (resource->dirty == 1) {
153  coap_add_option(response, COAP_OPTION_OBSERVE,
154  coap_encode_var_bytes(buf, ctx->observe), buf);
155  }
156 
157  coap_add_option(response, COAP_OPTION_CONTENT_FORMAT,
158  coap_encode_var_bytes(buf, COAP_MEDIATYPE_TEXT_PLAIN), buf);
159 
160  /* temp data is fresh in 60 seconds */
161  coap_add_option(response, COAP_OPTION_MAXAGE,
162  coap_encode_var_bytes(buf, 60), buf);
163 
164  if (temp_sensor_read(&val) != E_OK) {
165  temp_val = 0;
166  } else {
167  temp_val = val / 10;
168  }
169 
170  len = snprintf((char *)buf,
171  min(sizeof(buf), response->max_size - response->length),
172  "%d", temp_val);
173  coap_add_data(response, len, buf);
174 }
