69 static int32_t read_spi(uint32_t address, uint32_t size, uint32_t* checksum)
70 {
71  uint8_t local_buf[BUF_SIZE];
72  int32_t cnt, datalen, len;
73  uint32_t csum;
74  static void *ram = (void *)0x17f00000;
75  if( checksum == NULL)
76  return 0;
77 
78  flash_wait_ready();
79 
80  csum = *checksum;
81 
82  datalen = 0;
83  do {
84  cnt = flash_read(address, BUF_SIZE, local_buf);
85  if (cnt < 0) {
86  *checksum = 0;
87  return 0;
88  }
89  len = (size < cnt) ? size : cnt;
90  csum = make_checks(csum, local_buf, len);
91  memcpy(ram, local_buf, len);
92  ram = ram + len;
93  size -= len;
94  datalen += len;
95  address += len;
96  } while (size);
97 
98  *checksum = csum;
99 
100  return datalen;
101  }

103 static int32_t reload_btloader(void)
104 {
105  image_t header;
106  uint32_t datalen;
107  uint32_t checksum = 0;
108 
109  // read spi header
110  flash_init();
111  datalen = flash_read(BOOTIMAGE_START, IMAGE_HEADER_LEN, &header);
112 
113  if(BOOTIMAGE_HEAD == header.head && \
114  header.start >= BOOTIMAGE_START && header.start < BOOTIMAGE_START + BOOTIMAGE_MAX_SIZE && \
115  header.start >= BOOTIMAGE_START && header.start < BOOTIMAGE_START + BOOTIMAGE_MAX_SIZE && \
116  header.size > 0 && header.size <= BOOTIMAGE_MAX_SIZE ) {
117  datalen = read_spi(header.start, header.size, &checksum);
118 
119  if(checksum != header.checksum) {
120  datalen = 0;
121  }
122  } else {
123  datalen = 0;
124  }
125 
126  return datalen;
127 }
