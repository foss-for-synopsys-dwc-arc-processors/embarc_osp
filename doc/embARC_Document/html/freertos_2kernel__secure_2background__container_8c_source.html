125 static unsigned int perf_end(void)
126 {
127  unsigned int end = 0;
128 
129  if (timer_current(perf_id, (void *)(&end)) < 0) {
130  return 0;
131  }
132 
133  if (start < end) {
134  return (end - start);
135  } else {
136  return (0xFFFFFFFF - start + end);
137  }
