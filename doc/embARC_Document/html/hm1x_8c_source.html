99 void hm1x_flush(void)
100 {
101  uint32_t rd_cnt = 0;
102  uint8_t read_chr[MAX_READ_CNT];
103  uint32_t rd_avail;
104 
105  hm1x_ble_uart->uart_control(UART_CMD_GET_RXAVAIL, (void *)(&rd_avail));
106  do {
107  rd_cnt = (rd_avail>MAX_READ_CNT)?MAX_READ_CNT:rd_avail;
108  if (rd_avail > 0) {
109  hm1x_ble_uart->uart_read((void *)read_chr, rd_cnt);
110  }
111  hm1x_ble_uart->uart_control(UART_CMD_GET_RXAVAIL, (void *)(&rd_avail));
112  } while(rd_avail > 0);
113 }

177 int32_t hm1x_exec_command(char *cmd, char *resp, uint32_t wait_ms)
178 {
179  uint32_t cur_ofs = 0;
180  uint32_t read_cnt;
181  uint32_t cur_time;
182 
183  if ((cmd == NULL) || (resp == NULL)) return -1;
184 
185  if (hm1x_write((uint8_t *)cmd, strlen(cmd)) > 0) {
186  cur_time = HM1X_TIMENOW();
187  do {
188  read_cnt = hm1x_read((uint8_t *)(&resp[cur_ofs]), 2);
189  cur_ofs += read_cnt;
190  } while((HM1X_TIMENOW()-cur_time) < wait_ms);
191  resp[cur_ofs] = '\0';
192  return cur_ofs;
193  }
194  return -1;
195 }

206 int32_t hm1x_exec_cmd_chkresp(char *cmd, char *resp_prefix, char *resp_data, uint32_t max_try)
207 {
208  char resp[MAX_RESP_LENGTH];
209  uint32_t try_cnt = 0;
210  int32_t rd_cnt;
211  int32_t resp_prefix_len = 0;
212  int32_t resp_data_len = -1;
213 
214  if (cmd == NULL) return -1;
215 
216  if (resp_prefix != NULL) {
217  resp_prefix_len = strlen(resp_prefix);
218  }
219 
220  do {
221  rd_cnt = hm1x_exec_command(cmd, resp, HM1X_READ_TMOUT);
222  if (rd_cnt >= resp_prefix_len) {
223  if (resp_prefix_len > 0) {
224  if (strncmp(resp, resp_prefix, resp_prefix_len) != 0) {
225  continue;
226  }
227  }
228  if (resp_data == NULL) {
229  resp_data_len = 0;
230  } else {
231  resp_data_len = rd_cnt - resp_prefix_len;
232  if (resp_data_len > 0) {
233  strcpy(resp_data, &resp[resp_prefix_len]);
234  }
235  }
236  break;
237  }
238  } while (try_cnt++ < max_try);
239 
240  return resp_data_len;
241 }

245 int32_t hm1x_set_param(char *param, char *val)
246 {
247  char cmd_buf[MAX_CMD_LENGTH];
248  uint32_t cmd_length = 0;
249 
250  if ((param == NULL) || (val == NULL)) return -1;
251 
252  cmd_length = strlen(param) + strlen(val) + strlen(AT_CMD_PREFIX);
253  if (cmd_length > MAX_CMD_LENGTH) {
254  HM1X_DEBUG("HM1X command is too long to execute!\r\n");
255  return -1;
256  }
257  sprintf(cmd_buf, "%s%s%s", AT_CMD_PREFIX, param, val);
258 
259  if (hm1x_exec_cmd_chkresp(cmd_buf, OK_SET_PREFIX, NULL, 2) != -1) {
260  return 0;
261  } else {
262  return -1;
263  }
264 }
