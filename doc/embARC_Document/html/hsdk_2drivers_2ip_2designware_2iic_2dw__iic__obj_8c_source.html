350 DEV_IIC_PTR iic_get_dev(int32_t iic_id)
351 {
352  static uint32_t install_flag = 0;
353 
354  /* intall device objects */
355  if (install_flag == 0) {
356  install_flag = 1;
357  dw_iic_all_install();
358  }
359 
360  switch (iic_id) {
361 #if (USE_DW_IIC_0)
362  case DW_IIC_0_ID:
363  return &dw_iic_0;
364  break;
365 #endif
366 #if (USE_DW_IIC_1)
367  case DW_IIC_1_ID:
368  return &dw_iic_1;
369  break;
370 #endif
371 #if (USE_DW_IIC_2)
372  case DW_IIC_2_ID:
373  return &dw_iic_2;
374  break;
375 #endif
376  default:
377  break;
378  }
379  return NULL;
380 }
