97 /* gpio ops */
98 static int32_t pmwifi_0_gpio_open(uint32_t pin_mask, uint32_t pin_dir)
99 {
100  int32_t ercd = E_OK;
101  uint32_t mask;
102  DEV_GPIO *pmwifi_gpio_ptr;
103 
104  pmwifi_gpio_ptr = gpio_get_dev(HSDK_PMWIFI_0_GPIO_ID);
105  PMWIFI_CHECK_EXP(pmwifi_gpio_ptr != NULL, E_OBJ);
106 
107  pin_mask = ((pin_mask & WF_GPIO_PIN_MASK_ALL)<<HSDK_PMWIFI_0_PIN_GAP);
108  mask = (pin_dir<<HSDK_PMWIFI_0_PIN_GAP) & pin_mask;
109  ercd = pmwifi_gpio_ptr->gpio_open(mask);
110  if (ercd == E_OPNED) {
111  /* direction */
112  pmwifi_gpio_ptr->gpio_control(GPIO_CMD_SET_BIT_DIR_OUTPUT, (void *)mask);
113  mask = (~(pin_dir<<HSDK_PMWIFI_0_PIN_GAP)) & pin_mask;
114  pmwifi_gpio_ptr->gpio_control(GPIO_CMD_SET_BIT_DIR_INPUT, (void *)mask);
115  }
116  /* Default all set to pull */
117  ercd = pmwifi_gpio_ptr->gpio_control(GPIO_CMD_DIS_BIT_INT, (void *)pin_mask);
118 
119 error_exit:
120  return ercd;
121 }
122 static int32_t pmwifi_0_gpio_close(uint32_t pin_mask)
123 {
124  static uint32_t close_mask = 0;
125  int32_t ercd = E_OK;
126  DEV_GPIO *pmwifi_gpio_ptr;
127 
128  pmwifi_gpio_ptr = gpio_get_dev(HSDK_PMWIFI_0_GPIO_ID);
129  PMWIFI_CHECK_EXP(pmwifi_gpio_ptr != NULL, E_OBJ);
130 
131  close_mask |= pin_mask;
132  close_mask &= WF_GPIO_PIN_MASK_ALL;
133 
134  if (close_mask == WF_GPIO_PIN_MASK_ALL) { /* only close it when all pin need to be closed */
135  ercd = pmwifi_gpio_ptr->gpio_close();
136  close_mask = 0;
137  }
138 
139 error_exit:
140  return ercd;
141 }

172 static int32_t pmwifi_0_gpio_eint_install(DEV_GPIO_HANDLER gpio_handler)
173 {
174  int32_t ercd = E_OK;
175  DEV_GPIO *pmwifi_gpio_ptr;
176  DEV_GPIO_BIT_ISR bit_isr;
177  DEV_GPIO_INT_CFG int_cfg;
178 
179  pmwifi_gpio_ptr = gpio_get_dev(HSDK_PMWIFI_0_GPIO_ID);
180  PMWIFI_CHECK_EXP(pmwifi_gpio_ptr != NULL, E_OBJ);
181 
182  int_cfg.int_bit_mask = (1<<HSDK_PMWIFI_0_INT_PIN);
183  int_cfg.int_bit_type = GPIO_INT_BIT_LEVEL_TRIG(HSDK_PMWIFI_0_INT_PIN);
184  int_cfg.int_bit_polarity = GPIO_INT_BIT_POL_ACT_LOW(HSDK_PMWIFI_0_INT_PIN);
185  int_cfg.int_bit_debounce = GPIO_INT_BIT_DIS_DEBOUNCE(HSDK_PMWIFI_0_INT_PIN);
186  ercd = pmwifi_gpio_ptr->gpio_control(GPIO_CMD_SET_BIT_INT_CFG, (void *)(&int_cfg));
187  bit_isr.int_bit_ofs = HSDK_PMWIFI_0_INT_PIN;
188  bit_isr.int_bit_handler = gpio_handler;
189  ercd = pmwifi_gpio_ptr->gpio_control(GPIO_CMD_SET_BIT_ISR, (void *)(&bit_isr));
190 
191 error_exit:
192  return ercd;
193 }
194 static int32_t pmwifi_0_gpio_eint_is_enabled(void)
195 {
196  int32_t ercd = E_OK;
197  uint32_t mask;
198  uint32_t mthd;
199  DEV_GPIO *pmwifi_gpio_ptr;
200 
201  pmwifi_gpio_ptr = gpio_get_dev(HSDK_PMWIFI_0_GPIO_ID);
202  PMWIFI_CHECK_EXP(pmwifi_gpio_ptr != NULL, E_OBJ);
203 
204  mask = (1 << HSDK_PMWIFI_0_INT_PIN);
205  ercd = pmwifi_gpio_ptr->gpio_control(GPIO_CMD_GET_BIT_MTHD, (void *)(&mthd));
206  if (ercd == E_OK) {
207  mthd &= mask;
208  ercd = ((mthd==0) ? WF_EINT_DISABLE:WF_EINT_ENABLE);
209  }
210 
211 error_exit:
212  return ercd;
213 }
214 static int32_t pmwifi_0_gpio_eint_control(int32_t state)
215 {
216  int32_t ercd = E_OK;

219 
220  pmwifi_gpio_ptr = gpio_get_dev(HSDK_PMWIFI_0_GPIO_ID);
221  PMWIFI_CHECK_EXP(pmwifi_gpio_ptr != NULL, E_OBJ);
222 
223  mask = (1 << HSDK_PMWIFI_0_INT_PIN);
224  if (state == WF_EINT_ENABLE) {
225  ercd = pmwifi_gpio_ptr->gpio_control(GPIO_CMD_ENA_BIT_INT, (void *)mask);
226  } else {
227  ercd = pmwifi_gpio_ptr->gpio_control(GPIO_CMD_DIS_BIT_INT, (void *)mask);
228  }
229 
230 error_exit:
231  return ercd;
232 }
