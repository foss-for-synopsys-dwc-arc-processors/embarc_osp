63 static void arc_connect_int_isr(void *ptr)
64 {
65  uint32_t reg;
66  uint32_t i;
67  uint32_t core;
68 
69  core = (_arc_lr_reg(AUX_IDENTITY) >> 8) & 0xff;
70 
71  for (i=HSDC_CREG_SW0_ISR; i<HSDK_MAX_NUM_ISR; i++) {
72  reg = arc_connect_idu_read_mask(i-HSDK_MAX_NUM_EXCP);
73  if (reg & 0x1) { /* check whether the interrupt is masked */
74  continue;
75  }
76  if ( (i == HSDC_CREG_UPDATE_ISR) || (i == HSDC_CGU_PLL_LOCK_ISR) \
77  || (i == HSDC_CGU_PLL_UNLOCK_ISR) || (i == HSDC_CGU_PLL_LOCKERROR_ISR) \
78  || (i == HSDC_TUNNEL_ISR) ) {
79  //Edge triggered (creg(update) + cgu(3x))
80  reg = arc_connect_idu_check_status(i-HSDK_MAX_NUM_EXCP);
81  if (reg & (1<<core)) {
82  arc_connect_idu_ack_cirq(i-HSDK_MAX_NUM_EXCP);
83  hsdk_int_handler_table[i-HSDK_MAX_NUM_EXCP]((void *)i);
84  }
85  } else {
86  //Level triggered
87  reg = arc_connect_idu_read_mode(i-HSDK_MAX_NUM_EXCP) & 0x3U;
88  if (reg == 1) {
89  //Level triggered IRQ
90  reg = arc_connect_idu_check_first(i-HSDK_MAX_NUM_EXCP);
91  if (reg == 1) {
92  //Execute handler
93  hsdk_int_handler_table[i-HSDK_MAX_NUM_EXCP]((void *)i);
94  }
95  }
96  }
97  }
98 }

362 void hsdk_interrupt_init(void)
363 {
364  uint32_t i;
365  uint32_t core;
366 
367  core = (_arc_lr_reg(AUX_IDENTITY) >> 8) & 0xff;
368 
369  for (i = HSDK_MAX_NUM_EXCP; i < HSDK_MAX_NUM_ISR; i ++) {
370  if ( (i == HSDC_CREG_UPDATE_ISR) || (i == HSDC_CGU_PLL_LOCK_ISR) \
371  || (i == HSDC_CGU_PLL_UNLOCK_ISR) || (i == HSDC_CGU_PLL_LOCKERROR_ISR) \
372  || (i == HSDC_TUNNEL_ISR) ) {
373  arc_connect_init_isr(1<<core, i-HSDK_MAX_NUM_EXCP, ARC_CONNECT_INTRPT_TRIGGER_EDGE, ARC_CONNECT_DISTRI_MODE_FIRST_ACK);
374  } else {
375  arc_connect_init_isr(1<<core, i-HSDK_MAX_NUM_EXCP, ARC_CONNECT_INTRPT_TRIGGER_LEVEL, ARC_CONNECT_DISTRI_MODE_FIRST_ACK);
376  }
377  }
