77 static void * prv_connect_server(uint16_t secObjInstID, void * userData)
78 {
79  client_data_t * dataP;
80  char * uri;
81  char * host;
82  char * portStr;
83  int port;
84  char * ptr;
85  connection_t * newConnP = NULL;
86 
87  dataP = (client_data_t *)userData;
88 
89  uri = get_server_uri(dataP->securityObjP, secObjInstID);
90  if (uri == NULL) return NULL;
91 
92  // parse uri in the form "coaps://[host]:[port]"
93  if (0==strncmp(uri, "coaps://", strlen("coaps://"))) {
94  host = uri+strlen("coaps://");
95  } else if (0==strncmp(uri, "coap://", strlen("coap://"))) {
96  host = uri+strlen("coap://");
97  } else {
98  goto exit;
99  }
100 
101  portStr = strchr(host, ':');
102  if (portStr == NULL) goto exit;
103  // split strings
104  *portStr = 0;
105  portStr++;
106  port = strtol(portStr, &ptr, 10);
107  if (*ptr != 0) goto exit;
108 
109  EMBARC_PRINTF("Trying to connect to LWM2M Server at %s:%d\r\n", host, port);
110  newConnP = connection_create(dataP->connList, dataP->sock, host, port);
111  if (newConnP == NULL) {
112  EMBARC_PRINTF("Connection creation failed.\r\n");
113  } else {
114  dataP->connList = newConnP;
115  }
116 
117 exit:
118  lwm2m_free(uri);
119  return (void *)newConnP;
120 }

141 int lwm2mclient(lwm2m_client_info *client_info)
142 {
143  client_data_t data;
144  int result;
145  lwm2m_context_t * lwm2mH = NULL;
146  const char * server = client_info->server;
147  const char * serverPort = client_info->serverPort;
148  const char * name = client_info->ep_name;
149  int lifetime = client_info->lifetime;
150  time_t reboot_time = 0;
151 
152 
153  EMBARC_PRINTF("LwM2M client(%s) try to connect to LwM2M server @%s:%s.\n", name, server, serverPort);
154 
155  memset(&data, 0, sizeof(client_data_t));
156 
157  /*
158  *This call an internal function that create an IPV6 socket on the port 5683.
159  */
160  EMBARC_PRINTF("Trying to open a socket for LwM2M Connection\r\n");
161  data.sock = create_socket(0, LWM2M_SERVER_PORT);
162  if (data.sock < 0) {
163  EMBARC_PRINTF("Failed to open socket: %d\r\n", errno);
164  return -1;
165  }
166 
167  /*
168  * Now the main function fill an array with each object, this list will be later passed to liblwm2m.
169  * Those functions are located in their respective object file.
170  */
171 
172  char serverUri[50];
173  int serverId = 123;
174 
175  sprintf(serverUri, "coap://%s:%s", server, serverPort);
176  objArray[0] = get_security_object(serverId, serverUri, false);
177  if (NULL == objArray[0]) {
178  EMBARC_PRINTF("Failed to create security object\r\n");
179  return -1;
180  }
181  data.securityObjP = objArray[0];
182 
183  objArray[1] = get_server_object(serverId, "U", lifetime, false);
184  if (NULL == objArray[1]) {
185  EMBARC_PRINTF("Failed to create server object\r\n");
186  return -1;
187  }
188 
189  objArray[2] = get_object_device();
190  if (NULL == objArray[2]) {
191  EMBARC_PRINTF("Failed to create device object\r\n");
192  return -1;
193  }
194 
195  objArray[3] = get_object_firmware();
196  if (NULL == objArray[3]) {
197  EMBARC_PRINTF("Failed to create firmware object\r\n");
198  return -1;
199  }
200 /* get the user object*/
201  obj_info *next_obj;
202  next_obj = client_info->obj;
203  int i = 4;
204  while (next_obj) {
205  switch (next_obj->oid) {
206 /* temperature*/
207  case 3303:
208  objArray[i] = get_object_temperature(next_obj->number);
209  break;
210 /* light*/
211  case 3311:
212  objArray[i] = get_object_light(next_obj->number);
213  break;
214 /* button*/
215  case 3347:
216  objArray[i] = get_object_button(next_obj->number);
217  break;
218 /*display*/
219  case 3341:
220  objArray[i] = get_object_display(next_obj->number);
221  break;
222  }
223 
224  EMBARC_PRINTF("create object: %d\r\n", next_obj->oid);
225  if (NULL == objArray[i]) {
226  EMBARC_PRINTF("Failed to create object: %d\r\n", next_obj->oid);
227  return -1;
228  }
229  next_obj = next_obj->next;
230  i++;
231  }
232  /*
233  * The liblwm2m library is now initialized with the functions that will be in
234  * charge of communication
235  */
236  lwm2mH = lwm2m_init(prv_connect_server, prv_buffer_send, &data);
237  if (NULL == lwm2mH) {
238  EMBARC_PRINTF("lwm2m_init() failed\r\n");
239  return -1;
240  }
241 
242  /*
243  * We configure the liblwm2m library with the name of the client - which shall be unique for each client -
244  * the number of objects we will be passing through and the objects array
245  */
246 
247  result = lwm2m_configure(lwm2mH, (char *)(name), NULL, NULL, i, objArray);
248 
249  if (result != 0) {
250  lwm2m_free(lwm2mH);
251  EMBARC_PRINTF("lwm2m_set_objects() failed: 0x%X\r\n", result);
252  return -1;
253  }
254 
255  /*
256  * This function start your client to the LWM2M servers
257  */
258  result = lwm2m_start(lwm2mH);
259  if (result != 0) {
260  lwm2m_free(lwm2mH);
261  EMBARC_PRINTF("lwm2m_register() failed: 0x%X\r\n", result);
262  return -1;
263  }
264 
268  init_value_change(lwm2mH);
269 
270  EMBARC_PRINTF("lwm2m client started.\n");
271  /*
272  * We now enter in a while loop that will handle the communications from the server
273  */
274  g_quit = 0;
275  g_reboot = 0;
276  while (0 == g_quit) {
277  struct timeval tv;
278  fd_set readfds;
279 
280  if (g_reboot) {
281  time_t tv_sec;
282 
283  tv_sec = lwm2m_gettime();
284 
285  if (0 == reboot_time) {
286  reboot_time = tv_sec + 5;
287  }
288  if (reboot_time < tv_sec){
289  /*
290  * Message should normally be lost with reboot ...
291  */
292  EMBARC_PRINTF("reboot time expired, rebooting ...\n");
293  g_quit = 1;
294  } else {
295  tv.tv_sec = reboot_time - tv_sec;
296  }
297  } else {
298  tv.tv_sec = 60;
299  }
300  tv.tv_usec = 0;
301 
302  FD_ZERO(&readfds);
303  FD_SET(data.sock, &readfds);
304 
305  /*
306  * This function does two things:
307  * - first it does the work needed by liblwm2m (eg. (re)sending some packets).
308  * - Secondly it adjust the timeout value (default 60s) depending on the state of the transaction
309  * (eg. retransmission) and the time between the next operation
310  */
311  result = lwm2m_step(lwm2mH, (time_t *)(&(tv.tv_sec)));
312  if (result != 0) {
313  EMBARC_PRINTF("lwm2m_step() failed: 0x%X\r\n", result);
314  return -1;
315  }
316 
317  /*
318  * This part will set up an interruption until an event happen on SDTIN or the socket until "tv" timed out (set
319  * with the precedent function)
320  */
321  result = select(FD_SETSIZE, &readfds, NULL, NULL, &tv);
322 
323  if (result < 0) {
324  if (errno != EINTR) {
325  EMBARC_PRINTF("Error in select(): %d\r\n", errno);
326  }
327  } else if (result > 0) {
328  uint8_t buffer[MAX_PACKET_SIZE];
329  int numBytes;
330 
331  /*
332  * If an event happen on the socket
333  */
334  if (FD_ISSET(data.sock, &readfds)) {
335  struct sockaddr_storage addr;
336  socklen_t addrLen;
337 
338  addrLen = sizeof(addr);
339 
340  /*
341  * We retrieve the data received
342  */
343  numBytes = recvfrom(data.sock, buffer, MAX_PACKET_SIZE, 0, (struct sockaddr *)&addr, &addrLen);
344 
345  if (0 > numBytes) {
346  EMBARC_PRINTF("Error in recvfrom(): %d\r\n", errno);
347  } else if (0 < numBytes) {
348  char s[INET_ADDRSTRLEN];
349  in_port_t port = 0;
350  connection_t * connP;
351 
352  if (AF_INET == addr.ss_family) {
353  struct sockaddr_in *saddr = (struct sockaddr_in *)&addr;
354  inet_ntop(saddr->sin_family, &(saddr->sin_addr), s, INET_ADDRSTRLEN);
355  port = ntohs(saddr->sin_port);
356  }
357 
358  EMBARC_PRINTF("%d bytes received from [%s]:%d\r\n", numBytes, s, port);
359 
360  /*
361  * Display it in the STDERR
362  */
363  //output_buffer(buffer, numBytes);
364 
365  connP = connection_find(data.connList, &addr, addrLen);
366  if (connP != NULL) {
367  /*
368  * Let liblwm2m respond to the query depending on the context
369  */
370  lwm2m_handle_packet(lwm2mH, buffer, numBytes, connP);
371  } else {
372  EMBARC_PRINTF("received bytes ignored!\r\n");
373  }
374  }
375  }
376  }
377  }
378 
379  /*
380  * Finally when the loop is left smoothly - asked by user in the command line interface - we unregister our client from it
381  */
382  if (g_quit != 0) {
383  lwm2m_close(lwm2mH);
384  }
385  close(data.sock);
386  connection_free(data.connList);
387  if (g_reboot != 0){
388  system_reboot();
389  }
390  return 0;
391 }
