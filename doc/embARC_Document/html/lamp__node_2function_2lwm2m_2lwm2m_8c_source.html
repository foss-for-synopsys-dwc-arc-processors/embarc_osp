84 static void task_lwm2m_client(void *par)
85 {
86  unsigned int cpu_status;
87 
88  while (1) {
89  lwm2m_client_conn_stat = 1;
90 
91  /* get into sub function(lwm2mclient) to do all work about data interaction with gateway(lwm2m server) */
92  if (lwm2mclient(&c_info) == 0) {
93  EMBARC_PRINTF("LwM2M client end successfully\r\n");
94  } else {
95  EMBARC_PRINTF("Error: LwM2M client end failed\r\n");
96  }
97 
98  cpu_status = cpu_lock_save();
99  if (c_info.server != NULL) free((void *)(c_info.server));
100  if (c_info.ep_name != NULL) free((void *)(c_info.ep_name));
101  if (c_info.serverPort != NULL) free((void *)(c_info.serverPort));
102  lwm2m_client_conn_stat = 0;
103  cpu_unlock_restore(cpu_status);
104 
105  vTaskSuspend(task_lwm2m_client_handle);
106  }
107 }

110 extern int32_t lwm2m_client_start(void)
111 {
112  int c_quit = 0;
113  task_lwm2m_client_handle = 0;
114 
115  /* check to see if wifi works */
116  if (!lwip_pmwifi_isup()) {
117  EMBARC_PRINTF("Error: Wifi is not ready for lwM2M client.\r\n");
118  goto error_exit;
119  }
120 
121  /* exit lwm2mClient */
122  if (c_quit == 1) {
123  EMBARC_PRINTF("Error: Try to exit existing client.\r\n");
124  handle_sigint(2);
125  goto error_exit;
126  }
127 
128  if (lwm2m_client_conn_stat == 1) {
129  EMBARC_PRINTF("Error: LwM2M client is already running.\r\n");
130  goto error_exit;
131  }
132 
133  if (p_server == NULL) {
134  EMBARC_PRINTF("Error: Server Url is not specified, please check it.\r\n");
135  goto error_exit;
136  }
137 
138  c_info.server = p_server;
139  c_info.ep_name = p_client_name;
140  c_info.serverPort = p_port;
141 
142  lwm2m_client_start_flag = 1;
143  EMBARC_PRINTF("Start lwm2m client.\r\n");
144 
145  /* create or resume task for lwm2mClient to realize communication with iBaby Smarthome Gateway */
146  if (xTaskCreate(task_lwm2m_client, "lwm2m client", STACK_DEPTH_LWM2M, NULL, TSKPRI_HIGH,
147  &task_lwm2m_client_handle) != pdPASS) {
148  EMBARC_PRINTF("Error: Create task_lwm2m_client failed\r\n");
149  return E_SYS;
150  } else {
151  vTaskResume(task_lwm2m_client_handle);
152  return E_OK;
153  }
154 
155  return E_OK;
156 
157 error_exit:
158  return E_SYS;
159 }
160 #endif/* LWM2M_CLIENT */
