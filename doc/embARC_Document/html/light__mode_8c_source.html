354 void light_mode_ride(uint8_t freq, uint8_t bri)
355 {
356  uint8_t i;
357  float y_radian, z_radian;
358  uint8_t brightness = bri * 0x04;
359  int16_t y_cos, z_cos, z_judge_acc, z_acc_radian;
360  static int16_t old_y_acc = 0, old_z_acc = LIGHT_RIDE_ACC_SUM, y_acc;
361  light_ctr_rgb_update(LIGHT_MASK_SIDE_MID, LIGHT_ROW_SIDE, 0xff, 0xff, 0xff);
362  light_ctr_rgb_update(0x0180, LIGHT_ROW_MIDDLE, 0xff, 0xff, 0xff);
363 
364  /* Light beats from the middle to both sides in green gradually */
365  for (i = 1; i < (LIGHT_NUM_MIDDLE / 2); i++) {
366  light_ctr_rgb_update(LIGHT_MASK_MIDDLE_M1 >> i, LIGHT_ROW_MIDDLE, 0x00, brightness, bri);
367  light_ctr_rgb_update((0x000000100 >> i) & 0x0000000fe, LIGHT_ROW_SIDE, 0x00, brightness, bri);
368  light_ctr_rgb_update((0x002000000 >> i) & 0x001fc0000, LIGHT_ROW_SIDE, 0x00, brightness, bri);
369  light_ctr_rgb_update((0x000000100 << i) & 0x00000fe00, LIGHT_ROW_SIDE, 0x00, brightness, bri);
370  light_ctr_rgb((light_ctr_mask_lshift(0x002000000, i)) & 0x1fc000000, LIGHT_ROW_SIDE, 0x00, brightness, bri);
371  light_ctr_rgb(LIGHT_MASK_MIDDLE_M2 << i, LIGHT_ROW_MIDDLE, 0x00, brightness, bri);
372  vTaskDelay(30);
373  }
374 
375  light_ctr_rgb(0x200030001, LIGHT_ROW_SIDE, 0xff, 0xff, 0xff);
376  vTaskDelay(40);
377  light_ctr_rgb(0x200030001, LIGHT_ROW_SIDE, 0x00, 0x00, 0x00);
378 
379  /* Light beats from the middle to both sides in yellow gradually */
380  for (i = 1; i < (LIGHT_NUM_MIDDLE / 2); i++) {
381  light_ctr_rgb_update(LIGHT_MASK_MIDDLE_M1 >> i, LIGHT_ROW_MIDDLE, brightness, brightness, bri / 4);
382  light_ctr_rgb_update((0x000000100 >> i) & 0x0000000fe, LIGHT_ROW_SIDE, brightness, brightness, bri / 4);
383  light_ctr_rgb_update((0x002000000 >> i) & 0x001fc0000, LIGHT_ROW_SIDE, brightness, brightness, bri / 4);
384  light_ctr_rgb_update((0x000000100 << i) & 0x00000fe00, LIGHT_ROW_SIDE, brightness, brightness, bri / 4);
385  light_ctr_rgb((light_ctr_mask_lshift(0x002000000, i)) & 0x1fc000000, LIGHT_ROW_SIDE, brightness, brightness, bri / 4);
386  light_ctr_rgb(LIGHT_MASK_MIDDLE_M2 << i, LIGHT_ROW_MIDDLE, brightness, brightness, bri / 4);
387  vTaskDelay(30);
388  }
389 
390  /* Calculate the cosine of the Y axis */
391  y_radian = (float)y_acc / (float)LIGHT_RIDE_ACC_SUM;
392 
393  /* Calculate the angle between the Y axis and the acceleration of gravity */
394  y_cos = acos(y_radian) * 180 / PI;
395 
396  /* According to the angle between the Y axis and the acceleration of gravity,caculate the angle of Y axis */
397  z_cos = 90 - y_cos;
398 
399  /* Caculate the cosine of the Y axis */
400  z_radian = cos(z_cos * PI / 180);
401 
402  /* Increase the Z-axis cosine by a factor of 1000 */
403  z_acc_radian = z_radian * 1000;
404 
405  /* Find the actual acceleration on the Z axis */
406  z_judge_acc = abs(imu_mpu6050_update_ptr -> acc_z) - z_acc_radian;
407 
408  if ((imu_mpu6050_update_ptr -> acc_z < (-RIDE_STOP_THRESHOLD + z_acc_radian)) && z_judge_acc > RIDE_Z_ACC_JUDGE) {
409 
410  for (i = 0; i < (LIGHT_NUM_MIDDLE / 2); i++) {
411  light_ctr_rgb_update(0x200000000 >> i, LIGHT_ROW_SIDE, brightness / 4, 0x00, 0x00);
412  light_ctr_rgb_update(0x000000001 << i, LIGHT_ROW_SIDE, brightness / 4, 0x00, 0x00);
413  light_ctr_rgb_update((LIGHT_MASK_MIDDLE_FST << i) & 0x7f, LIGHT_ROW_MIDDLE, brightness, 0x00, 0x00);
