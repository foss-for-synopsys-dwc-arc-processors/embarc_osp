74 static uint8_t prv_get_value(lwm2m_tlv_t * tlvP,
75  security_instance_t * targetP)
76 {
77  // There are no multiple instance ressources
78  tlvP->type = LWM2M_TYPE_RESOURCE;
79 
80  switch (tlvP->id)
81  {
82  case LWM2M_SECURITY_URI_ID:
83  tlvP->value = (uint8_t*)targetP->uri;
84  tlvP->length = strlen(targetP->uri);
85  tlvP->flags = LWM2M_TLV_FLAG_STATIC_DATA;
86  tlvP->dataType = LWM2M_TYPE_STRING;
87  return COAP_205_CONTENT;
88 
89  case LWM2M_SECURITY_BOOTSTRAP_ID:
90  lwm2m_tlv_encode_bool(targetP->isBootstrap, tlvP);
91  if (0 != tlvP->length) return COAP_205_CONTENT;
92  else return COAP_500_INTERNAL_SERVER_ERROR;
93 
94  case LWM2M_SECURITY_SECURITY_ID:
95  lwm2m_tlv_encode_int(LWM2M_SECURITY_MODE_NONE, tlvP);
96  if (0 != tlvP->length) return COAP_205_CONTENT;
97  else return COAP_500_INTERNAL_SERVER_ERROR;
98 
99  case LWM2M_SECURITY_PUBLIC_KEY_ID:
100  // Here we return an opaque of 1 byte containing 0
101  tlvP->value = (uint8_t*)"";
102  tlvP->length = 1;
103  tlvP->flags = LWM2M_TLV_FLAG_STATIC_DATA;
104  tlvP->dataType = LWM2M_TYPE_OPAQUE;
105  return COAP_205_CONTENT;
106 
107  case LWM2M_SECURITY_SERVER_PUBLIC_KEY_ID:
108  // Here we return an opaque of 1 byte containing 0
109  tlvP->value = (uint8_t*)"";
110  tlvP->length = 1;
111  tlvP->flags = LWM2M_TLV_FLAG_STATIC_DATA;
112  tlvP->dataType = LWM2M_TYPE_OPAQUE;
113  return COAP_205_CONTENT;
114 
115  case LWM2M_SECURITY_SECRET_KEY_ID:
116  // Here we return an opaque of 1 byte containing 0
117  tlvP->value = (uint8_t*)"";
118  tlvP->length = 1;
119  tlvP->flags = LWM2M_TLV_FLAG_STATIC_DATA;
120  tlvP->dataType = LWM2M_TYPE_OPAQUE;
121  return COAP_205_CONTENT;
122 
123  case LWM2M_SECURITY_SMS_SECURITY_ID:
124  lwm2m_tlv_encode_int(LWM2M_SECURITY_MODE_NONE, tlvP);
125  if (0 != tlvP->length) return COAP_205_CONTENT;
126  else return COAP_500_INTERNAL_SERVER_ERROR;
127 
128  case LWM2M_SECURITY_SMS_KEY_PARAM_ID:
129  // Here we return an opaque of 6 bytes containing a buggy value
130  tlvP->value = (uint8_t*)"12345";
131  tlvP->length = 6;
132  tlvP->flags = LWM2M_TLV_FLAG_STATIC_DATA;
133  tlvP->dataType = LWM2M_TYPE_OPAQUE;
134  return COAP_205_CONTENT;
135 
136  case LWM2M_SECURITY_SMS_SECRET_KEY_ID:
137  // Here we return an opaque of 32 bytes containing a buggy value
138  tlvP->value = (uint8_t*)"1234567890abcdefghijklmnopqrstu";
139  tlvP->length = 32;
140  tlvP->flags = LWM2M_TLV_FLAG_STATIC_DATA;
141  tlvP->dataType = LWM2M_TYPE_OPAQUE;
142  return COAP_205_CONTENT;
143 
144  case LWM2M_SECURITY_SMS_SERVER_NUMBER_ID:
145  lwm2m_tlv_encode_int(0, tlvP);
146  if (0 != tlvP->length) return COAP_205_CONTENT;
147  else return COAP_500_INTERNAL_SERVER_ERROR;
148 
149  case LWM2M_SECURITY_SHORT_SERVER_ID:
150  lwm2m_tlv_encode_int(targetP->shortID, tlvP);
151  if (0 != tlvP->length) return COAP_205_CONTENT;
152  else return COAP_500_INTERNAL_SERVER_ERROR;
153 
154  case LWM2M_SECURITY_HOLD_OFF_ID:
155  lwm2m_tlv_encode_int(targetP->clientHoldOffTime, tlvP);
156  if (0 != tlvP->length) return COAP_205_CONTENT;
157  else return COAP_500_INTERNAL_SERVER_ERROR;
158 
159  default:
160  return COAP_404_NOT_FOUND;
161  }
162 }

164 static uint8_t prv_security_read(uint16_t instanceId,
165  int * numDataP,
166  lwm2m_tlv_t ** dataArrayP,
167  lwm2m_object_t * objectP)
168 {
169  security_instance_t * targetP;
170  uint8_t result;
171  int i;
172 
173  targetP = (security_instance_t *)lwm2m_list_find(objectP->instanceList, instanceId);
174  if (NULL == targetP) return COAP_404_NOT_FOUND;
175 
176  // is the server asking for the full instance ?
177  if (*numDataP == 0)
178  {
179  uint16_t resList[] = {LWM2M_SECURITY_URI_ID,
180  LWM2M_SECURITY_BOOTSTRAP_ID,
181  LWM2M_SECURITY_SECURITY_ID,
182  LWM2M_SECURITY_PUBLIC_KEY_ID,
183  LWM2M_SECURITY_SERVER_PUBLIC_KEY_ID,
184  LWM2M_SECURITY_SECRET_KEY_ID,
185  LWM2M_SECURITY_SMS_SECURITY_ID,
186  LWM2M_SECURITY_SMS_KEY_PARAM_ID,
187  LWM2M_SECURITY_SMS_SECRET_KEY_ID,
188  LWM2M_SECURITY_SMS_SERVER_NUMBER_ID,
189  LWM2M_SECURITY_SHORT_SERVER_ID,
190  LWM2M_SECURITY_HOLD_OFF_ID};
191  int nbRes = sizeof(resList)/sizeof(uint16_t);
192 
193  *dataArrayP = lwm2m_tlv_new(nbRes);
194  if (*dataArrayP == NULL) return COAP_500_INTERNAL_SERVER_ERROR;
195  *numDataP = nbRes;
196  for (i = 0 ; i < nbRes ; i++)
197  {
198  (*dataArrayP)[i].id = resList[i];
199  }
200  }
201 
202  i = 0;
203  do
204  {
205  result = prv_get_value((*dataArrayP) + i, targetP);
206  i++;
207  } while (i < *numDataP && result == COAP_205_CONTENT);
208 
209  return result;
210 }

214 static uint8_t prv_security_write(uint16_t instanceId,
215  int numData,
216  lwm2m_tlv_t * dataArray,
217  lwm2m_object_t * objectP)
218 {
219  security_instance_t * targetP;
220  int i;
221  uint8_t result = COAP_204_CHANGED;
222 
223  if ((dataArray->flags & LWM2M_TLV_FLAG_BOOTSTRAPPING) == 0) return COAP_401_UNAUTHORIZED;
224 
225  targetP = (security_instance_t *)lwm2m_list_find(objectP->instanceList, instanceId);
226  if (NULL == targetP)
227  {
228  return COAP_404_NOT_FOUND;
229  }
230 
231  i = 0;
232  do {
233  switch (dataArray[i].id)
234  {
235  case LWM2M_SECURITY_URI_ID:
236  if (targetP->uri != NULL) lwm2m_free(targetP->uri);
237  targetP->uri = (char *)lwm2m_malloc(dataArray[i].length + 1);
238  memset(targetP->uri, 0, dataArray[i].length + 1);
239  if (targetP->uri != NULL)
240  {
241  strncpy(targetP->uri, (char*)dataArray[i].value, dataArray[i].length);
242  result = COAP_204_CHANGED;
243  }
244  else
245  {
246  result = COAP_500_INTERNAL_SERVER_ERROR;
247  }
248  break;
249 
250  case LWM2M_SECURITY_BOOTSTRAP_ID:
251  if (1 == lwm2m_tlv_decode_bool(dataArray + i, &(targetP->isBootstrap)))
252  {
253  result = COAP_204_CHANGED;
254  }
255  else
256  {
257  result = COAP_400_BAD_REQUEST;
258  }
259  break;
260 
261  case LWM2M_SECURITY_SECURITY_ID:
262  // Let just ignore this
263  result = COAP_204_CHANGED;
264  break;
265 
266  case LWM2M_SECURITY_PUBLIC_KEY_ID:
267  // Let just ignore this
268  result = COAP_204_CHANGED;
269  break;
270 
271  case LWM2M_SECURITY_SERVER_PUBLIC_KEY_ID:
272  // Let just ignore this
273  result = COAP_204_CHANGED;
274  break;
275 
276  case LWM2M_SECURITY_SECRET_KEY_ID:
277  // Let just ignore this
278  result = COAP_204_CHANGED;
279  break;
280 
281  case LWM2M_SECURITY_SMS_SECURITY_ID:
282  // Let just ignore this
283  result = COAP_204_CHANGED;
284  break;
285 
286  case LWM2M_SECURITY_SMS_KEY_PARAM_ID:
287  // Let just ignore this
288  result = COAP_204_CHANGED;
289  break;
290 
291  case LWM2M_SECURITY_SMS_SECRET_KEY_ID:
292  // Let just ignore this
293  result = COAP_204_CHANGED;
294  break;
295 
296  case LWM2M_SECURITY_SMS_SERVER_NUMBER_ID:
297  // Let just ignore this
298  result = COAP_204_CHANGED;
299  break;
300 
301  case LWM2M_SECURITY_SHORT_SERVER_ID:
302  {
303  int64_t value;
304 
305  if (1 == lwm2m_tlv_decode_int(dataArray + i, &value))
306  {
307  if (value >= 0 && value <= 0xFFFF)
308  {
309  targetP->shortID = value;
310  result = COAP_204_CHANGED;
311  }
312  else
313  {
314  result = COAP_406_NOT_ACCEPTABLE;
315  }
316  }
317  else
318  {
319  result = COAP_400_BAD_REQUEST;
320  }
321  }
322  break;
323 
324  case LWM2M_SECURITY_HOLD_OFF_ID:
325  {
326  int64_t value;
327 
328  if (1 == lwm2m_tlv_decode_int(dataArray + i, &value))
329  {
330  if (value >= 0 && value <= 0xFFFF)
331  {
332  targetP->clientHoldOffTime = value;
333  result = COAP_204_CHANGED;
334  }
335  else
336  {
337  result = COAP_406_NOT_ACCEPTABLE;
338  }
339  }
340  else
341  {
342  result = COAP_400_BAD_REQUEST;
343  }
344  break;
345  }
346  default:
347  return COAP_404_NOT_FOUND;
348  }
349  i++;
350  } while (i < numData && result == COAP_204_CHANGED);
351 
352  return result;
353 }

373  int numData,

382  memset(targetP, 0, sizeof(security_instance_t));

384  targetP->instanceId = instanceId;
385  objectP->instanceList = LWM2M_LIST_ADD(objectP->instanceList, targetP);
386 
387  result = prv_security_write(instanceId, numData, dataArray, objectP);
388 
389  if (result != COAP_204_CHANGED)
390  {
391  (void)prv_security_delete(instanceId, objectP);
392  }
393  else
394  {
395  result = COAP_201_CREATED;
396  }
397 
398  return result;
399 }

416 void copy_security_object(lwm2m_object_t * objectDest, lwm2m_object_t * objectSrc)
417 {
418  memcpy(objectDest, objectSrc, sizeof(lwm2m_object_t));
419  objectDest->instanceList = NULL;
420  objectDest->userData = NULL;
421  security_instance_t * instanceSrc = (security_instance_t *)objectSrc->instanceList;
422  security_instance_t * previousInstanceDest = NULL;
423  while (instanceSrc != NULL)
424  {
425  security_instance_t * instanceDest = (security_instance_t *)lwm2m_malloc(sizeof(security_instance_t));
426  if (NULL == instanceDest)
427  {
428  return;
429  }
430  memcpy(instanceDest, instanceSrc, sizeof(security_instance_t));
431  instanceDest->uri = (char*)lwm2m_malloc(strlen(instanceSrc->uri) + 1);
432  strcpy(instanceDest->uri, instanceSrc->uri);
433  instanceSrc = (security_instance_t *)instanceSrc->next;
434  if (previousInstanceDest == NULL)
435  {
436  objectDest->instanceList = (lwm2m_list_t *)instanceDest;
437  }
438  else
439  {
440  previousInstanceDest->next = instanceDest;
441  }
442  previousInstanceDest = instanceDest;
443  }

462 lwm2m_object_t * get_security_object(int serverId, const char* serverUri, bool isBootstrap)
463 {
464  lwm2m_object_t * securityObj;
465 
466  securityObj = (lwm2m_object_t *)lwm2m_malloc(sizeof(lwm2m_object_t));
467 
468  if (NULL != securityObj)
469  {
470  security_instance_t * targetP;
471 
472  memset(securityObj, 0, sizeof(lwm2m_object_t));
473 
474  securityObj->objID = 0;
475 
476  // Manually create an hardcoded instance
477  targetP = (security_instance_t *)lwm2m_malloc(sizeof(security_instance_t));
478  if (NULL == targetP)
479  {
480  lwm2m_free(securityObj);
481  return NULL;
482  }
483 
484  memset(targetP, 0, sizeof(security_instance_t));
485  targetP->instanceId = 0;
486  targetP->uri = (char*)lwm2m_malloc(strlen(serverUri)+1);
487  strcpy(targetP->uri, serverUri);
488  targetP->isBootstrap = isBootstrap;
489  targetP->shortID = serverId;
490  targetP->clientHoldOffTime = 10;
491 
492  securityObj->instanceList = LWM2M_LIST_ADD(securityObj->instanceList, targetP);
493 
494  securityObj->readFunc = prv_security_read;
495 #ifdef LWM2M_BOOTSTRAP
496  securityObj->writeFunc = prv_security_write;
497  securityObj->createFunc = prv_security_create;
498  securityObj->deleteFunc = prv_security_delete;
499 #endif
500  securityObj->closeFunc = prv_security_close;
501  }
502 
503  return securityObj;
