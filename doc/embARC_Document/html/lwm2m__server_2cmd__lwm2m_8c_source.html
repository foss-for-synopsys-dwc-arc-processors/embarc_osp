64 void task_lwm2m_server(void *par)
65 {
66  unsigned int cpu_status;
67  while (1) {
68  lwm2m_server_conn_stat = 1;
69  if (lwm2mserver(&c_info) == 0) {
70  EMBARC_PRINTF("LwM2M server end successfully\r\n");
71  } else {
72  EMBARC_PRINTF("LwM2M server end failed\r\n");
73  }
74  cpu_status = cpu_lock_save();
75  if (c_info.serverPort != 0) c_info.serverPort = 0;
76  lwm2m_server_conn_stat = 0;
77  cpu_unlock_restore(cpu_status);
78  vTaskSuspend(NULL);
79  }
80 }

104 static int cmd_lwm2m_server(int argc, char **argv, void *extobj)
105 {
106  int ercd = E_OK;
107  int opt;
108  int opt_enter_cnt = 0;
109  int c_quit = 0;
110  long p_port = -1;
111 
112  NTSHELL_IO_PREDEF;
113 
114  VALID_EXTOBJ(extobj, -1);
115  NTSHELL_IO_GET(extobj);
116 
117  opterr = 0;
118  optind = 1;
119 
120  if (argc < 1) {
121  cmd_lwm2m_server_help(argv[0], extobj);
122  return E_OK;
123  }
124 
125  if (!lwip_pmwifi_isup()) {
126  CMD_DEBUG("Wifi is not ready for lwM2M server.\r\n");
127  goto error_exit;
128  }
129  while ((opt=getopt(argc, argv, "p:qhH?")) != -1) {
130  opt_enter_cnt ++;
131  switch (opt) {
132  case 'h':
133  case '?':
134  case 'H':
135  cmd_lwm2m_server_help(argv[0], extobj);
136  return E_OK;
137  break;
138  case 'q':
139  c_quit = 1;
140  break;
141  case 'p':
142  if (xatoi((char **)(&optarg), (long *)(&p_port)) != 1) {
143  CMD_DEBUG("Please check the port number is valid!\r\n");
144  return E_PAR;
145  }
146  break;
147  default:
148  opt_enter_cnt --;
149  CMD_DEBUG("unrecognized option:%c\r\n", opt);
150  break;
151  }
152  }
153  if (c_quit == 1) {
154  CMD_DEBUG("Try to exit existing server.\r\n");
155  handle_sigint(2);
156  goto error_exit;
157  }
158  if (lwm2m_server_conn_stat == 1) {
159  CMD_DEBUG("LwM2M server is already running.\r\n");
160  goto error_exit;
161  }
162 
163  c_info.serverPort = p_port;
164 
165  lwm2m_server_start_flag = 1;
166  CMD_DEBUG("Start LwM2M server.");
167 
168  if (task_lwm2m_server_handle == NULL) {
169  if (xTaskCreate(task_lwm2m_server, "lwm2m server", 20480, \
170  (void *)1, TSKPRI_LWM2M_SERVER, &task_lwm2m_server_handle) != pdPASS) {
171  EMBARC_PRINTF("create lwm2m server failed\r\n");
172  }
173  } else {
174  vTaskResume(task_lwm2m_server_handle);
175  }
176 
177  return E_OK;
178 
179 error_exit:
180  return ercd;
181 }

205 static lwm2m_handler get_op_handler(char *op_method, lwm2m_op_info *op_info)
206 {
207  if ((op_method == NULL) || (op_info == NULL)) {
208  return NULL;
209  } else {
210  if (LWM2M_CMP(op_method, LWM2M_OP_METHOD_LIST) == 0) {
211  return (lwm2m_handler)lwm2m_server_output_clients;
212  } else if (LWM2M_CMP(op_method, LWM2M_OP_METHOD_READ) == 0) {
213  if (op_info->clientId < 0) return NULL;
214  if (op_info->r_uri == NULL) return NULL;
215  return (lwm2m_handler)lwm2m_server_read_client;
216  } else if (LWM2M_CMP(op_method, LWM2M_OP_METHOD_WRITE) == 0) {
217  if (op_info->clientId < 0) return NULL;
218  if (op_info->r_uri == NULL) return NULL;
219  if (op_info->r_data == NULL) return NULL;
220  return (lwm2m_handler)lwm2m_server_write_client;
221  } else if (LWM2M_CMP(op_method, LWM2M_OP_METHOD_EXEC) == 0) {
222  if (op_info->clientId < 0) return NULL;
223  if (op_info->r_uri == NULL) return NULL;
224  return (lwm2m_handler)lwm2m_server_exec_client;
225  } else if (LWM2M_CMP(op_method, LWM2M_OP_METHOD_CREATE) == 0) {
226  if (op_info->clientId < 0) return NULL;
227  if (op_info->r_uri == NULL) return NULL;
228  if (op_info->r_data == NULL) return NULL;
229  return (lwm2m_handler)lwm2m_server_create_client;
230  } else if (LWM2M_CMP(op_method, LWM2M_OP_METHOD_DELETE) == 0) {
231  if (op_info->clientId < 0) return NULL;
232  if (op_info->r_uri == NULL) return NULL;
233  return (lwm2m_handler)lwm2m_server_delete_client;
234  } else if (LWM2M_CMP(op_method, LWM2M_OP_METHOD_OBSERVE) == 0) {
235  if (op_info->clientId < 0) return NULL;
236  if (op_info->r_uri == NULL) return NULL;
237  return (lwm2m_handler)lwm2m_server_observe_client;
238  } else if (LWM2M_CMP(op_method, LWM2M_OP_METHOD_CANCEL) == 0) {
239  if (op_info->clientId < 0) return NULL;
240  if (op_info->r_uri == NULL) return NULL;
241  return (lwm2m_handler)lwm2m_server_cancel_client;
242  } else {
243  return NULL;
244  }
245  }

269  }
270 
271  if (!lwip_pmwifi_isup()) {
272  CMD_DEBUG("Wifi is not ready for lwM2M server.\r\n");
273  goto error_exit;
274  }
275  if (lwm2m_server_conn_stat == 0) {
276  CMD_DEBUG("lwm2m server is not running, please run it first using command:lwm2m_server.\r\n");
277  goto error_exit;
278  }
279  while ((opt=getopt(argc, argv, "o:c:r:d:hH?")) != -1) {
280  opt_enter_cnt ++;
281  switch (opt) {
282  case 'h':
283  case '?':
284  case 'H':
285  cmd_lwm2m_op_help(argv[0], extobj);
286  return E_OK;
287  break;
288  case 'o':
289  op_method = optarg;
290  break;
291  case 'c':
292  if (xatoi((char **)(&optarg), (long *)(&op_info->clientId)) != 1) {
293  CMD_DEBUG("Please check the client ID is valid!\r\n");
294  return E_PAR;
295  }
296  break;
297  case 'r':
298  op_info->r_uri = optarg;
299  break;
300  case 'd':
301  op_info->r_data = optarg;
302  break;
303  default:
304  opt_enter_cnt --;
305  CMD_DEBUG("unrecognized option:%c\r\n", opt);
306  }
307  }
308 
309  op_handler = get_op_handler(op_method, op_info);
310  if (op_handler == NULL) {
311  CMD_DEBUG("Selected operation is not supported, or miss some other data.\r\n");
312  } else {
313  op_handler(op_info);
314  }
315 
316  return E_OK;
317 
318 error_exit:
