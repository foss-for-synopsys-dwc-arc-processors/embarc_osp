109 static int32_t spi_read_long_ctrl_reg(MRF24J40_DEF *mrf24j40, uint16_t addr, uint8_t *val)
110 {
111  int32_t ercd = E_OK;
112  uint8_t msg[2];
113  uint8_t ret_val = 0;
114  int32_t ret0, ret1, ret2;
115  DEV_SPI_TRANSFER dev_spi_xfer;
116  uint32_t cpu_status;
117  uint32_t cs_line = mrf24j40->spi_cs;
118 
119  DEV_SPI_PTR dev_spi_ptr = spi_get_dev(mrf24j40->spi);
120 
121  msg[0] = (uint8_t)(((addr >> 3) & 0x7F) | 0x80);
122  msg[1] = (uint8_t)((addr << 5) & 0xE0);
123 
124  DEV_SPI_XFER_SET_TXBUF(&dev_spi_xfer, msg, 0, 2);
125  DEV_SPI_XFER_SET_RXBUF(&dev_spi_xfer, &ret_val, 2, 1);
126  DEV_SPI_XFER_SET_NEXT(&dev_spi_xfer, NULL);
127 
128 #ifdef EMSK_PMRF_0_SPI_CPULOCK_ENABLE
129  cpu_status = cpu_lock_save();
130 #endif
131  ret0 = dev_spi_ptr->spi_control(SPI_CMD_MST_SEL_DEV, CONV2VOID(cs_line));
132  ret1 = dev_spi_ptr->spi_control(SPI_CMD_TRANSFER_POLLING, CONV2VOID(&dev_spi_xfer));
133  ret2 = dev_spi_ptr->spi_control(SPI_CMD_MST_DSEL_DEV, CONV2VOID(cs_line));
134 #ifdef EMSK_PMRF_0_SPI_CPULOCK_ENABLE
135  cpu_unlock_restore(cpu_status);
136 #endif
137 
138  if (!((ret0 == E_OK) && (ret1 == E_OK) && (ret2 == E_OK)))
139  {
140  ercd = E_SYS;
141  }
142  else
143  {
144  *val = ret_val;
145  }
146 
147 error_exit:
148  return ercd;
149 }

566 void mrf24j40_initialize(MRF24J40_DEF *mrf24j40)
567 {
568  uint8_t softrst_status;
569  uint8_t rf_state;
570  uint8_t rxmcr;
571 
572  //mrf24j40_cs_pin(1);
573  //mrf24j40_wake_pin(1);
574  mrf24j40_wake_pin(mrf24j40, 1);
575 
576  mrf24j40_reset(mrf24j40);
577  mrf24j40_write_short_ctrl_reg(mrf24j40, MRF24J40_SOFTRST, (MRF24J40_RSTPWR | MRF24J40_RSTBB | MRF24J40_RSTMAC));
578  mrf24j40_delay_us(500);
579  do
580  {
581  mrf24j40_read_short_ctrl_reg(mrf24j40, MRF24J40_SOFTRST, &softrst_status);
582  } while((softrst_status & (MRF24J40_RSTPWR | MRF24J40_RSTBB | MRF24J40_RSTMAC)) != 0);
583 
584  mrf24j40_write_short_ctrl_reg(mrf24j40, MRF24J40_PACON2, MRF24J40_FIFOEN | MRF24J40_TXONTS(0x18));
585  mrf24j40_write_short_ctrl_reg(mrf24j40, MRF24J40_TXSTBL, MRF24J40_RFSTBL(9) | MRF24J40_MSIFS(5));
586  mrf24j40_write_long_ctrl_reg(mrf24j40, MRF24J40_RFCON0, MRF24J40_CHANNEL(0) | MRF24J40_RFOPT(0x03));
587  mrf24j40_write_long_ctrl_reg(mrf24j40, MRF24J40_RFCON1, MRF24J40_VCOOPT(0x02)); // Initialize VCOOPT = 0x02
588  mrf24j40_write_long_ctrl_reg(mrf24j40, MRF24J40_RFCON2, MRF24J40_PLLEN);
589  mrf24j40_write_long_ctrl_reg(mrf24j40, MRF24J40_RFCON6, (MRF24J40_TXFIL | MRF24J40__20MRECVR));
590  mrf24j40_write_long_ctrl_reg(mrf24j40, MRF24J40_RFCON7, MRF24J40_SLPCLKSEL(0x02));
591  mrf24j40_write_long_ctrl_reg(mrf24j40, MRF24J40_RFCON8, MRF24J40_RFVCO);
592  mrf24j40_write_long_ctrl_reg(mrf24j40, MRF24J40_SLPCON1, MRF24J40_SLPCLKDIV(1) | MRF24J40_CLKOUTDIS);
593 
594  mrf24j40_write_short_ctrl_reg(mrf24j40, MRF24J40_RXFLUSH, (MRF24J40_WAKEPAD | MRF24J40_WAKEPOL));
595 
596  mrf24j40_read_short_ctrl_reg(mrf24j40, MRF24J40_RXMCR, &rxmcr);
597  mrf24j40_write_short_ctrl_reg(mrf24j40, MRF24J40_RXMCR, rxmcr & (~MRF24J40_NOACKRSP));
