141 static void ntshell_uart_nt_printf(NTSHELL_IO *nt_io, const char *fmt, va_list arp)
142 {
143  int32_t ercd = E_OK;
144  NTSHELL_IO_UART *nt_uart;
145  void (*pf)(unsigned char);
146 
147  NSIM_NTSHELL_UART_CHECK_EXP(nt_io!=NULL, E_OBJ);
148  NSIM_NTSHELL_UART_CHECK_EXP(nt_io->extra_info!=NULL, E_OBJ);
149 
150  nt_uart = (NTSHELL_IO_UART *)nt_io->extra_info;
151 
152  if (nt_uart->write_byte) {
153  pf = xfunc_out; /* Save current output device */
154  xfunc_out = nt_uart->write_byte; /* Switch output to specified device */
155  xvprintf(fmt, arp);
156  xfunc_out = pf; /* Restore output device */
157  }
158 
159 error_exit:
160  return ;
161 }
