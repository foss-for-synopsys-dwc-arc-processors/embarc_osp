71 /*
72  * Multiple instance objects can use userdata to store data that will be shared between the different instances.
73  * The lwm2m_object_t object structure - which represent every object of the liblwm2m as seen in the single instance
74  * object - contain a chained list called instanceList with the object specific structure prv_instance_t:
75  */
76 typedef struct _prv_instance_
77 {
78  /*
79  * The first two are mandatories and represent the pointer to the next instance and the ID of this one. The rest
80  * is the instance scope user data (uint8_t test in this case)
81  */
82  struct _prv_instance_ * next; // matches lwm2m_list_t::next
83  uint16_t shortID; // matches lwm2m_list_t::id
84  bool state;
85  uint32_t counter;
86 } prv_instance_t;

88 static uint8_t prv_read(uint16_t instanceId, int * numDataP, lwm2m_tlv_t ** dataArrayP,
89  lwm2m_object_t * objectP)
90 {
91  prv_instance_t * instance_ptr;
92  int i;
93 
94  instance_ptr = (prv_instance_t *)lwm2m_list_find(objectP->instanceList, instanceId);
95  if (NULL == instance_ptr) return COAP_404_NOT_FOUND;
96 
97  if (*numDataP == 0) {
98  *dataArrayP = lwm2m_tlv_new(2);
99  if (*dataArrayP == NULL) return COAP_500_INTERNAL_SERVER_ERROR;
100  *numDataP = 2;
101  (*dataArrayP)[0].id = LWM2M_STATE;
102  (*dataArrayP)[1].id = LWM2M_COUNTER;
103  }
104 
105  for (i = 0 ; i < *numDataP ; i++) {
106  switch ((*dataArrayP)[i].id) {
107  case LWM2M_STATE:
108  (*dataArrayP)[i].type = LWM2M_TYPE_RESOURCE;
109  if(instance_ptr->state == 0){
110  instance_ptr->state = (bool)button_read(1 << instanceId);
111  }
112  lwm2m_tlv_encode_bool(instance_ptr->state, *dataArrayP + i);
113  instance_ptr->state = 0;
114  break;
115 
116  case LWM2M_COUNTER:
117  (*dataArrayP)[i].type = LWM2M_TYPE_RESOURCE;
118  lwm2m_tlv_encode_int(instance_ptr->counter, *dataArrayP + i);
119  break;
120  default:
121  return COAP_404_NOT_FOUND;
122  }
123  if ((*dataArrayP)[i].length == 0) return COAP_500_INTERNAL_SERVER_ERROR;
124  }
125 
126  return COAP_205_CONTENT;
127 }

162 static void prv_button_int(uint8_t button_id, uint8_t enable)
163 {
164 /*install the button isr*/
165  DEV_GPIO_PTR port;
166  uint32_t mask = (1 << button_id);
167  DEV_GPIO_BIT_ISR bit_isr;
168  DEV_GPIO_INT_CFG int_cfg;
169 
170  port = gpio_get_dev(BUTTON_SENSOR_ID);
171  port->gpio_control(GPIO_CMD_SET_BIT_DIR_INPUT, (void *)(mask));
172  int_cfg.int_bit_mask = mask;
173  int_cfg.int_bit_type = GPIO_INT_BITS_EDGE_TRIG(mask);
174  int_cfg.int_bit_polarity = GPIO_INT_BITS_POL_FALL_EDGE(mask);
175  int_cfg.int_bit_debounce = GPIO_INT_BITS_ENA_DEBOUNCE(mask);
176  port->gpio_control(GPIO_CMD_SET_BIT_INT_CFG, (void *)(&int_cfg));
177  bit_isr.int_bit_ofs = button_id;
178 
179  switch (button_id) {
180  case 0:
181  bit_isr.int_bit_handler = button0_isr;
182  break;
183  case 1:
184  bit_isr.int_bit_handler = button1_isr;
185  break;
186  case 2:
187  bit_isr.int_bit_handler = button2_isr;
188  break;
189  default:
190  break;
191  }
192 
193  port->gpio_control(GPIO_CMD_SET_BIT_ISR, (void *)(&bit_isr));
194 
195  if (enable) {
196  port->gpio_control(GPIO_CMD_ENA_BIT_INT, (void *)mask);
197  }
198  else {
199  port->gpio_control(GPIO_CMD_DIS_BIT_INT, (void *)mask);
200  }
201 
202 }

204 lwm2m_object_t * get_object_button(uint32_t num)
205 {
206  lwm2m_object_t * obj;
207  obj = (lwm2m_object_t *)lwm2m_malloc(sizeof(lwm2m_object_t));
208 
209  if (NULL != obj) {
210  int i;
211  prv_instance_t * instance_ptr;
212 
213  memset(obj, 0, sizeof(lwm2m_object_t));
214 
215  obj->objID = OBJECT_BUTTON_ID;
216 
217  for (i=0 ; i < num ; i++) {
218  instance_ptr = (prv_instance_t *)lwm2m_malloc(sizeof(prv_instance_t));
219  if (NULL == instance_ptr) {
220  lwm2m_free(obj);
221  return NULL;
222  }
223 
224  memset(instance_ptr, 0, sizeof(prv_instance_t));
225  instance_ptr->shortID = LWM2M_EMSK_INSTANCE_ID+i;
226  instance_ptr->state = 0;
227  instance_ptr->counter = 0;
228  obj->instanceList = LWM2M_LIST_ADD(obj->instanceList, instance_ptr);
229  prv_button_int(i, 1);
230  }
231  /*
232  * From a single instance object, two more functions are available.
233  * - The first one (createFunc) create a new instance and filled it with the provided informations. If an ID is
234  * provided a check is done for verifying his disponibility, or a new one is generated.
235  * - The other one (deleteFunc) delete an instance by removing it from the instance list (and freeing the memory
236  * allocated to it)
237  */
238  obj->readFunc = prv_read;
239  obj->writeFunc = prv_write;
240  obj->createFunc = prv_create;
241  obj->deleteFunc = prv_delete;
242  obj->executeFunc = prv_exec;
243  obj->closeFunc = prv_close;
244  }
245  button_obj = obj;
246  return obj;
247 }
