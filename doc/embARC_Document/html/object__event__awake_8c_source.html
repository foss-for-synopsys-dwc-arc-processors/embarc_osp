94 /*
95  * Multiple instance objects can use userdata to store data that will be shared between the different instances.
96  * The lwm2m_object_t object structure - which represent every object of the liblwm2m as seen in the single instance
97  * object - contain a chained list called instanceList with the object specific structure prv_instance_t:
98  */
99 typedef struct _prv_instance_
100 {
101  /*
102  * The first two are mandatories and represent the pointer to the next instance and the ID of this one. The rest
103  * is the instance scope user data (uint8_t test in this case)
104  */
105  struct _prv_instance_ * next; /* matches lwm2m_list_t::next */
106  uint16_t shortID; /* matches lwm2m_list_t::id */
107  bool event_awake;
108 } prv_instance_t;

111 static uint8_t prv_get_value(lwm2m_tlv_t * tlvP,
112  prv_instance_t * targetP)
113 {
114  /* There are no multiple instance resources */
115  tlvP->type = LWM2M_TYPE_RESOURCE;
116 
117  switch (tlvP->id) {
118  case WARN_AWAKE_ID:
119  targetP->event_awake = data_report_wn.event_awake;
120  lwm2m_tlv_encode_bool(targetP->event_awake, tlvP);
121  if (0 != tlvP->length) return COAP_205_CONTENT;
122  else return COAP_500_INTERNAL_SERVER_ERROR;
123  break;
124 
125  default:
126  return COAP_404_NOT_FOUND;
127  }

132 static uint8_t prv_read(uint16_t instanceId,
133  int * numDataP,
134  lwm2m_tlv_t ** dataArrayP,
135  lwm2m_object_t * objectP)
136 {
137  prv_instance_t * targetP;
138  uint8_t result;
139  int i;
140 
141  targetP = (prv_instance_t *)lwm2m_list_find(objectP->instanceList, instanceId);
142  if (NULL == targetP) return COAP_404_NOT_FOUND;
143 
144  if (*numDataP == 0) {
145  uint16_t resList[] = {
146  WARN_AWAKE_ID
147  };
148  int nbRes = sizeof(resList)/sizeof(uint16_t);
149 
150  *dataArrayP = lwm2m_tlv_new(nbRes);
151  if (*dataArrayP == NULL) return COAP_500_INTERNAL_SERVER_ERROR;
152  *numDataP = nbRes;
153  for (i = 0 ; i < nbRes ; i++) {
154  (*dataArrayP)[i].id = resList[i];
155  }
156 
157  }
158 
159  i = 0;
160  do {
161  result = prv_get_value((*dataArrayP) + i, targetP);
162  i++;
163  } while (i < *numDataP && result == COAP_205_CONTENT);
164 
165  return result;
166 }

223 lwm2m_object_t * get_awakestatus_object(void)
224 {
225  lwm2m_object_t * awakestatusObj;
226 
227  awakestatusObj = (lwm2m_object_t *)lwm2m_malloc(sizeof(lwm2m_object_t));
228 
229  if (NULL != awakestatusObj) {
230  int i;
231 
232  prv_instance_t * targetP;
233 
234  memset(awakestatusObj, 0, sizeof(lwm2m_object_t));
235 
236  awakestatusObj->objID = LWM2M_AWAKE_STA_OBJECT_ID;
237  for (i=0 ; i < 1 ; i++) {
238  targetP = (prv_instance_t *)lwm2m_malloc(sizeof(prv_instance_t));
239  if (NULL == targetP) {
240  lwm2m_free(awakestatusObj);
241  return NULL;
242  }
243  memset(targetP, 0, sizeof(prv_instance_t));
244 
245  targetP->shortID = LWM2M_EMSK_INSTANCE_ID + i;
246  targetP->event_awake = data_report_wn.event_awake;
247  awakestatusObj->instanceList = LWM2M_LIST_ADD(awakestatusObj->instanceList, targetP);
248  }
249  /*
250  * From a single instance object, two more functions are available.
251  * - The first one (createFunc) create a new instance and filled it with the provided informations. If an ID is
252  * provided a check is done for verifying his disponibility, or a new one is generated.
253  * - The other one (deleteFunc) delete an instance by removing it from the instance list (and freeing the memory
254  * allocated to it)
255  */
256  awakestatusObj->readFunc = prv_read;
257  awakestatusObj->writeFunc = prv_write;
258  awakestatusObj->createFunc = prv_create;
259  awakestatusObj->deleteFunc = prv_delete;
260  awakestatusObj->executeFunc = prv_exec;
261  awakestatusObj->closeFunc = prv_close;
262  }
263 
264  return awakestatusObj;
265 }
