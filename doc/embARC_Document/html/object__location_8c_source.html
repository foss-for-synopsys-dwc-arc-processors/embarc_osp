86 static uint8_t prv_res2tlv(lwm2m_tlv_t* tlvP,
87  location_data_t* locDataP)
88 {
89  //-------------------------------------------------------------------- JH --
90  uint8_t ret = COAP_205_CONTENT;
91  switch (tlvP->id) // location resourceId
92  {
93  case RES_M_LATITUDE:
94  tlvP->value = (uint8_t*)locDataP->latitude;
95  tlvP->length = strlen(locDataP->latitude);
96  tlvP->flags = LWM2M_TLV_FLAG_STATIC_DATA;
97  tlvP->type = LWM2M_TYPE_RESOURCE;
98  tlvP->dataType = LWM2M_TYPE_STRING;
99  break;
100  case RES_M_LONGITUDE:
101  tlvP->value = (uint8_t*)locDataP->longitude;
102  tlvP->length = strlen(locDataP->latitude);
103  tlvP->flags = LWM2M_TLV_FLAG_STATIC_DATA;
104  tlvP->type = LWM2M_TYPE_RESOURCE;
105  tlvP->dataType = LWM2M_TYPE_STRING;
106  break;
107  case RES_O_ALTITUDE:
108  tlvP->value = (uint8_t*)locDataP->altitude;
109  tlvP->length = strlen(locDataP->altitude);
110  tlvP->flags = LWM2M_TLV_FLAG_STATIC_DATA;
111  tlvP->type = LWM2M_TYPE_RESOURCE;
112  tlvP->dataType = LWM2M_TYPE_STRING;
113  break;
114  case RES_O_UNCERTAINTY:
115  tlvP->value = (uint8_t*)locDataP->uncertainty;
116  tlvP->length = strlen(locDataP->uncertainty);
117  tlvP->flags = LWM2M_TLV_FLAG_STATIC_DATA;
118  tlvP->type = LWM2M_TYPE_RESOURCE;
119  tlvP->dataType = LWM2M_TYPE_STRING;
120  break;
121  case RES_O_VELOCITY:
122  tlvP->value = locDataP->velocity;
123  tlvP->length = VELOCITY_OCTETS;
124  tlvP->flags = LWM2M_TLV_FLAG_STATIC_DATA;
125  tlvP->type = LWM2M_TYPE_RESOURCE;
126  tlvP->dataType = LWM2M_TYPE_OPAQUE;
127  break;
128  case RES_M_TIMESTAMP:
129  lwm2m_tlv_encode_int(locDataP->timestamp, tlvP);
130  tlvP->type = LWM2M_TYPE_RESOURCE;
131  tlvP->dataType = LWM2M_TYPE_TIME;
132  break;
133  default:
134  ret = COAP_404_NOT_FOUND;
135  break;
136  }
137 
138  return ret;
139 }

153 static uint8_t prv_location_read(uint16_t objInstId,
154  int* numDataP,
155  lwm2m_tlv_t** tlvArrayP,
156  lwm2m_object_t* objectP)
157 {
158  //-------------------------------------------------------------------- JH --
159  int i;
160  uint8_t result = COAP_500_INTERNAL_SERVER_ERROR;
161  location_data_t* locDataP = (location_data_t*)(objectP->userData);
162 
163  // defined as single instance object!
164  if (objInstId != 0) return COAP_404_NOT_FOUND;
165 
166  if (*numDataP == 0) // full object, readable resources!
167  {
168  uint16_t readResIds[] = {
169  RES_M_LATITUDE,
170  RES_M_LONGITUDE,
171  RES_O_ALTITUDE,
172  RES_O_UNCERTAINTY,
173  RES_O_VELOCITY,
174  RES_M_TIMESTAMP
175  }; // readable resources!
176 
177  *numDataP = sizeof(readResIds)/sizeof(uint16_t);
178  *tlvArrayP = lwm2m_tlv_new(*numDataP);
179  if (*tlvArrayP == NULL) return COAP_500_INTERNAL_SERVER_ERROR;
180 
181  // init readable resource id's
182  for (i = 0 ; i < *numDataP ; i++)
183  {
184  (*tlvArrayP)[i].id = readResIds[i];
185  }
186  }
187 
188  for (i = 0 ; i < *numDataP ; i++)
189  {
190  result = prv_res2tlv ((*tlvArrayP)+i, locDataP);
191  if (result!=COAP_205_CONTENT) break;
192  }
193 
194  return result;
195 }

206 void location_setVelocity(lwm2m_object_t* locationObj,
207  uint16_t bearing,
208  uint16_t horizontalSpeed,
209  uint8_t speedUncertainty)
210 {
211  //-------------------------------------------------------------------- JH --
212  location_data_t* pData = locationObj->userData;
213  pData->velocity[0] = HORIZONTAL_VELOCITY_WITH_UNCERTAINTY << 4;
214  pData->velocity[0] = (bearing & 0x100) >> 8;
215  pData->velocity[1] = (bearing & 0x0FF);
216  pData->velocity[2] = horizontalSpeed >> 8;
217  pData->velocity[3] = horizontalSpeed & 0xff;
218  pData->velocity[4] = speedUncertainty;
219 }

286 lwm2m_object_t * get_object_location(void)
287 {
288  //-------------------------------------------------------------------- JH --
289  lwm2m_object_t * locationObj;
290 
291  locationObj = (lwm2m_object_t *)lwm2m_malloc(sizeof(lwm2m_object_t));
292  if (NULL != locationObj)
293  {
294  memset(locationObj, 0, sizeof(lwm2m_object_t));
295 
296  // It assigns its unique ID
297  // The 6 is the standard ID for the optional object "Location".
298  locationObj->objID = LWM2M_LOCATION_OBJECT_ID;
299 
300  // and its unique instance
301  locationObj->instanceList = (lwm2m_list_t *)lwm2m_malloc(sizeof(lwm2m_list_t));
302  if (NULL != locationObj->instanceList)
303  {
304  memset(locationObj->instanceList, 0, sizeof(lwm2m_list_t));
305  }
306  else
307  {
308  lwm2m_free(locationObj);
309  return NULL;
310  }
311 
312  // And the private function that will access the object.
313  // Those function will be called when a read query is made by the server.
314  // In fact the library don't need to know the resources of the object, only the server does.
315  //
316  locationObj->readFunc = prv_location_read;
317  locationObj->closeFunc = prv_location_close;
318  locationObj->userData = lwm2m_malloc(sizeof(location_data_t));
319 
320  // initialize private data structure containing the needed variables
321  if (NULL != locationObj->userData)
322  {
323  location_data_t* data = (location_data_t*)locationObj->userData;
324  strcpy (data->latitude, "27.986065"); // Mount Everest :)
325  strcpy (data->longitude, "86.922623");
326  strcpy (data->altitude, "8495.0000");
327  strcpy (data->uncertainty, "0.01");
328  location_setVelocity(locationObj, 0, 0, 255); // 255: speedUncertainty not supported!
329  data->timestamp = time(NULL);
330  }
331  else
332  {
333  lwm2m_free(locationObj);
334  locationObj = NULL;
335  }
336  }
337 
338  return locationObj;
339 }
