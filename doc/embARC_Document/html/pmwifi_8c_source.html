113 static int32_t pmwifi_0_gpio_open(uint32_t pin_mask, uint32_t pin_dir)
114 {
115  int32_t ercd = E_OK;
116  uint32_t mask;
117  DEV_GPIO *pmwifi_gpio_ptr;
118 
119  pmwifi_gpio_ptr = gpio_get_dev(EMSK_PMWIFI_0_GPIO_ID);
120  PMWIFI_CHECK_EXP(pmwifi_gpio_ptr != NULL, E_OBJ);
121 
122  pin_mask = ((pin_mask & WF_GPIO_PIN_MASK_ALL)<<EMSK_PMWIFI_0_PIN_GAP);
123  mask = (pin_dir<<EMSK_PMWIFI_0_PIN_GAP) & pin_mask;
124  ercd = pmwifi_gpio_ptr->gpio_open(mask);
125  if (ercd == E_OPNED) {
126  /* direction */
127  pmwifi_gpio_ptr->gpio_control(GPIO_CMD_SET_BIT_DIR_OUTPUT, (void *)mask);
128  mask = (~(pin_dir<<EMSK_PMWIFI_0_PIN_GAP)) & pin_mask;
129  pmwifi_gpio_ptr->gpio_control(GPIO_CMD_SET_BIT_DIR_INPUT, (void *)mask);
130  }
131  /* Default all set to pull */
132  ercd = pmwifi_gpio_ptr->gpio_control(GPIO_CMD_DIS_BIT_INT, (void *)pin_mask);
133 
134 error_exit:
135  return ercd;
136 }
137 static int32_t pmwifi_0_gpio_close(uint32_t pin_mask)
138 {
139  static uint32_t close_mask = 0;
140  int32_t ercd = E_OK;
141  DEV_GPIO *pmwifi_gpio_ptr;
142 
143  pmwifi_gpio_ptr = gpio_get_dev(EMSK_PMWIFI_0_GPIO_ID);
144  PMWIFI_CHECK_EXP(pmwifi_gpio_ptr != NULL, E_OBJ);
145 
146  close_mask |= pin_mask;
147  close_mask &= WF_GPIO_PIN_MASK_ALL;
148 
149  if (close_mask == WF_GPIO_PIN_MASK_ALL) { /* only close it when all pin need to be closed */
150  ercd = pmwifi_gpio_ptr->gpio_close();
151  close_mask = 0;
152  }
153 
154 error_exit:
155  return ercd;
156 }

220  ercd = pmwifi_gpio_ptr->gpio_control(GPIO_CMD_GET_BIT_MTHD, (void *)(&mthd));
