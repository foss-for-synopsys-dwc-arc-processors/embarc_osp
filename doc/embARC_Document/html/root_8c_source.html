119 static void multicast_send(void)
120 {
121  uint32_t id;
122 
123  id = uip_htonl(seq_id);
124  memset(buf, 0, MAX_PAYLOAD_LEN);
125  memcpy(buf, &id, sizeof(seq_id));
126 
127  EMBARC_PRINTF("Send to: ");
128  PRINT6ADDR(&mcast_conn->ripaddr);
129  EMBARC_PRINTF(" Remote Port %u,", uip_ntohs(mcast_conn->rport));
130  EMBARC_PRINTF(" (msg=0x%08lx)", (unsigned long)uip_ntohl(*((uint32_t *)buf)));
131  EMBARC_PRINTF(" %lu bytes\r\n", (unsigned long)sizeof(id));
132 
133  seq_id++;
134  uip_udp_packet_send(mcast_conn, buf, sizeof(id));
135 }

137 static void prepare_mcast(void)
138 {
139  uip_ipaddr_t ipaddr;
140 
141  /*
142  * IPHC will use stateless multicast compression for this destination
143  * (M=1, DAC=0), with 32 inline bits (1E 89 AB CD)
144  */
145  uip_ip6addr(&ipaddr, 0xFF1E,0,0,0,0,0,0x89,0xABCD);
146  mcast_conn = udp_new(&ipaddr, UIP_HTONS(MCAST_SINK_UDP_PORT), NULL);
147 }

149 static void set_own_addresses(void)
150 {
151  int i;
152  uint8_t state;
153  rpl_dag_t *dag;
154  uip_ipaddr_t ipaddr;
155 
156  uip_ip6addr(&ipaddr, 0xaaaa, 0, 0, 0, 0, 0, 0, 0);
157  uip_ds6_set_addr_iid(&ipaddr, &uip_lladdr);
158  uip_ds6_addr_add(&ipaddr, 0, ADDR_AUTOCONF);
159 
160  EMBARC_PRINTF("Our IPv6 addresses:\r\n");
161  for(i = 0; i < UIP_DS6_ADDR_NB; i++) {
162  state = uip_ds6_if.addr_list[i].state;
163  if(uip_ds6_if.addr_list[i].isused && (state == ADDR_TENTATIVE || state
164  == ADDR_PREFERRED)) {
165  EMBARC_PRINTF(" ");
166  PRINT6ADDR(&uip_ds6_if.addr_list[i].ipaddr);
167  EMBARC_PRINTF("\r\n");
168  if(state == ADDR_TENTATIVE) {
169  uip_ds6_if.addr_list[i].state = ADDR_PREFERRED;
170  }
171  }
172  }
173 
174  /* Become root of a new DODAG with ID our global v6 address */
175  dag = rpl_set_root(RPL_DEFAULT_INSTANCE, &ipaddr);
176  if(dag != NULL) {
177  rpl_set_prefix(dag, &ipaddr, 64);
178  EMBARC_PRINTF("Created a new RPL dag with ID: ");
179  PRINT6ADDR(&dag->dag_id);
180  EMBARC_PRINTF("\r\n");
181  }
182 }

184 PROCESS_THREAD(rpl_root_process, ev, data)
185 {
186  static struct etimer et;
187 
188  PROCESS_BEGIN();
189 
190  EMBARC_PRINTF("------IPv6 multcast demo (root node)------\r\n");
191  EMBARC_PRINTF("Multicast Engine: '%s'\r\n", UIP_MCAST6.name);
192 
193  init_net(EMSK_ID);
194  NETSTACK_MAC.off(1);
195 
196  set_own_addresses();
197 
198  prepare_mcast();
199 
200  etimer_set(&et, START_DELAY * CLOCK_SECOND);
201  while(1) {
202  PROCESS_YIELD();
203  if(etimer_expired(&et)) {
204  multicast_send();
205  etimer_set(&et, SEND_INTERVAL);
206  }
207  }
208 
209  PROCESS_END();
210 }
