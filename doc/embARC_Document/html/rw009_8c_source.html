291 {

305  rw009_gpio_ptr->gpio_control(GPIO_CMD_SET_BIT_DIR_OUTPUT, (void *)mask);
306  mask = RW009_INT_BUSY_PIN_MASK;
307  rw009_gpio_ptr->gpio_control(GPIO_CMD_SET_BIT_DIR_INPUT, (void *)mask);
308  }
309 
310  int_cfg.int_bit_mask = RW009_INT_BUSY_PIN_MASK;
311  int_cfg.int_bit_type = GPIO_INT_BITS_EDGE_TRIG(RW009_INT_BUSY_PIN_MASK);
312  int_cfg.int_bit_polarity = GPIO_INT_BITS_POL_FALL_EDGE(RW009_INT_BUSY_PIN_MASK);
313  int_cfg.int_bit_debounce = GPIO_INT_BITS_DIS_DEBOUNCE(RW009_INT_BUSY_PIN_MASK);

316  bit_isr.int_bit_handler = rw009_isr;

422 static int cmd_resp_handler(DEV_WNIC *rw009_wnic, struct rw009_resp* resp)
423 {
424  int32_t ercd = E_OK;
425  DEV_WNIC_INFO *rw009_info;
426  DEV_WNIC_ON_OPS *rw009_on_ops;
427 
428  RW009_CHECK_EXP(rw009_wnic != NULL, E_OBJ);
429  rw009_info = &(rw009_wnic->wnic_info);
430  rw009_on_ops = &(rw009_wnic->wnic_on_ops);
431 
432 
433  if (resp->cmd != last_cmd) {
434  DBG("current command is %d, cmd and response do not match!\n", resp->cmd);
435  }
436 
437  switch (resp->cmd) {
438  case RW009_CMD_INIT:
439  RW009_CHECK_EXP((rw009_info->init_status == WNIC_DURING_INITIALIZATION), E_PAR);
440 
441  DBG("response for rw009_cmd_init\n");
442  DBG("sn:%s\n", resp->resp.init.sn);
443  DBG("ver:%s\n", resp->resp.init.version);
444  memcpy(rw009_info->mac_addr, resp->resp.init.mac, WNIC_HDR_LEN);
445  rw009_info->init_status = WNIC_INIT_SUCCESSFUL;
446  init_try = 0;
447 
448  if (rw009_on_ops->on_init_success != NULL) {
449  rw009_on_ops->on_init_success(rw009_wnic);
450  }
451  break;
452  case RW009_CMD_SCAN:
453  RW009_CHECK_EXP((rw009_info->scan_status == WNIC_DURING_SCAN), E_PAR);
454  DBG("response for rw009_cmd_scan\n");
455  if (scan_cnt < RW009_SCAN_AP_NUM) {
456  memcpy(&scanned_ap[scan_cnt], &resp->resp.ap_info, sizeof(rw009_ap_info));
457  scan_cnt++;
458  }
459  break;
460  case RW009_CMD_JOIN:
461  if (rw009_info->conn_status != WNIC_CONNECTED) {
462  break;
463  }
464  case RW009_CMD_EASY_JOIN:
465  RW009_CHECK_EXP((rw009_info->init_status == WNIC_INIT_SUCCESSFUL), E_PAR);
466  RW009_CHECK_EXP((rw009_info->conn_status == WNIC_DURING_CONNECTION) || \
467  (rw009_info->conn_status == WNIC_CONNECTED), E_CLS);
468  rw009_info->busy_status = WNIC_IS_FREE;
469  DBG("response for rw009_cmd_join or easy_join(cmd:%d, resp:%d)\n", resp->cmd, resp->result);
470  if (resp->result == 0) {
471  /* connected */
472  DBG("rw009 connection success\n");
473  if (rw009_info->conn_status != WNIC_CONNECTED) {
474  rw009_info->conn_status = WNIC_CONNECTED;
475  if (rw009_on_ops->on_connected != NULL) {
476  rw009_on_ops->on_connected(rw009_wnic);
477  }
478  }
479  } else {
480  if (! (((resp->cmd == RW009_CMD_JOIN) && (resp->result == 12)) || \
481  (resp->cmd == RW009_CMD_EASY_JOIN)) ) {
482  break;
483  }
484  if ((rw009_info->conn_status == WNIC_CONNECTED) && \
485  (resp->cmd == RW009_CMD_EASY_JOIN)) {
486  break;
487  }
488  DBG("rw009 connection failure\n");
489  if (rw009_info->conn_status == WNIC_CONNECTED) {
490  rw009_info->conn_status = WNIC_NOT_CONNECTED;
491  if (rw009_on_ops->on_disconnected != NULL) {
492  rw009_on_ops->on_disconnected(rw009_wnic);
493  }
494  } else {
495  rw009_info->conn_status = WNIC_NOT_CONNECTED;
496  }
497  }
498  break;
499  case RW009_CMD_RSSI:
500  RW009_CHECK_EXP((rw009_info->init_status == WNIC_INIT_SUCCESSFUL), E_PAR);
501  DBG("response for rw009_cmd_rssi\n");
502  /* \todo use to poll the status of connection ? */
503  break;
504  default:
505  DBG("unknown cmd response\n");
506  ercd = E_PAR;
507  break;
508  }
509 
510 error_exit:
511  return ercd;
512 }

522 static int rw009_xfer(DEV_WNIC *rw009_wnic, struct rw009_spi_data_packet* out, struct rw009_spi_data_packet* in)
523 {
524  int32_t ercd = E_OK;
525  DEV_WNIC_INFO *rw009_info;
526  DEV_WNIC_ON_OPS *rw009_on_ops;
527 
528  struct rw009_spi_request req;
529  struct rw009_spi_response resp;
530  struct dev_spi_transfer xfer;
531  uint32_t to_send = 0;
532 
533 
534  RW009_CHECK_EXP(rw009_wnic != NULL, E_OBJ);
535  rw009_info = &(rw009_wnic->wnic_info);
536  rw009_on_ops = &(rw009_wnic->wnic_on_ops);
537 
538 
539  if (out == NULL || in == NULL) {
540  return -1;
541  }
542 
543  rw009_int_disable();
544 
545  while (rw009_is_busy()); /* wait if rw009 is busy */
546 
547  req.magic1 = RW009_REQ_MAGIC1;
548  req.magic2 = RW009_REQ_MAGIC2;
549  req.flag = RW009_REQ_FLAG_MRDY;
550  if (out->len) {
551  req.len = out->len + RW009_DATA_PACKET_HDR_LEN;
552  } else {
553  req.len = 0;
554  }
555 
556  /* send cmd request */
557  rw009_select();
558  rw009_spi_ptr->spi_write(&req, sizeof(req));
559  rw009_deselect();
560 
561  while (rw009_is_busy());
562 
563  rw009_select();
564 
565  /* receive cmd request response */
566  rw009_spi_ptr->spi_read(&resp, sizeof(resp));
567 
568  if ((resp.magic1 != RW009_RESP_MAGIC1) || (resp.magic2 != RW009_RESP_MAGIC2)) {
569  DBG("bad resp magic, abort!\n");
570  return E_SYS;
571  }
572 
573  if (resp.flag & RW009_RESP_FLAG_SRDY) {
574  //DBG("slave is ready to receive\n");
575  to_send = req.len;
576  }
577 
578  if (resp.len > RW009_SPI_MAX_DATA_LEN) {
579  resp.len = 0;
580  }
581 
582  DEV_SPI_XFER_SET_TXBUF(&xfer, out, 0, to_send);
583  DEV_SPI_XFER_SET_RXBUF(&xfer, in, 0, resp.len);
584  DEV_SPI_XFER_SET_NEXT(&xfer, NULL);
585 
586  rw009_spi_ptr->spi_control(SPI_CMD_TRANSFER_POLLING, &xfer);
587 
588  rw009_deselect();
589 
590  if (resp.len > 0) {
591  if (in->type == data_type_eth_data) {
592  //DBG("eth data:%d\n", resp.len);
593  /* ethernet data */
594  if (rw009_info->init_status == WNIC_INIT_SUCCESSFUL) {
595  if (rw009_on_ops->on_rxdata_comes != NULL) {
596  rw009_on_ops->on_rxdata_comes(rw009_wnic);
597  }
598  }
599  /* eth packet it recevied */
600  } else if (in->type == data_type_resp) {
601  /* cmd respone received */
602  cmd_resp_handler(rw009_wnic, (struct rw009_resp *)in->buffer);
603  } else {
604  DBG("unsupported data type\n");
605  ercd = E_SYS;
606  }
607  }
608  rw009_int_enable();
609  //DBG("master send %d bytes, rw009 send %d bytes\n", to_send, resp.len);
610 error_exit:
611  return ercd;
612 }

622 static int32_t rw009_cmd(DEV_WNIC *rw009_wnic, uint32_t cmd, void *args)
623 {
624  uint32_t cmd_len;
625  uint32_t cpu_status;
626  struct rw009_cmd *cmd_packet;
627  struct rw009_spi_data_packet* out;
628 
629  cmd_len = 0;
630  switch (cmd) {
631  case RW009_CMD_INIT:

634  break;
635  case RW009_CMD_SCAN:
636  cmd_len = 0;
