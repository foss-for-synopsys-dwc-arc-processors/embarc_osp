92 Inline int32_t spi_send_cmd(DEV_SPI_TRANSFER *xfer)
93 {
94  uint32_t cpu_status;
95  int32_t ercd = 0;
96 
97  cpu_status = cpu_lock_save();
98 
99  /* select device */
100  ercd = spi_flash->spi_control(SPI_CMD_MST_SEL_DEV, CONV2VOID(cs_flash));
101  SPIFLASH_CHECK_EXP(ercd == E_OK, -1);
102  ercd = spi_flash->spi_control(SPI_CMD_TRANSFER_POLLING, CONV2VOID(xfer));
103  /* deselect device */
104  spi_flash->spi_control(SPI_CMD_MST_DSEL_DEV, CONV2VOID(cs_flash));
105  SPIFLASH_CHECK_EXP(ercd == E_OK, -1);
106 
107  cpu_unlock_restore(cpu_status);
108  if (ercd == E_OK) ercd = 0;
109 
110 error_exit:
111  return ercd;
112 }

145 uint32_t flash_read_id(void)
146 {
147  uint32_t id = 0;
148  uint8_t local_buf[4];
149  DEV_SPI_TRANSFER cmd_xfer;
150 
151  local_buf[0] = RDID;
152 
153  DEV_SPI_XFER_SET_TXBUF(&cmd_xfer, local_buf, 0, 1);
154  DEV_SPI_XFER_SET_RXBUF(&cmd_xfer, local_buf, 1, 3);
155  DEV_SPI_XFER_SET_NEXT(&cmd_xfer, NULL);
156 
157  if (spi_send_cmd(&cmd_xfer) == 0) {
158  id = (local_buf[0] << 16) | (local_buf[1] << 8) | local_buf[2];
159  } else {
160  id = SPI_FLASH_NOT_VALID;
161  }
162 
163  return id;

242 int32_t flash_write_enable(void)
243 {
244  uint8_t local_buf[3];
245  DEV_SPI_TRANSFER cmd_xfer;
246 
247  uint32_t status = 0;
248  do {
249  local_buf[0] = WREN;
250 
251  DEV_SPI_XFER_SET_TXBUF(&cmd_xfer, local_buf, 0, 1);
252  DEV_SPI_XFER_SET_RXBUF(&cmd_xfer, NULL, 0, 0);
253  DEV_SPI_XFER_SET_NEXT(&cmd_xfer, NULL);
254 
255  if (spi_send_cmd(&cmd_xfer) != 0) {
256  return -1;
257  }
258  status = flash_read_status();
259  if (status == SPI_FLASH_NOT_VALID) {
260  return -1;
261  }
262  // clear protection bits
263  // Write Protect. and Write Enable.
264  if( (status & 0xfc) && (status & 0x02) ) {
265  local_buf[0] = WRSR; // write status
266  local_buf[1] = 0x00; // write status
267  local_buf[2] = 0x00; // write status
268 
269  DEV_SPI_XFER_SET_TXBUF(&cmd_xfer, local_buf, 0, 3);
270  DEV_SPI_XFER_SET_RXBUF(&cmd_xfer, NULL, 0, 0);
271  DEV_SPI_XFER_SET_NEXT(&cmd_xfer, NULL);
272 
273  if (spi_send_cmd(&cmd_xfer) != 0) {
274  return -1;
275  }
276  status = 0;
277  }
278  } while ( status != 0x02);
279 
280  return 0;
281 }

293 int32_t flash_erase(uint32_t address, uint32_t size)
294 {
295  uint32_t last_address;
296  uint32_t count = 0;
297  uint8_t local_buf[4];
298  DEV_SPI_TRANSFER cmd_xfer;
299 
300  // start address of last sector
301  last_address = (address + size) & (~(FLASH_SECTOR_SIZE - 1));
302 
303  // start address of first sector
304  address &= ~(FLASH_SECTOR_SIZE - 1);
305 
306  do {
307  if (flash_write_enable() != 0) {
308  return -1;
309  }
310 
311  if (flash_wait_ready() != 0) {
312  return -1;
313  }
314 
315  local_buf[0] = SE;
316  local_buf[1] = (address >> 16) & 0xff;
317  local_buf[2] = (address >> 8) & 0xff;
318  local_buf[3] = address & 0xff;
319 
320  DEV_SPI_XFER_SET_TXBUF(&cmd_xfer, local_buf, 0, 4);
321  DEV_SPI_XFER_SET_RXBUF(&cmd_xfer, NULL, 0, 0);
322  DEV_SPI_XFER_SET_NEXT(&cmd_xfer, NULL);
323 
324  if (spi_send_cmd(&cmd_xfer) != 0) {
325  return -1;
326  }
327 
328  address += FLASH_SECTOR_SIZE;
329  count++;
330  } while(address <= last_address);
331  if (flash_wait_ready() != 0) {
332  return -1;
333  }
334 
335  return (int32_t)count;
336 }

348 int32_t flash_write(uint32_t address, uint32_t size, const void *data)
349 {
350 
351  uint8_t local_buf[4];
352  DEV_SPI_TRANSFER cmd_xfer;
353  DEV_SPI_TRANSFER data_xfer;
354 
355  uint32_t first = 0;
356  uint32_t size_orig = size;
357 
358  if (flash_wait_ready() != 0) {
359  return -1;
360  }
361 
362  first = FLASH_PAGE_SIZE - (address & (FLASH_PAGE_SIZE - 1));
363 
364  do {
365  // send write enable command to flash
366  if (flash_write_enable() != 0) {
367  return -1;
368  }
369  if (flash_wait_ready() != 0) {
370  return -1;
371  }
372 
373  first = first < size ? first : size;
374 
375  local_buf[0] = PP;
376  local_buf[1] = (address >> 16) & 0xff;
377  local_buf[2] = (address >> 8) & 0xff;
378  local_buf[3] = address & 0xff;
379 
380  DEV_SPI_XFER_SET_TXBUF(&data_xfer, data, 0, first);
381  DEV_SPI_XFER_SET_RXBUF(&data_xfer, NULL, 0, 0);
382  DEV_SPI_XFER_SET_NEXT(&data_xfer, NULL);
383 
384  DEV_SPI_XFER_SET_TXBUF(&cmd_xfer, local_buf, 0, 4);
385  DEV_SPI_XFER_SET_RXBUF(&cmd_xfer, NULL, 0, 0);
386  DEV_SPI_XFER_SET_NEXT(&cmd_xfer, &data_xfer);
387 
388  if (spi_send_cmd(&cmd_xfer) != 0) {
389  return -1;
390  }
391 
392  size -= first;
393  address += first;
394  data += first;
395  first = FLASH_PAGE_SIZE;
396 
397  } while (size);
398 
399  if (flash_wait_ready() != 0) {
400  return -1;
401  }
402 
403  return (int32_t)(size_orig);
404 }
