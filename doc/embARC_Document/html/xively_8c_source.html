231 PROCESS_THREAD(dhcpc_process, ev, data)
232 {
233  DEV_WNIC_PTR wifi;
234 
235  PROCESS_BEGIN();
236 
237  EMBARC_PRINTF("\r\nNow trying to connect to WIFI Hotspot, Please wait about 30s!\r\n");
238  /* wait wifi connected */
239  while (1) {
240  PROCESS_WAIT_EVENT();
241  if (ev == wnic_event) {
242  wifi = (DEV_WNIC_PTR)data;
243  if (wifi->wnic_info.conn_status == WNIC_CONNECTED) {
244  break;
245  }
246  }
247  }
248  EMBARC_PRINTF("\r\nNow start get ip address using DHCP, Please wait about 30s!\r\n");
249  dhcpc_init(uip_lladdr.addr, sizeof(uip_lladdr.addr));
250  dhcpc_request();
251  while(1) {
252  PROCESS_WAIT_EVENT();
253  if (ev == PROCESS_EVENT_EXIT) {
254  process_exit(&dhcpc_process);
255  }
256  dhcpc_appcall(ev, data);
257  }
258  PROCESS_END();
259 }
